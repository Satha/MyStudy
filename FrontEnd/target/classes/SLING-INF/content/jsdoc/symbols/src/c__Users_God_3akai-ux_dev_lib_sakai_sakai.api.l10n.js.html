 44 sakai.api.l10n.getUserDefaultLocale = function() {
 45     var ret = sakai.config.defaultLanguage;
 46     // Get the browser language preference - IE uses userLanguage, all other browsers user language
 47     var locale = navigator.language ? navigator.language : navigator.userLanguage;
 48     if (locale) {
 49         var split = locale.split("-");
 50         if (split.length > 1) {
 51             split[1] = split[1].toUpperCase();
 52             var langs = sakai.config.Languages;
 53             // Loop over all the available languages - if the user's browser language preference matches
 54             // then set their locale to that so they don't have to set it manually
 55             for (var i=0,j=langs.length; i<j; i++) {
 56                 if (langs[i].language === split[0] && langs[i].country === split[1]) {
 57                     ret = split[0] + "_" + split[1];
 58                     break;
 59                 }
 60             }
 61         }
 62     }
 63     return ret;
 64 };

 66 /**
 67  * Parse a date string into a date object and adjust that date to the timezone
 68  * set by the current user.
 69  * @param {Object} dateString    date to parse in the format 2010-10-06T14:45:54+01:00
 70  */
 71 sakai.api.l10n.parseDateString = function(dateString){
 72     var d = new Date();
 73     d.setFullYear(parseInt(dateString.substring(0,4),10));
 74     d.setMonth(parseInt(dateString.substring(5,7),10) - 1);
 75     d.setDate(parseInt(dateString.substring(8,10),10));
 76     d.setHours(parseInt(dateString.substring(11,13),10));
 77     d.setMinutes(parseInt(dateString.substring(14,16),10));
 78     d.setSeconds(parseInt(dateString.substring(17,19),10));
 79     // Localization
 80     d.setTime(d.getTime() - (parseInt(dateString.substring(19,22),10)*60*60*1000));
 81     d.setTime(d.getTime() + sakai.data.me.user.locale.timezone.GMT*60*60*1000);
 82     return d;
 83 };

151 /**
152  * Function that will take in a date object and will transform that
153  * date object into a GMT date object. This should always be done before
154  * we try to save a date back to a file or the database. The timezone
155  * we are currently in will be determined from our timezone set in the
156  * personal preferences page
157  * @param {Date} date
158  *  JavaScript date object that we would like to transform in a
159  *  GMT date object
160  * @return {Date}
161  *  Date object, that will have transformed the given date and time into
162  *  GMT date and time
163  */
164 sakai.api.l10ntoGMT = function(date){
165     date.setHours(date.getHours() - sakai.data.me.locale.timezone.GMT);
166     return date;
167 };

169 /**
170  * Function that will take in a GMT date object and will transform that
171  * date object into a local date object. This should always be done after
172  * we load a date back from a file or the database. The timezone
173  * we are currently in will be determined from our timezone set in the
174  * personal preferences page
175  * @param {Date} date
176  *  JavaScript GMT date object that we would like to transform to a local date object
177  * @return {Date}
178  *  Date object, that will have transformed the given GMT date and time into
179  *  a local date and time
180  */
181 sakai.api.l10n.fromGMT = function(date){

184 };

199 sakai.api.l10n.getDateFormatString = function() {
200     var pattern = Globalization.culture.calendar.patterns.d;
201     var split = pattern.split("/");
202     for (var i=0, j=split.length; i<j; i++) {
203         if (split[i] === "m" || split[i] === "M") {
204             split[i] = "mm";
205         } else if (split[i] === "d" || split[i] === "D") {
206             split[i] = "dd";
207         }
208     }
209     return split.join("/").toUpperCase();
210 };
