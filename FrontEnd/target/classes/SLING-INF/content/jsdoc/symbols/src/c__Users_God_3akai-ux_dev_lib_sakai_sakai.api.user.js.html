 37 /**
 38  * @param {Object} extraOptions can include recaptcha: {challenge, response}, locale : "user_LOCALE", template: "templateName"
 39  */
 40 sakai.api.User.createUser = function(username, firstName, lastName, email, password, passwordConfirm, extraOptions, callback) {
 41     var profileData = {}; profileData.basic = {}; profileData.basic.elements = {};
 42     profileData.basic.elements["firstName"] = {};
 43     profileData.basic.elements["firstName"].value = firstName;
 44     profileData.basic.elements["lastName"] = {};
 45     profileData.basic.elements["lastName"].value = lastName;
 46     profileData.basic.elements["email"] = {};
 47     profileData.basic.elements["email"].value = email;
 48     profileData.basic.access = "everybody";
 49     var user = {
 50         "_charset_": "utf-8",
 51         "locale": sakai.api.l10n.getUserDefaultLocale(),
 52         "pwd": password,
 53         "pwdConfirm": passwordConfirm,
 54         ":name": username,
 55         ":sakai:pages-template": "/var/templates/site/" + sakai.config.defaultUserTemplate,
 56         ":sakai:profile-import": $.toJSON(profileData)
 57     };
 58     for (var i in extraOptions) {
 59         if (extraOptions.hasOwnProperty(i)) {
 60             switch(i) {
 61                 case "recaptcha":
 62                     user[":create-auth"] = "reCAPTCHA.net";
 63                     user[":recaptcha-challenge"] = extraOptions[i].challenge;
 64                     user[":recaptcha-response"] = extraOptions[i].response;
 65                     break;
 66                 case "locale":
 67                     user["locale"] = extraOptions[i];
 68                     break;
 69                 case "template":
 70                     user["template"] = "/var/templates/site/" + extraOptions[i];
 71                     break;
 72             }
 73         }
 74     }
 75     // Send an Ajax POST request to create a user
 76     $.ajax({
 77         url: sakai.config.URL.CREATE_USER_SERVICE,
 78         type: "POST",
 79         data: user,
 80         success: function(data){
 81 
 82             // Call callback function if set
 83             if ($.isFunction(callback)) {
 84                 callback(true, data);
 85             }
 86 
 87         },
 88         error: function(xhr, textStatus, thrownError){
 89 
 90             // Call callback function if set
 91             if ($.isFunction(callback)) {
 92                 callback(false, xhr);
 93             }
 94 
 95         }
 96     });
 97 };

143 /**
144  * Log-in to Sakai3
145  *
146  * @example
147  * sakai.api.user.login({
148  *     "username": "user1",
149  *     "password": "test"
150  * });
151  *
152  * @param {Object} credentials JSON object container the log-in information. Contains the username and password.
153  * @param {Function} [callback] Callback function that is called after sending the log-in request to the server.
154  */
155 sakai.api.User.login = function(credentials, callback) {
156 
157     // Argument check
158     if (!credentials || !credentials.username || !credentials.password) {
159         debug.info("sakai.api.user.login: Not enough or invalid arguments!");
160         callback(false, null);
161         return;
162     }
163 
164     /*
165      * sakaiauth:un : the username for the user
166      * sakaiauth:pw : the password for the user
167      * sakaiauth:login : set to 1 because we want to perform a login action
168      */
169     var data = {
170         "sakaiauth:login": 1,
171         "sakaiauth:un": credentials.username,
172         "sakaiauth:pw": credentials.password,
173         "_charset_": "utf-8"
174     };
175 
176     // Send the Ajax request
177     $.ajax({
178         url : sakai.config.URL.LOGIN_SERVICE,
179         type : "POST",
180         success: function(data){
181 
182             // Call callback function if set
183             if ($.isFunction(callback)) {
184                 callback(true, data);
185             }
186 
187         },
188         error: function(xhr, textStatus, thrownError){
189 
190             // Call callback function if set
191             if ($.isFunction(callback)) {
192                 callback(false, xhr);
193             }
194 
195         },
196         data : data
197     });
198 
199 };

202 /**
203  * Log-out from Sakai3
204  *
205  * @example sakai.api.user.logout();
206  * @param {Function} [callback] Callback function that is called after sending the log-in request to the server.
207  */
208 sakai.api.User.logout = function(callback) {
209 
210     /*
211      * POST request to the logout service,
212      * which will destroy the session.
213      */
214     $.ajax({
215         url: sakai.config.URL.PRESENCE_SERVICE,
216         type: "POST",
217         data: {
218             "sakai:status": "offline",
219             "_charset_": "utf-8"
220         },
221         complete: function(xhr, textStatus) {
222             // hit the logout service to destroy the session
223             $.ajax({
224                 url: sakai.config.URL.LOGOUT_SERVICE,
225                 type: "GET",
226                 complete: function(xhrInner, textStatusInner) {
227                     callback(textStatusInner === "success");
228                 }
229             });
230         }
231     });
232 
233 };

236 /**
237  * Retrieves all available information about a logged in user and stores it under sakai.data.me object. When ready it will call a specified callback function
238  *
239  * @param {Function} [callback] A function which will be called when the information is retrieved from the server.
240  * The first argument of the callback is a boolean whether it was successful or not, the second argument will contain the retrieved data or the xhr object
241  * @return {Void}
242  */
243 sakai.api.User.loadMeData = function(callback) {
244 
245     // Get the service url from the config file
246     var data_url = sakai.config.URL.ME_SERVICE;
247 
248     // Start a request to the service
249     $.ajax({
250         url: data_url,
251         cache: false,
252         success: function(data){
253 
254             sakai.data.me = sakai.api.Server.loadJSON.convertObjectToArray(data, null, null);
255 
256             // Check for firstName and lastName property - if not present use "rep:userId" for both (admin for example)
257             if (sakai.api.User.getProfileBasicElementValue(sakai.data.me.profile, "firstName") === "") {
258                 sakai.api.User.setProfileBasicElementValue(sakai.data.me.profile, "firstName", sakai.data.me.profile["rep:userId"]);
259             }
260             if (sakai.api.User.getProfileBasicElementValue(sakai.data.me.profile, "lastName") === "") {
261                 sakai.api.User.setProfileBasicElementValue(sakai.data.me.profile, "lastName", sakai.data.me.profile["rep:userId"]);
262             }
263 
264             // Call callback function if set
265             if ($.isFunction(callback)) {
266                 callback(true, sakai.data.me);
267             }
268         },
269         error: function(xhr, textStatus, thrownError) {
270 
271             // Log error
272             debug.error("sakai.api.User.loadMeData: Could not load logged in user data from the me service!");
273 
274             if (xhr.status === 500 && window.location.pathname !== "/dev/500.html"){
275                 document.location = "/dev/500.html";
276             }
277 
278             // Call callback function if set
279             if ($.isFunction(callback)) {
280                 callback(false, xhr);
281             }
282         }
283     });
284 };

288 /**
289  * Retrieves the display name to use for the user from config
290  * and parses it from the profile elements
291  *
292  * @param {Object} profile the user's profile (sakai.data.me.profile for the current user)
293  * @return {String} the name to show for a user
294  */
295 sakai.api.User.getDisplayName = function(profile) {
296     var configDisplayName = [sakai.config.Profile.userNameDisplay, sakai.config.Profile.userNameDefaultDisplay];
297     var nameToReturn = "";
298     var done = false;
299     var idx = 0;
300 
301     var parseName = function(i,key) {
302         if (profile &&
303             profile.basic &&
304             profile.basic.elements &&
305             profile.basic.elements[key] !== undefined &&
306             profile.basic.elements[key].value !== undefined) {
307            nameToReturn += profile.basic.elements[key].value + " ";
308            done = true;
309        }
310     };
311 
312     // iterate over the configDisplayName object until a valid non-empty display name is found
313     while (!done && idx < 2) {
314         if (configDisplayName[idx] !== undefined && configDisplayName[idx] !== "") {
315             var configEltsArray = configDisplayName[idx].split(" ");
316             $(configEltsArray).each(parseName);
317         }
318         idx++;
319     }
320 
321     return unescape(sakai.api.Security.saneHTML($.trim(nameToReturn)));
322 };

324 /**
325  * Safely retrieves an element value from the user's profile
326  *
327  * @param {Object} profile the user's profile (sakai.data.me.profile for the current user)
328  * @param {String} eltName the element name to retrieve the value for
329  * @return {String} the value of the element name provided
330  */
331 sakai.api.User.getProfileBasicElementValue = function(profile, eltName) {
332     var ret = "";
333     if (profile !== undefined &&
334         profile.basic !== undefined &&
335         profile.basic.elements !== undefined &&
336         profile.basic.elements[eltName] !== undefined &&
337         profile.basic.elements[eltName].value !== undefined) {
338             ret = profile.basic.elements[eltName].value;
339         }
340     return unescape(sakai.api.Security.saneHTML($.trim(ret)));
341 };

360 /**
361  * Get a user's short description from their profile
362  * This is based off of the configuration in config.js
363  * Example: "${role} in ${department}" could translate to "Undergraduate Student in Computer Science"
364  *           based on the configuration in config.js and the user's profile information
365  * If the user doesn't have the profile information requested by config.js, the function
366  * will remove the token from the string and any modifiers before the token after the previous token
367  * In the above example, if the user only had a department, the resulting string would be "Computer Science"
368  *
369  * @param {Object} profile The user's profile to get a description from
370  * @return {String} the user's short description
371  */
372 sakai.api.User.getShortDescription = function(profile) {
373     var shortDesc = sakai.config.Profile.shortDescription || "";
374     var tokenRegex = /\$\{[A-Za-z]+\}/gi;
375     var tokens = shortDesc.match(tokenRegex);
376     var lastReplacementValue = "";
377     $(tokens).each(function(i, val) {
378         var profileNode = val.match(/[A-Za-z]+/gi)[0];
379         if (profile.basic.elements[profileNode] && $.trim(profile.basic.elements[profileNode].value) !== "") {
380             /*if (lastReplacementValue === "" && tokens[i-1]) {
381                 // replace everything before this and after the last token
382             } */
383             if (sakai.config.Profile.configuration.defaultConfig.basic.elements[profileNode].type === "select") {
384                 lastReplacementValue = profile.basic.elements[profileNode].value;
385                 lastReplacementValue = sakai.config.Profile.configuration.defaultConfig.basic.elements[profileNode].select_elements[lastReplacementValue];
386                 lastReplacementValue = sakai.api.i18n.General.process(lastReplacementValue, sakai.data.i18n.localBundle, sakai.data.i18n.defaultBundle);
387             } else {
388                 lastReplacementValue = profile.basic.elements[profileNode].value;
389             }
390 
391             shortDesc = shortDesc.replace(val, lastReplacementValue);
392         } else {
393             if (tokens[i-1]) { // this is not the first time through
394                 var indexToStart = 0;
395                 // if the previous token's replaced value exists
396                 if (lastReplacementValue !== "" && shortDesc.indexOf(shortDesc.indexOf(lastReplacementValue)) !== -1) {
397                     // the index to start replacing at is the end of the last replacement
398                     indexToStart = shortDesc.indexOf(shortDesc.indexOf(lastReplacementValue)) + lastReplacementValue.length;
399                 }
400                 var indexToEnd = shortDesc.indexOf(val) + val.length;
401                 shortDesc = $.trim(shortDesc.replace(shortDesc.substring(indexToStart, indexToEnd), ""));
402             } else {
403                 shortDesc = $.trim(shortDesc.replace(val, ""));
404             }
405         }
406     });
407     return $.trim(shortDesc);
408 };

410 sakai.api.User.getContacts = function(callback) {
411     if (sakai.data.me.mycontacts) {
412         if ($.isFunction(callback)) {
413             callback();
414         }
415     } else {
416         // has to be synchronous
417         $.ajax({
418             url: sakai.config.URL.SEARCH_USERS_ACCEPTED,
419             data: {"q": "*"},
420             async: false,
421             success: function(data) {
422                 sakai.data.me.mycontacts = data.results;
423                 if ($.isFunction(callback)) {
424                     callback();
425                 }
426             }
427         });
428     }
429 };

431 sakai.api.User.checkIfConnected = function(userid) {
432     var ret = false;
433     sakai.api.User.getContacts(function() {
434         for (var i in sakai.data.me.mycontacts) {
435             if (i && sakai.data.me.mycontacts.hasOwnProperty(i)) {
436                 if (sakai.data.me.mycontacts[i].user === userid) {
437                     ret = true;
438                 }
439             }
440         }
441     });
442     return ret;
