 34 /**
 35  * @class Container
 36  *
 37  * @description
 38  * This will expose 2 funcions that can be called by widgets to inform
 39  * the container that the widget has finished doing things in its settings
 40  * mode. The container can then do whatever it needs to do according to the
 41  * context it's in (f.e.: if in the personal dashboard environment, the container
 42  * will want to render the view mode of that widget, in a site page edit context
 43  * the container will want to insert the widget into the WYSIWYG editor).
 44  *
 45  * This will also allow the container to register 2 functions related to widget
 46  * settings mode. First of all, the container can register a finish function,
 47  * which will be executed when a widget notifies the container that it has
 48  * successfully finished its settings mode. It can also register a cancel
 49  * function, which will be executed when a widget notifies the container that
 50  * its settings mode has been cancelled.
 51  *
 52  * @namespace
 53  * Widget container functions
 54  *
 55  */
 56

126     performLoad : function() {
127         for (var i = 0, il = sakai.api.Widgets.Container.toLoad.length; i<il; i++){
128             var fct = window["sakai"][sakai.api.Widgets.Container.toLoad[i]];
129             if ($.isFunction(fct)) {
130                 fct();
131             } else {
132                 debug.error("sakai magic - sakai.api.Widgets.Container.performLoad - The function couldn't execute correctly: '" + fct + "'");
133             }
134         }
135         sakai.api.Widgets.Container.toLoad = [];
136     },

139         sakai.api.Widgets.Container.readyToLoad = set;

214     /**
215      * Load the actual widgets
216      * @param {String} id The id of the widget
217      * @param {Boolean} showSettings
218      *  true  : render the settings mode of the widget
219      *  false : render the view mode of the widget
220      * @param {String} context The context of the widget (e.g. siteid)

222     loadWidgets : function(id, showSettings, context){
223         // Configuration variables
224         var widgetNameSpace = "sakai";
225         var widgetSelector = ".widget_inline";

228         var widgets = {}, settings = false;

230         /**
231          * Inform the widget that is is loaded and execute the main JavaScript function
232          * If the widget name is "createsite", then the function sakai.createsite will be executed.
233          * @param {String} widgetname The name of the widget
234          */
235         var informOnLoad = function(widgetname){
236             var doDelete;
237             // Check if the name of the widget is inside the widgets object.
238             if (widgets[widgetname] && widgets[widgetname].length > 0){
239 
240                 // Run through all the widgets with a specific name
241                 for (var i = 0, j = widgets[widgetname].length; i<j; i++){
242                     widgets[widgetname][i].done++;
243 
244                     if (widgets[widgetname][i].done === widgets[widgetname][i].todo){
245                          // Save the placement in the widgets variable
246                         sakai.api.Widgets.widgetLoader.widgets[widgets[widgetname][i].uid] = {
247                             "placement": widgets[widgetname][i].placement + widgets[widgetname][i].uid + "/" + widgetname,
248                             "name" : widgetname
249                         };
250 
251                         // Run the widget's main JS function
252                         var initfunction = window[widgetNameSpace][widgetname];
253                         initfunction(widgets[widgetname][i].uid, settings);
254 
255                         // Send out a "loaded" event for this widget
256                         $(window).trigger(widgetname + "_loaded", [widgets[widgetname][i].uid]);
257 
258                         doDelete = true;
259                     }
260                 }
261 
262                 // Remove the widget from the widgets object (clean up)
263                 if (doDelete){
264                     delete widgets[widgetname];
265                 }
266             }

269         /**
270          * Locate a tag and remove it from the content
271          * @param {String} content The complete content of a file (e.g. <div>...)
272          * @param {String} tagName The name of the tag you want to remove (link/script)
273          * @param {String} URLIdentifier The part that identifies the URL (href/src)
274          */
275         var locateTagAndRemove = function(content, tagName, URLIdentifier){
276             var returnObject = {
277                 URL : [],
278                 content : content
279             };
280             var regexp = new RegExp('<'+tagName+'.*?'+URLIdentifier+'\\s?=\\s?["|'+'\''+']([^"]*)["|'+'\''+'].*/.*?>', "gi");
281             var regexp_match_result = regexp.exec(content);
282             while (regexp_match_result !== null) {
283                 returnObject.URL[returnObject.URL.length] = regexp_match_result[1]; // value of URLIdentifier attrib
284                 returnObject.content = returnObject.content.replace(regexp_match_result[0],""); // whole tag
285                 regexp_match_result = regexp.exec(content);
286             }
287             return returnObject;
288         };

290         var sethtmlover = function(content,widgets,widgetname){
291 
292             var CSSTags = locateTagAndRemove(content, "link", "href");
293             content = CSSTags.content;
294             var stylesheets = [];
295 
296             for (var i = 0, j = CSSTags.URL.length; i<j; i++) {
297                 // SAKIII-1524 - Instead of loading all of the widget CSS files independtly,
298                 // we collect all CSS file declarations from all widgets in the current pass
299                 // of the WidgetLoader. These will then be loaded in 1 go.
300                 if ($.browser.msie && !sakai.api.Widgets.cssCache[CSSTags.URL[i]]) {
301                     stylesheets.push(CSSTags.URL[i]);
302                     sakai.api.Widgets.cssCache[CSSTags.URL[i]] = true;
303                 } else {
304                     sakai.api.Util.include.css(CSSTags.URL[i]);
305                 }
306             }
307 
308             var JSTags = locateTagAndRemove(content, "script", "src");
309             content = JSTags.content;
310 
311             for (var widget = 0, k = widgets[widgetname].length; widget < k; widget++){
312                 var container = $("<div>");
313                 container.html(content);
314                 $("#" + widgets[widgetname][widget].uid).append(container);
315 
316                 widgets[widgetname][widget].todo = JSTags.URL.length;
317                 widgets[widgetname][widget].done = 0;
318             }
319 
320             for (var JSURL = 0, l = JSTags.URL.length; JSURL < l; JSURL++){
321                 sakai.api.Util.include.js(JSTags.URL[JSURL]);
322             }
323 
324             return stylesheets;
325

328         /**
329          * Load the files that the widget needs (HTML/CSS and JavaScript)
330          * @param {Object} widgets
331          * @param {Object} batchWidgets A list of all the widgets that need to load
332          */
333         var loadWidgetFiles = function(widgets, batchWidgets){
334             var urls = [];
335             var requestedURLsResults = [];
336 
337             for(var k in batchWidgets){
338                 if(batchWidgets.hasOwnProperty(k)){
339                     var item = {
340                         "url" : k,
341                         "method" : "GET"
342                     };
343                     urls[urls.length] = item;
344                 }
345             }
346 
347             if(urls.length > 0){
348                 $.ajax({
349                     url: sakai.config.URL.BATCH,
350                     traditional: true,
351                     cache: false,
352                     data: {
353                         requests: $.toJSON(urls)
354                     },
355                     success: function(data){
356                         requestedURLsResults = data.results;
357                         var current_locale_string = false;
358                         if (typeof sakai.data.me.user.locale === "object") {
359                             current_locale_string = sakai.data.me.user.locale.language + "_" + sakai.data.me.user.locale.country;
360                         }
361                         var bundles = [];
362                         for (var i = 0, j = requestedURLsResults.length; i<j; i++) {
363                             var jsonpath = requestedURLsResults[i].url;
364                             var widgetname = batchWidgets[jsonpath];
365                             if ($.isPlainObject(sakai.widgets.widgets[widgetname].i18n)) {
366                                 if (sakai.widgets.widgets[widgetname].i18n["default"]){
367                                     var item = {
368                                         "url" : sakai.widgets.widgets[widgetname].i18n["default"],
369                                         "method" : "GET"
370                                     };
371                                     bundles.push(item);
372                                 }
373                                 if (sakai.widgets.widgets[widgetname].i18n[current_locale_string]) {
374                                     var item1 = {
375                                         "url" : sakai.widgets.widgets[widgetname].i18n[current_locale_string],
376                                         "method" : "GET"
377                                     };
378                                     bundles.push(item1);
379                                 }
380                             }
381                         }
382                         $.ajax({
383                             url: sakai.config.URL.BATCH,
384                             traditional: true,
385                             cache: false,
386                             data: {
387                                 requests: $.toJSON(bundles)
388                             },
389                             success: function(data){
390                                 var stylesheets = [];
391                                 for (var i = 0, j = requestedURLsResults.length; i < j; i++) {
392                                     // Current widget name
393                                     var widgetName = requestedURLsResults[i].url.split("/")[2];
394                                     // Check if widget has bundles
395                                     var hasBundles = false;
396                                     // Array containing language bundles
397                                     var bundleArr = [];
398                                     // Local and default bundle
399                                     for (var ii = 0, jj = data.results.length; ii < jj; ii++) {
400                                         if (widgetName === data.results[ii].url.split("/")[2]){
401                                             hasBundles = true;
402                                             if(data.results[ii].url.split("/")[4].split(".")[0] === "default"){
403                                                 sakai.data.i18n.widgets[widgetName] = sakai.data.i18n.widgets[widgetName] || {};
404                                                 sakai.data.i18n.widgets[widgetName]["default"] = sakai.data.i18n.changeToJSON(data.results[ii].body);
405                                             } else {
406                                                 sakai.data.i18n.widgets[widgetName] = sakai.data.i18n.widgets[widgetName] || {};
407                                                 sakai.data.i18n.widgets[widgetName][current_locale_string] = sakai.data.i18n.changeToJSON(data.results[ii].body);
408                                             }
409                                         }
410                                     }
411 
412                                     // Change messages
413                                     var translated_content = "",
414                                         lastend = 0;
415                                     if (hasBundles) {
416                                         var expression = new RegExp(".{1}__MSG__(.*?)__", "gm");
417                                         while (expression.test(requestedURLsResults[i].body)) {
418                                             var replace = RegExp.lastMatch;
419                                             var lastParen = RegExp.lastParen;
420                                             var quotes = "";
421 
422                                             // need to add quotations marks if key is adjacent to an equals sign which means its probably missing quotes - IE
423                                             if (replace.substr(0,2) !== "__"){
424                                                 if (replace.substr(0,1) === "="){
425                                                     quotes = '"';
426                                                 }
427                                                 replace = replace.substr(1, replace.length);
428                                             }
429                                             var toreplace;
430                                             // check for i18n debug
431                                             if (sakai.config.displayDebugInfo === true && sakai.data.me.user.locale && sakai.data.me.user.locale.language === "lu" && sakai.data.me.user.locale.country === "GB"){
432                                                 toreplace = quotes + replace.substr(7, replace.length - 9) + quotes;
433                                                 translated_content += requestedURLsResults[i].body.substring(lastend, expression.lastIndex - replace.length) + toreplace;
434                                                 lastend = expression.lastIndex;
435                                             } else {
436                                                 toreplace = quotes + sakai.api.i18n.Widgets.getValueForKey(widgetName, current_locale_string, lastParen) + quotes;
437                                                 translated_content += requestedURLsResults[i].body.substring(lastend, expression.lastIndex - replace.length) + toreplace;
438                                                 lastend = expression.lastIndex;
439                                             }
440                                         }
441                                         translated_content += requestedURLsResults[i].body.substring(lastend);
442                                     } else {
443                                         translated_content = sakai.api.i18n.General.process(requestedURLsResults[i].body, sakai.data.i18n.localBundle, sakai.data.i18n.defaultBundle);
444                                     }
445                                     var ss = sethtmlover(translated_content, widgets, widgetName);
446                                     for (var s = 0; s < ss.length; s++){
447                                         stylesheets.push(ss[s]);
448                                     }
449                                 }
450                                 // SAKIII-1524 - IE has a limit of maximum 32 CSS files (link or style tags). When
451                                 // a lot of widgets are loaded into 1 page, we can easily hit that limit. Therefore,
452                                 // we adjust the widgetloader to load all CSS files of 1 WidgetLoader pass in 1 style
453                                 // tag filled with import statements
454                                 if ($.browser.msie && stylesheets.length > 0) {
455                                     var numberCSS = $("head style, head link").length;
456                                     // If we have more than 30 stylesheets, we will merge all of the previous style
457                                     // tags we have created into the lowest possible number
458                                     if (numberCSS >= 30){
459                                         $("head style").each(function(index){
460                                             if ($(this).attr("title") && $(this).attr("title") === "sakai_widgetloader") {
461                                                 $(this).remove();
462                                             }
463                                         });
464                                     }
465                                     var allSS = [];
466                                     var newSS = document.createStyleSheet();
467                                     newSS.title = "sakai_widgetloader";
468                                     var totalImportsInCurrentSS = 0;
469                                     // Merge in the previously created style tags
470                                     if (numberCSS >= 30){
471                                         for (var k in sakai.api.Widgets.cssCache){
472                                             if (sakai.api.Widgets.cssCache.hasOwnProperty(k)) {
473                                                  if (totalImportsInCurrentSS >= 30){
474                                                      allSS.push(newSS);
475                                                      newSS = document.createStyleSheet();
476                                                      newSS.title = "sakai_widgetloader";
477                                                      totalImportsInCurrentSS = 0;
478                                                  }
479                                                  newSS.addImport(k);
480                                                  totalImportsInCurrentSS++;
481                                             }
482                                         }
483                                     }
484                                     // Add in the stylesheets declared in the widgets loaded
485                                     // in the current pass of the WidgetLoader
486                                     for (var m = 0, mm = stylesheets.length; m < mm; m++) {
487                                         if (totalImportsInCurrentSS >= 30){
488                                             allSS.push(newSS);
489                                             newSS = document.createStyleSheet();
490                                             newSS.title = "sakai_widgetloader";
491                                             totalImportsInCurrentSS = 0;
492                                         }
493                                         newSS.addImport(stylesheets[m]);
494                                     }
495                                     allSS.push(newSS);
496                                     // Add the style tags to the document
497                                     for (var z = 0; z < allSS.length; z++) {
498                                         $("head").append(allSS[z]);
499                                     }
500                                 }
501                             }
502                         });
503                     }
504                 });
505             }
506         };

508         /**
509          * Insert the widgets into the page
510          * @param {String} containerId The id of the container element
511          * @param {Boolean} showSettings Show the settings for the widget
512          * @param {String} context The context of the widget (e.g. siteid)
513          */
514         var insertWidgets = function(containerId, showSettings, context){
515 
516             // Use document.getElementById() to avoid jQuery selector escaping issues with '/'
517             var el = containerId ? document.getElementById(containerId) : $(document.body);
518 
519             // Array of jQuery objects that contains all the elements in the with the widget selector class.
520             var divarray = $(widgetSelector, el);
521 
522             // Check if the showSettings variable is set, if not set the settings variable to false
523             settings = showSettings || false;
524 
525             // Array that will contain all the URLs + names of the widgets that need to be fetched with batch get
526             var batchWidgets = [];
527 
528             // Run over all the elements and load them
529             for (var i = 0, j = divarray.length; i < j; i++){
530                 var id = divarray[i].id;
531                 var split = id.split("_");
532                 var widgetname = split[1];
533 
534                 // Set the id for the container of the widget
535                 var widgetid;
536                 if (split[2]){
537                     widgetid = split[2];
538                 } else if(widgetname) {
539                     widgetid = widgetname + "container" + Math.round(Math.random() * 10000000);
540                 }
541 
542                 // Check if the widget is an iframe widget
543                 if (sakai.widgets.widgets[widgetname] && sakai.widgets.widgets[widgetname].iframe){
544 
545                     // Get the information about the widget in the widgets.js file
546                     var portlet = sakai.widgets.widgets[widgetname];
547 
548                     // Check if the scrolling property has been set to true
549                     var scrolling = portlet.scrolling ? "auto" : "no";
550 
551                     var src = portlet.url;
552 
553                     // Construct the HTML for the iframe
554                     var html = '<div id="widget_content_'+ widgetname + '">' +
555                                    '<iframe src="'+ src +'" ' +
556                                    'frameborder="0" ' +
557                                    'height="'+ portlet.height +'px" ' +
558                                    'width="100%" ' +
559                                    'scrolling="' + scrolling + '"' +
560                                    '></iframe></div>';
561 
562                     // Add the HTML for to the iframe widget container
563                     $("#" + widgetid + "_container").html(html).addClass("fl-widget-content").parent().append('<div class="fl-widget-no-options fl-fix"><div class="widget-no-options-inner"><!-- --></div></div>');
564 
565                 }
566 
567                 // The widget isn't an iframe widget
568                 else if (sakai.widgets.widgets[widgetname]){
569 
570                     // Set the placement for the widget
571                     var placement = "";
572                     if (split[3] !== undefined){
573                         var length = split[0].length + 1 + widgetname.length + 1 + widgetid.length + 1;
574                         placement = id.substring(length);
575                     } else if (context){
576                         placement = context;
577                     }
578 
579                     // Check if the widget exists
580                     if (!widgets[widgetname]){
581                         widgets[widgetname] = [];
582                     }
583 
584                     // Set the initial properties for the widget
585                     var index = widgets[widgetname].length;
586                     widgets[widgetname][index] = {
587                         uid : widgetid,
588                         placement : placement,
589                         id : id
590                     };
591                     var floating = "inline_class_widget_nofloat";
592 
593                     // Check if the browser supports cssFloat (other browsers) or styleFloat (IE)
594                     var styleFloat = $.support.cssFloat ? "cssFloat" : "styleFloat";
595                     if (divarray[i].style[styleFloat]) {
596                         floating = divarray[i].style[styleFloat] === "left" ? "inline_class_widget_leftfloat" : "inline_class_widget_rightfloat";
597                     }
598                     widgets[widgetname][index].floating = floating;
599                 }
600             }
601 
602             for (i in widgets){
603                 if (widgets.hasOwnProperty(i)) {
604                     for (var ii = 0, jj = widgets[i].length; ii<jj; ii++) {
605 
606                         // Replace all the widgets with id "widget_" to widgets with new id's
607                         // and add set the appropriate float class
608                         $(document.getElementById(widgets[i][ii].id)).replaceWith($('<div id="'+widgets[i][ii].uid+'" class="' + widgets[i][ii].floating + '"></div>'));
609                     }
610 
611                     var url = sakai.widgets.widgets[i].url;
612                     batchWidgets[url] = i; //i is the widgetname
613                 }
614             }
615 
616             // Load the HTML files for the widgets
617             loadWidgetFiles(widgets, batchWidgets);
618 
619         };
620 
621         insertWidgets(id, showSettings, context);
