 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.addcontent
 25  *
 26  * @class addcontent
 27  *
 28  * @description
 29  * Initialize the add content widget - This widget adds content to a site
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.addcontent = function(tuid, showSettings){
 36 
 37 
 38     /////////////////////////////
 39     // Configuration variables //
 40     /////////////////////////////
 41 
 42 
 43     ///////////////////
 44     // CSS selectors //
 45     ///////////////////
 46 
 47     var $rootel = $("#" + tuid);
 48     var $addcontent_button = $("#addcontent_button", $rootel);
 49     var $addcontent_button_cancel = $("#addcontent_button_cancel", $rootel);
 50     var $addcontent_container_buttons = $("#addcontent_container_buttons", $rootel);
 51     var $addcontent_container = $("#addcontent_container", $rootel);
 52     var $addcontent_container_content = $("#addcontent_container_content", $rootel);
 53     var $addcontent_container_content_box_container = $(".addcontent_container_content_box_container", $rootel);
 54     var $addcontent_container_content_box_selected = $(".addcontent_container_content_box_selected", $rootel);
 55     var $addcontent_container_content_template = $("#addcontent_container_content_template", $rootel);
 56     var $addcontent_container_show = $("#addcontent_container_show", $rootel);
 57     var $addcontent_container_show_box_download = $("#addcontent_container_show_box_download", $rootel);
 58     var $addcontent_container_show_box_profile = $("#addcontent_container_show_box_profile", $rootel);
 59     var $addcontent_container_show_template = $("#addcontent_container_show_template", $rootel);
 60 
 61     var addcontent_container_content_box_selected = "addcontent_container_content_box_selected";
 62     var addcontent_container_place_selected = "addcontent_container_place_selected";
 63 
 64 
 65     //////////////////////
 66     // General function //
 67     //////////////////////
 68 
 69     /**
 70      * Render the general template
 71      * @param {Object} data A JSON object that is used by the main template
 72      */
 73     var renderTemplate = function(data){
 74 
 75         // Execute the render function
 76         $.TemplateRenderer($addcontent_container_content_template, data, $addcontent_container_content);
 77 
 78         // Show the content container
 79         $addcontent_container_content.show();
 80 
 81         // Show the buttons
 82         $addcontent_container_buttons.show();
 83 
 84     };
 85 
 86     /**
 87      * Render the show template (in non settings mode)
 88      * @param {Object} data The JSON data you get back from the server
 89      */
 90     var renderTemplateShow = function(data){
 91 
 92         // Execute the render function
 93         $.TemplateRenderer($addcontent_container_show_template, data, $addcontent_container_show);
 94 
 95         // Show the show container
 96         $addcontent_container_show.show();
 97 
 98     };
 99 
100     /**
101      * The general add binding function
102      */
103     var addBinding = function(){
104 
105         // Reinitialize the jQuery object(s)
106         $addcontent_container_content_box_container = $($addcontent_container_content_box_container.selector);
107 
108         // Bind the click event to the content box container
109         $addcontent_container_content_box_container.bind("click", function(){
110 
111             // Reinitialize the jQuery object(s)
112             $addcontent_container_content_box_selected = $($addcontent_container_content_box_selected.selector);
113 
114             // Remove the previous selection(s)
115             $addcontent_container_content_box_selected.removeClass(addcontent_container_content_box_selected);
116 
117             // Add the selected class
118             $(this).addClass(addcontent_container_content_box_selected);
119         });
120 
121         // Bind the add content button
122         $addcontent_button.bind("click", function(){
123 
124             // Reinitialize the jQuery object(s)
125             $addcontent_container_content_box_selected = $($addcontent_container_content_box_selected.selector);
126 
127             var data ={
128                 path: $addcontent_container_content_box_selected.find("img").attr("src")
129             };
130 
131             // Save the widget data
132             sakai.api.Widgets.saveWidgetData(tuid, data, function(success){
133                 if(success){
134                     sakai.api.Widgets.Container.informFinish(tuid, "addcontent");
135                 } else {
136                     debug.warn("Widget data could not be saved");
137                     // TODO show a valid error message to the user
138                 }
139             });
140 
141         });
142 
143         // Bind the cancel buttons
144         $addcontent_button_cancel.bind("click", function(){
145             sakai.api.Widgets.Container.informCancel(tuid, "addcontent");
146         });
147 
148     };
149 
150     /**
151      * Add binding to the show container
152      * @param {Object} json JSON object containing useful data
153      */
154     var addBindingShow = function(json){
155 
156         // Reinitialise the jQuery selectors
157         $addcontent_container_show_box_download = $($addcontent_container_show_box_download.selector);
158         $addcontent_container_show_box_profile = $($addcontent_container_show_box_profile.selector);
159 
160         // Add binding to the download button
161         $addcontent_container_show_box_download.bind("click", function(){
162 
163             // Open a new window to download the actual resource
164             window.open(json.url);
165 
166         });
167 
168         // Add binding to the profile button
169         $addcontent_container_show_box_profile.bind("click", function(){
170 
171             // Redirect the user to the profile page
172             document.location = "/dev/content_profile.html#content_path=" + json.url.replace(sakai.config.SakaiDomain, "/");
173 
174         });
175 
176     };
177 
178     var filesLoaded = function(bool, data){
179         // Render the template for a file
180         renderTemplate(data);
181         
182         // Add binding
183         addBinding();
184         
185         // Show the container
186         $addcontent_container.show();
187     };
188 
189     /**
190      * Load the files from a specific location
191      * @param {String} location The location where you want to load the files from
192      */
193     var loadFiles = function(location){
194         if(location === "myfiles"){
195             var data = {
196                 q: "*",
197                 sortOn: "jcr:created",
198                 sortOrder: "descending"
199             };
200             sakai.api.Server.loadJSON("/var/search/pool/me/manager.1", filesLoaded, data);
201         }
202 
203     };
204 
205     /**
206      * Load the information about a file
207      */
208     var loadFileInfo = function(){
209 
210         // Load the widget data
211         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
212 
213             // Only do something on success
214             // TODO show a valid error message if something goes wrong
215             if(success && data.path){
216                 if (sakai.content_profile.content_data && sakai.content_profile.content_data.data) {
217                     json = sakai.content_profile.content_data;
218                     var splitslash = sakai.content_profile.content_data.contentpath.split("/");
219                     json.data.name = splitslash[splitslash.length -1];
220                     // Render the show template
221                     renderTemplateShow(json);
222 
223                     // Add binding to the show container
224                     addBindingShow(json);
225                 } else {
226                     sakai.content_profile.loadContentProfile(function(success) {
227                         if (success) {
228                             json = sakai.content_profile.content_data;
229                             var splitslash = sakai.content_profile.content_data.contentpath.split("/");
230                             json.data.name = splitslash[splitslash.length -1];
231                             // Render the show template
232                             renderTemplateShow(json);
233 
234                             // Add binding to the show container
235                             addBindingShow(json);
236                         }
237                     });
238                 }
239             }
240         });
241     };
242 
243 
244     ////////////////////
245     // Initialisation //
246     ////////////////////
247 
248     /**
249      * The main initialisation function
250      */
251     var init = function(){
252 
253         // Check whether it is in settings mode or not
254         if (showSettings) {
255 
256             loadFiles("myfiles");
257 
258         }
259         else {
260 
261             loadFileInfo();
262 
263         }
264 
265     };
266 
267     // Execute the init function
268     init();
269 };
270 sakai.api.Widgets.widgetLoader.informOnLoad("addcontent");
