 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.contentpicker
 24  *
 25  * @class contentpicker
 26  *
 27  * @description
 28  * Initialize the contentpicker widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  * @param {Boolean} showSettings Show the settings of the widget or not
 33  */
 34 sakai.contentpicker = function(tuid, showSettings) {
 35 
 36     var $rootel = $("#" + tuid);
 37 
 38     var $contentpicker_dialog = $("#contentpicker_dialog", $rootel);
 39     var $contentpicker_page_name = $("#contentpicker_page_name", $rootel);
 40     var $contentpicker_placing_content_label = $("#contentpicker_placing_content_label", $rootel);
 41     var $contentpicker_share_files_with_label = $("#contentpicker_share_files_with_label", $rootel);
 42     var $contentpicker_place_content = $("#contentpicker_place_content", $rootel);
 43     var $contentpicker_cancel = $("#contentpicker_cancel", $rootel);
 44     var $contentpicker_content_input = $("#contentpicker_content_input", $rootel);
 45     var $contentpicker_search_for_content = $("#contentpicker_search_for_content", $rootel);
 46 
 47     var $contentpicker_page_name_template = $("#contentpicker_page_name_template", $rootel);
 48     var $contentpicker_new_item_template = $("#contentpicker_new_item_template", $rootel);
 49 
 50     var $fileuploadContainer = $("#fileupload_container", $rootel);
 51     var $uploadContentLink = $("#upload_content", $rootel);
 52 
 53     var selectedItems = [];
 54     var firstTime = true;
 55     var widgetData = false;
 56 
 57     var pickerConfig = {
 58         "mode": "picker",
 59         "name": "Page",
 60         "limit": false,
 61         "filter": false,
 62         "type": "choose"
 63     };
 64 
 65     /**
 66      * Render the embed screen
 67      */
 68     var render = function() {
 69         selectedItems = [];
 70         $contentpicker_share_files_with_label.hide();
 71         $contentpicker_placing_content_label.hide();
 72         if (pickerConfig.type && pickerConfig.type === "share") {
 73             $contentpicker_share_files_with_label.show();
 74         } else {
 75             $contentpicker_placing_content_label.show();
 76         }
 77         $.TemplateRenderer($contentpicker_page_name_template, {"name": pickerConfig.name}, $contentpicker_page_name);
 78 
 79         if (firstTime) {
 80             setupAutoSuggest();
 81             firstTime = false;
 82         } else {
 83             doReset();
 84         }
 85         $("#as-values-" + tuid).val("");
 86         $(".as-selection-item").remove();
 87     };
 88 
 89     /**
 90      * Do a reset of the embed screen
 91      */
 92     var doReset = function() {
 93         $("#as-values-" + tuid).val("");
 94         $(".as-selection-item").remove();
 95     };
 96 
 97     /**
 98      * Get the mimetype of a provided file
 99      * @param {Object} file File provided to get mimetype of
100      */
101     var getMimeType = function(file) {
102         var mimetype = "";
103         mimetype = file["jcr:content"] ? file["jcr:content"]["jcr:mimeType"] : "";
104         return mimetype;
105     };
106 
107     /**
108      * Creates an object out of results provided
109      * This object contains valuable information about the file like path, name, type,...
110      * @param {Object} result results provided (eg through a search)
111      * @param {Object} name optional name provided
112      */
113     var createDataObject = function(result, name) {
114         var mimetype = getMimeType(result);
115         var dataObj = {
116             "value": name || result['jcr:name'],
117             "name": result['sakai:pooled-content-file-name'],
118             "type": "file",
119             "filetype": mimetype.split("/")[0],
120             "mimetype": mimetype,
121             "description": result["sakai:description"] || "",
122             "path": "/p/" + (name || result['jcr:name']),
123             "fileSize": sakai.api.Util.convertToHumanReadableFileSize(result["jcr:content"][":jcr:data"]),
124             "link": "/p/" + (name || result['jcr:name']) + "/" + result['sakai:pooled-content-file-name'],
125             "extension": result['sakai:fileextension']
126         };
127         return dataObj;
128     };
129 
130     /**
131      * When typing in the suggest box this function is executed to provide the user with a list of possible autocompletions
132      */
133     var setupAutoSuggest = function() {
134         $contentpicker_content_input.autoSuggest("",{
135             source: function(query, add) {
136                 searchUrl = sakai.config.URL.POOLED_CONTENT_MANAGER;
137                 sakai.api.Server.loadJSON(searchUrl.replace(".json", ""), function(success, data){
138                     if (success) {
139                         var suggestions = [];
140                         $.each(data.results, function(i) {
141                             var dataObj = createDataObject(data.results[i]);
142                             var doAdd = true;
143                             if (pickerConfig.filter) {
144                                 if (dataObj.filetype !== pickerConfig.filter) {
145                                     doAdd = false;
146                                 }
147                             }
148                             if (doAdd) {
149                                 suggestions.push(dataObj);
150                             }
151                         });
152                         add(suggestions);
153                     }
154                 }, {"q": "*" + query.replace(/\s+/g, "* OR *") + "*", "page": 0, "items": 15});
155             },
156             retrieveLimit: 10,
157             asHtmlID: tuid,
158             selectedItemProp: "name",
159             searchObjProps: "name",
160             selectionLimit: pickerConfig.limit,
161             resultClick: function(data) {
162                 selectedItems.push(data.attributes);
163                 $contentpicker_place_content.removeAttr("disabled");
164             },
165             selectionRemoved: function(elem) {
166                 removeItemFromSelected(elem.html().split("</a>")[1]); // get filename
167                 elem.remove();
168                 if (selectedItems.length === 0) {
169                     $contentpicker_place_content.attr("disabled", "disabled");
170                 }
171             }
172         });
173     };
174 
175     /**
176      * Removes a previously selected item from the list of selected items
177      * @param {Object} fileName name of the selected item to be removed from the list
178      */
179     var removeItemFromSelected = function(fileName) {
180         var newItems = [];
181         $(selectedItems).each(function(i, val) {
182            if (val.name !== fileName) {
183                newItems.push(val);
184            }
185         });
186         selectedItems = newItems;
187     };
188 
189     /**
190      * Called when file(s) are selected in the picker advanced widget and need to be added to the list of files that will be embedded.
191      * @param {Object} files Array of files selected in the picker advanced widget
192      */
193     var addChoicesFromPickeradvanced = function(files) {
194         var filesPicked = 0;
195         $.each(files, function(i,val) {
196             filesPicked++;
197         });
198         // revisit this next conditional -- right now it'll clear out all selections, not just add up to the limit
199         if (pickerConfig.limit && filesPicked && ($(".as-selection-item").length + filesPicked) > pickerConfig.limit) {
200             $("#as-values-" + tuid).val('');
201             $(".as-selection-item").remove();
202         }
203         if (filesPicked > 0) {
204             $contentpicker_place_content.removeAttr("disabled");
205         }
206         $.each(files, function(i,val) {
207             var newObj = createDataObject(val, val["jcr:name"]);
208             selectedItems.push(newObj);
209             $contentpicker_content_input.autoSuggest.add_selected_item(newObj, newObj.value);
210         });
211         $("input#" + tuid).val('').focus();
212     };
213 
214     /**
215      * Called when newly uploaded files need to be added to the list of files that will be embedded
216      * @param {Object} files Array containing a list of files
217      */
218     var addChoicesFromFileUpload = function(files) {
219       $.each(files, function(i,val) {
220           $.ajax({
221              url: val.url + ".infinity.json",
222              success: function(data) {
223                  var newObj = createDataObject(data, val.url.split("/p/")[1]);
224                  selectedItems.push(newObj);
225                  $contentpicker_content_input.autoSuggest.add_selected_item(newObj, newObj.value);
226              }
227           });
228       });
229       $("input#" + tuid).val('').focus();
230       $contentpicker_place_content.removeAttr("disabled");
231     };
232 
233     /**
234      * Once the content has been placed on the page it has to be associated with the group
235      * The group is set as a viewer of the content
236      * @param {Object} embeddedItems Array of object containing information about the selected items. Only the path variable is used.
237      */
238     var associatedEmbeddedItemsWithGroup = function(embeddedItems){
239         var data = [];
240         for (var embeddedItem in embeddedItems) {
241             if (embeddedItems.hasOwnProperty(embeddedItem)) {
242                 var item = {
243                     "url": embeddedItems[embeddedItem].path + ".members.json",
244                     "method": "POST",
245                     "parameters": {
246                         ":viewer": sakai.currentgroup.id
247                     }
248                 };
249                 data[data.length] = item;
250             }
251         }
252 
253         $.ajax({
254             url: sakai.config.URL.BATCH,
255             traditional: true,
256             type: "POST",
257             cache: false,
258             data: {
259                 requests: $.toJSON(data)
260             }
261         });
262     };
263 
264     /**
265      * Fire the event to indicate the picker is done picking and
266      * call the function that associates the content with this group
267      */
268     var finish = function() {
269         // Associate embedded items with the group
270         associatedEmbeddedItemsWithGroup(selectedItems);
271 
272         $(window).trigger("sakai-contentpicker-finished", {"items": selectedItems});
273         $contentpicker_dialog.jqmHide();
274     };
275 
276     // Bind Events
277     $contentpicker_place_content.bind("click", function() {
278         finish();
279     });
280 
281     $contentpicker_cancel.bind("click", function() {
282         $contentpicker_dialog.jqmHide();
283     });
284 
285     $uploadContentLink.bind("click", function() {
286         $(window).trigger("sakai-fileupload-init");
287     });
288 
289     $(window).unbind("sakai-fileupload-complete");
290     $(window).bind("sakai-fileupload-complete", function(e, data) {
291         var files = data.files;
292         addChoicesFromFileUpload(files);
293     });
294 
295     $(window).unbind("sakai-pickeradvanced-finished");
296     $(window).bind("sakai-pickeradvanced-finished", function(e, data) {
297         addChoicesFromPickeradvanced(data.toAdd);
298     });
299 
300     $(window).unbind("sakai-contentpicker-init");
301     $(window).bind("sakai-contentpicker-init", function(e, config) {
302 
303         // position dialog box at users scroll position
304         var htmlScrollPos = $("html").scrollTop();
305         var docScrollPos = $(document).scrollTop();
306         if (htmlScrollPos > 0) {
307             $contentpicker_dialog.css({"top": htmlScrollPos + 50 + "px"});
308         } else if (docScrollPos > 0) {
309             $contentpicker_dialog.css({"top": docScrollPos + 50 + "px"});
310         }
311 
312         pickerConfig = $.extend(true, pickerConfig, config);
313         render();
314         $contentpicker_dialog.jqmShow();
315     });
316 
317     $(window).unbind("sakai-pickeradvanced-ready");
318     $(window).bind("sakai-pickeradvanced-ready", function(e) {
319         $contentpicker_search_for_content.bind("click", function() {
320             var pickerConfig = {
321                 "type": "content"
322             };
323             if (pickerConfig.limit) {
324                 pickerConfig.limit = pickerConfig.limit;
325             }
326             $(window).trigger("sakai-pickeradvanced-init", {"config": pickerConfig});
327         });
328     });
329 
330     $contentpicker_dialog.jqm({
331         modal: true,
332         overlay: 20,
333         zIndex: 3000,
334         toTop: true
335     });
336 
337     var doInit = function() {
338         $(window).trigger("sakai-contentpicker-ready");
339         sakai.api.Widgets.widgetLoader.insertWidgets("#"+tuid);
340     };
341 
342     doInit();
343 };
344 sakai.api.Widgets.widgetLoader.informOnLoad("contentpicker");
