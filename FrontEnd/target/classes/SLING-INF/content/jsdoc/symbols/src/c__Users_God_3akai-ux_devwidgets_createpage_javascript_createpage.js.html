 18 /*global $ */
 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.createpage
 24  *
 25  * @class createpage
 26  *
 27  * @description
 28  * createpage widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  * @param {Boolean} showSettings Show the settings of the widget or not
 33  */
 34 sakai.createpage = function(tuid, showSettings){
 35 
 36     /////////////////////////////
 37     // Configuration variables //
 38     /////////////////////////////
 39 
 40     // jQuery DOM objects
 41     var $rootel = $("#" + tuid);
 42     var $createpageContainer = $("#createpage_container", $rootel);
 43     var $selectPageType = $("#createpage_page_type", $rootel);
 44     var $inputPageTitleContainer = $("#createpage_title_container", $rootel);
 45     var $inputPageTitle = $("#createpage_title", $rootel);
 46     var $createpageForm = $("#createpage_form", $rootel);
 47     var $createpageSubmit = $("#createpage_save", $rootel);
 48     var $createpageCancel = $("#createpage_cancel", $rootel);
 49     var $createpageProcessing = $("#createpage_processing", $rootel);
 50     var $createpageTemplates = $("#createpage_templates", $rootel);
 51     var $createpageTemplatesTemplate = $("#createpage_templates_template", $rootel);
 52     
 53     // CSS Selectors
 54     var errorMsgClass = "createpage_error_msg";
 55     var invalidFieldClass = "invalid";
 56     var buttonClass = "s3d-button";
 57     var radioButtons = "createpagetemplate_radio";
 58 
 59     ///////////////////////
 60     // Utility functions //
 61     ///////////////////////
 62 
 63     /**
 64      * This function is called before the jqModal hides. It removes any error
 65      * notices, resets values for select and text inputs and hides templates
 66      */
 67     var resetModalDialog = function (hash) {
 68         // hide the overlay
 69         hash.w.hide();
 70         hash.o.remove();
 71 
 72         // remove any errors
 73         $("." + errorMsgClass).hide();
 74         $inputPageTitle.removeClass(invalidFieldClass);
 75 
 76         // clear input
 77         $inputPageTitle.val("");
 78 
 79         // reset page type select element
 80         $selectPageType.val("blank");
 81         
 82         // hide templates
 83         $createpageTemplates.hide();
 84 
 85         // hide processing message
 86         hideProcessing();
 87     };
 88 
 89     /**
 90      * Mark the page title field as erroneous and show an error message
 91      */
 92     var showPageTitleError = function () {
 93         $("." + errorMsgClass).show();
 94         $inputPageTitle.addClass(invalidFieldClass);
 95         $inputPageTitle.focus();
 96     };
 97 
 98     /**
 99      * Show page templates available to the user
100      */
101     var showTemplates = function () {
102         var json = {
103             templates: []
104         };
105         if(sakai.sitespages && sakai.sitespages.mytemplates) {
106             // create a custom list of templates with just the info we care about
107             for(var t in sakai.sitespages.mytemplates) {
108                 if(sakai.sitespages.mytemplates.hasOwnProperty(t)) {
109                     var template = {
110                         "id": t,
111                         "name": sakai.sitespages.mytemplates[t].name,
112                         "desc": sakai.sitespages.mytemplates[t].description
113                     };
114                     json.templates.push(template);
115                 }
116             }
117         }
118         // pass the array to HTML view
119         $createpageTemplates.html($.TemplateRenderer($createpageTemplatesTemplate,
120             json));
121         $createpageTemplates.show();
122 
123         // select the first item
124         $("input:radio:first", $createpageContainer).attr("checked","checked");
125 
126         // if no templates, disable the create button
127         if(json.templates.length === 0) {
128             $createpageSubmit.attr("disabled","disabled");
129         }
130     };
131 
132     /**
133      * Show the "Processing..." message to the user
134      */
135     var showProcessing = function () {
136         $("#createpage_container " + "." + buttonClass).hide();
137         $createpageProcessing.show();
138     };
139 
140     /**
141      * Hide the "Processing..." message to the user
142      */
143     var hideProcessing = function () {
144         $createpageProcessing.hide();
145         $("#createpage_container " + "." + buttonClass).show();
146     };
147 
148 
149     ////////////////////
150     // Event Handlers //
151     ////////////////////
152 
153     /**
154      * Validate input and initiate creating pages when the user submits the form
155      */
156     $createpageForm.submit(function () {
157         // check if we have valid input
158         var pageTitle = $inputPageTitle.val();
159         if(!pageTitle) {
160             showPageTitleError();
161         } else {
162             // we've got a valid title
163             showProcessing();  // display "Processing..."
164             var pageType = $selectPageType.val();
165             if(pageType === "blank") {
166                 sakai.sitespages.createNewPage(pageTitle, "", handleNewPageCreation);
167             } else if(pageType === "template") {
168                 // fetch the selected template and its content to create a new page
169                 var selectedTemplate = $("input:radio:checked", $createpageContainer).val();
170                 sakai.sitespages.createNewPage(pageTitle,
171                     sakai.sitespages.mytemplates[selectedTemplate]["pageContent"]["sakai:pagecontent"],
172                     handleNewPageCreation);
173             } else {
174                 sakai.sitespages.addDashboardPage(pageTitle, handleNewPageCreation);
175             }
176         }
177     });
178 
179     /**
180      * Callback function called upon completion of sakai.sitespages.createNewPage
181      * called on submit event of the create page form.
182      * @param {Boolean} success true if the call succeeded, false otherwise
183      */
184     var handleNewPageCreation = function (success) { 
185         hideProcessing();
186         if(success) {
187             // hide the modal
188             $createpageContainer.jqmHide();
189         } else {
190             debug.error("createpage.js - handleNewPageCreation: creating page failed.");
191             sakai.api.Util.notification.show($("#createpage_createpage").text(),
192                                             $("#createpage_cannot_create_page").text(),
193                                             sakai.api.Util.notification.type.ERROR);
194         }
195     };
196 
197     /**
198      * Show available templates when the user selects Create page from template
199      * in the page type dropdown. If the other types are selected, focus the 
200      * input are immediately.
201      */
202     $selectPageType.change(function () {
203         var pageType = $selectPageType.val();
204         if(pageType === "template") {
205             // display templates
206             showTemplates();
207         } else {
208             // hide templates, ensure submit button is visible
209             $createpageTemplates.hide();
210             $createpageSubmit.removeAttr("disabled");
211         }
212         $inputPageTitle.focus();
213     });
214 
215     /**
216      * Delete the template when the user clicks the delete link
217      */
218     $createpageContainer.delegate("a.createpage_delete_template_link", "click",
219     function () {
220         var templateDeleted = this.id.split("_")[1];
221         delete sakai.sitespages.mytemplates[templateDeleted];
222         // Save updated template preferences
223         // -- this has a bug right now. The save is not overwriting the templates
224         // -- at /~userid/private/templates as expected - maybe a special jcr flag (i.e. :replace)?
225         sakai.api.Server.saveJSON("/~" + sakai.data.me.user.userid + "/private/templates",
226             sakai.sitespages.mytemplates, function(success, response) {
227             if (success) {
228                 showTemplates();
229             } else {
230                 debug.error("createpage.js: Failed to delete template with id="+templateDeleted);
231             }
232         });
233     });
234 
235     //////////////////////////////
236     // Initialization function  //
237     //////////////////////////////
238 
239     /**
240      * Public init function that shows the createpage modal dialog
241      */
242     sakai.createpage.initialise = function(){
243         // add jqModal functionality to the container
244         $createpageContainer.jqm({
245             modal: true,
246             overlay: 20,
247             toTop: true,
248             onHide: resetModalDialog
249         })
250         .jqmAddClose($createpageCancel);
251         
252         // Load page templates
253         sakai.sitespages.loadTemplates();
254         $createpageSubmit.removeAttr("disabled");
255         // show container
256         $createpageContainer.jqmShow();
257         $inputPageTitle.focus();
258     };
259 
260 };
261 
262 sakai.api.Widgets.widgetLoader.informOnLoad("createpage");
