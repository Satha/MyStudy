145     /**
146      * Parse a json integer to a valid date
147      * @param {Integer} dateInput Integer of a date that needs to be parsed
148      * @returns {Date}
149      */
150     var parseDate = function(dateInput){
151         //2009-08-19 11:29:53+0100
152         //2009-08-19T10:58:27
153         if (dateInput !== null) {
154             /** Get the date with the use of regular expressions */
155             var match = /([0-9]{4})\-([0-9]{2})\-([0-9]{2}).([0-9]{2}):([0-9]{2}):([0-9]{2})/.exec(dateInput); // 2009-08-14T12:18:50
156             var d = new Date();
157             if (match !== undefined) {
158                 d.setYear(match[1]);
159                 d.setMonth(match[2] - 1);
160                 d.setDate(match[3]);
161                 d.setHours(match[4]);
162                 d.setMinutes(match[5]);
163                 d.setSeconds(match[6]);
164             }
165             return d;
166         }
167         return null;
168     };

419     /**
420      * Makes sure the post is properly formatted so the template engine can interpret it.
421      * @param {Object} o
422      */
423     var doMarkUpOnPost = function(o){
424         var post = o.post;
425         var uid = post["sakai:from"];
426         post.date = sakai.api.l10n.transformDateTimeShort(parseDate(post["sakai:created"]));
427         post['sakai:body'] = (""+post['sakai:body']).replace(/\n/g, "<br />");
428         post.showEdit = false;
429         post.showDelete = false;
430 
431         // Show or hide the edit or delete button
432         if (me.user.superUser) {
433             post.canEdit = true;
434             post.canDelete = true;
435         }
436 
437         // Get the user's firstName, lastName and picture if it's in the database
438         var profile = post.profile[0];
439         post.profile.fullName = parseName(uid, profile);
440         post.profile.picture = parsePicture(uid, profile);
441 
442         // Check if someone edited the post
443         // post.sakai:editedbyprofiles is an array of objects that contain all the editers for this post.
444         // TODO: Fix this weird assignment bug.
445         var editedByProfiles = post['sakai:editedbyprofiles'];
446         if (editedByProfiles) {
447             var lastEditter = editedByProfiles[editedByProfiles.length - 1];
448 
449             // Get the profile info from the user that edited the post
450             post.editedByUserid = lastEditter.userid;
451             post.editedByName = parseName(lastEditter.userid, lastEditter);
452             //post.editedByDate = formatDate(parseDate(lastEditter.date));
453         }
454         o.post = post;
455 
456         // Count all the replies;
457         countReplies = 0;
458         addCountReplies(o);
459         o.post.nrOfReplies = "" + countReplies;
460 
461 
462         // weird json bug.
463         o.post["sakai:deleted"] = (o.post["sakai:deleted"] && (o.post["sakai:deleted"] === "true" || o.post["sakai:deleted"] === true)) ? true : false;
464 
465         displayReplies(o);
466 
467         return o;
468     };

471         for (var i = 0, j = o.replies.length; i < j; i++) {
472             o.replies[i] = doMarkUpOnPost(o.replies[i]);
473         }
474     };

882     var submitSettings = function(){
883         // Determine which view we are on.
884         var post = {};
885         if ($(discussionSettingsExistingContainer, rootel).is(":visible")) {
886             // The user wants to add an existing item, we get the selected post and get the marker from it.
887             post = getSelectedDiscussion(selectedExistingDiscussionID);

900                     alert("Please fill in all the fields.");
901                 }
902                 else {
903                     if (initialPost !== false) {
904                         // We already have an initalpost.
905                         // edit this one.
906                         editPost(initialPost["sakai:id"], post['sakai:subject'], post['sakai:body']);
907                     }
908                     else {
909                         // create a new post.
910                         createInitialPost(post);
911                     }
912                 }
913             }
914             else
915                 if ($(discussionSettingsDisplayOptionsContainer, rootel).is(":visible")) {
916 
917                     if (initialPost === false) {
918                         var subject = $(discussionSettingsNewSubject, rootel).val();
919                         var body = $(discussionSettingsNewBody, rootel).val();
920                         if (subject.replace(/ /g, "") !== "" && body.replace(/ /g, "") !== "") {
921                             post = createPostObject();
922                             createInitialPost(post);
923                         }
924                         else {
925                             post = getSelectedDiscussion(selectedExistingDiscussionID);
926 
927                             if (post === false) {
928                                 sakai.api.Util.notification.show($("#discussion_discussion").text(),
929                                                                 $("#discussion_post_to_new_discussion").text(),
930                                                                 sakai.api.Util.notification.type.ERROR);
931                                 return;
932                             }
933                             widgetSettings.marker = post["sakai:marker"];
934                         }
935                     }
936 
937                     if ($('#' + tuid + ' #discussion_settings_link_display_button').is(":checked")) {
938                         widgetSettings.displayMode = 'link';
939                     }
940                     else
941                         if ($('#' + tuid + ' #discussion_settings_inline_display_button').is(":checked")) {
942                             widgetSettings.displayMode = 'inline';
943                         }
944 
945                     var callback1 = finishSettingsContainer;
946                     saveWidgetSettings(callback1);
947                 }
948     };
