 22 /**
 23  * @name sakai.embedcontent
 24  *
 25  * @class embedcontent

 34 sakai.embedcontent = function(tuid, showSettings) {
 35 
 36     var $rootel = $("#" + tuid);
 37 
 38     var $embedcontent_main_container = $("#embedcontent_main_container", $rootel);
 39     var $embedcontent_settings = $("#embedcontent_settings", $rootel);
 40     var $embedcontent_page_name = $("#embedcontent_page_name", $rootel);
 41     var $embedcontent_place_content = $("#embedcontent_place_content", $rootel);
 42     var $embedcontent_cancel = $("#embedcontent_cancel", $rootel);
 43     var $embedcontent_content_input = $("#embedcontent_content_input", $rootel);
 44     var $embedcontent_display_options = $("#embedcontent_display_options", $rootel);
 45     var $embedcontent_display_options_select = $("#embedcontent_display_options_select", $rootel);
 46     var $embedcontent_metadata_container = $("#embedcontent_metadata_container", $rootel);
 47     var $embedcontent_metadata = $("#embedcontent_metadata", $rootel);
 48     var $embedcontent_search_for_content = $("#embedcontent_search_for_content", $rootel);
 49 
 50     var $embedcontent_alternative_display_name_value = $("#embedcontent_alternative_display_name_value", $rootel);
 51     var $embedcontent_description_value = $("#embedcontent_description_value", $rootel);
 52 
 53     var $embedcontent_page_name_template = $("#embedcontent_page_name_template", $rootel);
 54     var $embedcontent_display_options_select_template = $("#embedcontent_display_options_select_template", $rootel);
 55     var $embedcontent_content = $("#embedcontent_content", $rootel);

 61     var $uploadContentLink = $("#upload_content", $rootel);

 63     var selectedItems = [];
 64     var firstTime = true;
 65     var widgetData = false;
 66 
 67     var embedConfig = {
 68         "name": "Page",
 69         "limit": false,
 70         "filter": false,
 71         "type": "choose"
 72     };
 73 
 74     /**
 75      * Render the embed screen
 76      */
 77     var renderSettings = function() {
 78         selectedItems = [];
 79         $.TemplateRenderer($embedcontent_page_name_template, {"name": embedConfig.name}, $embedcontent_page_name);
 80         if (firstTime) {
 81             setupAutoSuggest();
 82             sakai.api.Widgets.widgetLoader.insertWidgets("embedcontent_settings", false, "#"+tuid);
 83             firstTime = false;
 84         } else {
 85             doReset();
 86         }
 87         $("#as-values-" + tuid).val("");
 88         $(".as-selection-item").remove();
 89         if (widgetData && widgetData.items && widgetData.items.length) {
 90             setCurrentFiles();
 91         }
 92     };
 93 
 94     var renderWidget = function() {
 95         $.TemplateRenderer($embedcontent_content_html_template, widgetData, $embedcontent_content);
 96         sakai.api.Widgets.widgetLoader.insertWidgets("embedcontent_main_container", false, "#"+tuid);
 97     };

 99     /**
100      * Do a reset of the embed screen
101      */
102     var doReset = function() {
103         $("#as-values-" + tuid).val("");
104         $(".as-selection-item").remove();
105         $embedcontent_display_options.hide();
106         $embedcontent_metadata_container.hide();
107         $embedcontent_display_options_select.find("option:selected").removeAttr("selected");
108         $embedcontent_display_options_select.find("#show_content_only").attr("selected", "selected");
109         $embedcontent_alternative_display_name_value.val('');
110         $embedcontent_description_value.val('');
111     };

113     /**
114      * Get the mimetype of a provided file
115      * @param {Object} file File provided to get mimetype of
116      */
117     var getMimeType = function(file) {
118         var mimetype = "";
119         mimetype = file["jcr:content"] ? file["jcr:content"]["jcr:mimeType"] : "";
120         return mimetype;
121     };
122 
123     /**
124      * Creates an object out of results provided
125      * This object contains valuable information about the file like path, name, type,...
126      * @param {Object} result results provided (eg through a search)
127      * @param {Object} name optional name provided
128      */
129     var createDataObject = function(result, name) {
130         var mimetype = getMimeType(result);
131         var dataObj = {
132             "value": name || result['jcr:name'],
133             "name": result['sakai:pooled-content-file-name'],
134             "type": "file",
135             "filetype": mimetype.split("/")[0],
136             "mimetype": mimetype,
137             "description": result["sakai:description"] || "",
138             "path": "/p/" + (name || result['jcr:name']),
139             "fileSize": sakai.api.Util.convertToHumanReadableFileSize(result["jcr:content"][":jcr:data"]),
140             "link": "/p/" + (name || result['jcr:name']) + "/" + result['sakai:pooled-content-file-name'],
141             "extension": result['sakai:fileextension']
142         };
143         return dataObj;
144     };
145 
146     /**
147      * When typing in the suggest box this function is executed to provide the user with a list of possible autocompletions
148      */
149     var setupAutoSuggest = function() {
150         $embedcontent_content_input.autoSuggest("",{
151             source: function(query, add) {
152                 searchUrl = sakai.config.URL.POOLED_CONTENT_MANAGER;
153                 sakai.api.Server.loadJSON(searchUrl.replace(".json", ""), function(success, data){
154                     if (success) {
155                         var suggestions = [];
156                         $.each(data.results, function(i) {
157                             var dataObj = createDataObject(data.results[i]);
158                             var doAdd = true;
159                             if (embedConfig.filter) {
160                                 if (dataObj.filetype !== embedConfig.filter) {
161                                     doAdd = false;
162                                 }
163                             }
164                             if (doAdd) {
165                                 suggestions.push(dataObj);
166                             }
167                         });
168                         add(suggestions);
169                     }
170                 }, {"q": "*" + query.replace(/\s+/g, "* OR *") + "*", "page": 0, "items": 15});
171             },
172             retrieveLimit: 10,
173             asHtmlID: tuid,
174             selectedItemProp: "name",
175             searchObjProps: "name",
176             selectionLimit: embedConfig.limit,
177             resultClick: function(data) {
178                 selectedItems.push(data.attributes);
179                 showDisplayOptions();
180                 $embedcontent_place_content.removeAttr("disabled");
181             },
182             selectionRemoved: function(elem) {
183                 removeItemFromSelected(elem.html().split("</a>")[1]); // get filename
184                 elem.remove();
185                 if (selectedItems.length === 0) {
186                     $embedcontent_place_content.attr("disabled", "disabled");
187                     $embedcontent_display_options.hide();
188                     $embedcontent_metadata_container.hide();
189                 }
190             }
191         });
192     };
193 
194     /**
195      * Removes a previously selected item from the list of selected items
196      * @param {Object} fileName name of the selected item to be removed from the list
197      */
198     var removeItemFromSelected = function(fileName) {
199         var newItems = [];
200         $(selectedItems).each(function(i, val) {
201            if (val.name !== fileName) {
202                newItems.push(val);
203            }
204         });
205         selectedItems = newItems;
206     };
207 
208     var setCurrentFiles = function() {
209         $.each(widgetData.items, function(i,val) {
210             selectedItems.push(val);
211             $embedcontent_content_input.autoSuggest.add_selected_item(val, val.value);
212         });
213         $(".as-original input.as-input").val('').focus();
214         $embedcontent_place_content.removeAttr("disabled");
215         showDisplayOptions();
216         $embedcontent_display_options_select.val(widgetData.embedmethod).trigger("change");
217         $embedcontent_alternative_display_name_value.val(widgetData.title);
218         $embedcontent_description_value.val(widgetData.description);
219     };

221     /**
222      * Called when file(s) are selected in the picker advanced widget and need to be added to the list of files that will be embedded.
223      * @param {Object} files Array of files selected in the picker advanced widget
224      */
225     var addChoicesFromPickeradvanced = function(files) {
226         var filesPicked = 0;
227         $.each(files, function(i,val) {
228             filesPicked++;
229         });
230         // revisit this next conditional -- right now it'll clear out all selections, not just add up to the limit
231         if (embedConfig.limit && filesPicked && ($(".as-selection-item").length + filesPicked) > embedConfig.limit) {
232             $("#as-values-" + tuid).val('');
233             $(".as-selection-item").remove();
234         }
235         if (filesPicked > 0) {
236             $embedcontent_place_content.removeAttr("disabled");
237         }
238         $.each(files, function(i,val) {
239             var newObj = createDataObject(val, val["jcr:name"]);
240             selectedItems.push(newObj);
241             $embedcontent_content_input.autoSuggest.add_selected_item(newObj, newObj.value);
242         });
243         $("input#" + tuid).val('').focus();
244     };
245 
246     /**
247      * Called when newly uploaded files need to be added to the list of files that will be embedded
248      * @param {Object} files Array containing a list of files
249      */
250     var addChoicesFromFileUpload = function(files) {
251       $.each(files, function(i,val) {
252           $.ajax({
253              url: val.url + ".infinity.json",
254              success: function(data) {
255                  var newObj = createDataObject(data, val.url.split("/p/")[1]);
256                  selectedItems.push(newObj);
257                  $embedcontent_content_input.autoSuggest.add_selected_item(newObj, newObj.value);
258              }
259           });
260       });
261       $("input#" + tuid).val('').focus();
262       $embedcontent_place_content.removeAttr("disabled");
263     };

265     /**
266      * Shows the options the user has for displaying the content
267      */
268     var showDisplayOptions = function() {
269         $embedcontent_display_options.show();
270     };
271 
272     /**
273      * Once the content has been placed on the page it has to be associated with the group
274      * The group is set as a viewer of the content
275      * @param {Object} embeddedItems Array of object containing information about the selected items. Only the path variable is used.
276      */
277     var associatedEmbeddedItemsWithGroup = function(embeddedItems){
278         var data = [];
279         for (var embeddedItem in embeddedItems) {
280             if (embeddedItems.hasOwnProperty(embeddedItem)) {
281                 var item = {
282                     "url": embeddedItems[embeddedItem].path + ".members.json",
283                     "method": "POST",
284                     "parameters": {
285                         ":viewer": sakai.currentgroup.id
286                     }
287                 };
288                 data[data.length] = item;
289             }
290         }
291 
292         $.ajax({
293             url: sakai.config.URL.BATCH,
294             traditional: true,
295             type: "POST",
296             cache: false,
297             data: {
298                 requests: $.toJSON(data)
299             }
300         });
301     };
302 
303     var registerVideo = function(videoBatchData){
304         $.ajax({
305             url: sakai.config.URL.BATCH,
306             traditional: true,
307             type: "POST",
308             cache: false,
309             data: {
310                 requests: $.toJSON(videoBatchData)
311             }
312         });
313     };
314 
315     /**
316      * Embed the selected content on the page,
317      * Call the function that associates the content with this group
318      */
319     var doEmbed = function() {
320         var embedContentHTML = "";
321         var objectData = {
322             "embedmethod": $embedcontent_display_options_select.find("option:selected").val(),
323             "title": $embedcontent_alternative_display_name_value.val(),
324             "description": $embedcontent_description_value.val(),
325             "items": selectedItems
326         };
327 
328         var videoBatchData = [];
329         for (var i in objectData.items){
330             if(objectData.items.hasOwnProperty(i)){
331                 if(objectData.items[i].filetype === "video"){
332                     // Set random ID to the video
333                     objectData.items[i].uId = Math.ceil(Math.random() * 999999999);
334 
335                     var itemUrl;
336                     if (sakai.currentgroup.data.authprofile) {
337                         itemUrl = "/~" + sakai.currentgroup.data.authprofile["sakai:group-title"] + "/pages/_widgets/id" + objectData.items[i].uId + "/video";
338                     } else {
339                         itemUrl = "/~" + sakai.data.me.user.userid + "/pages/_widgets/id" + objectData.items[i].uId + "/video";
340                     }
341 
342                     // Create batch request data for the video
343                     var item = {
344                         "url": itemUrl,
345                         "method": "POST",
346                         "parameters": {
347                             "uid": sakai.data.me.user.userid,
348                             "source": " ",
349                             "URL": sakai.config.SakaiDomain + objectData.items[i].link + objectData.items[i].extension,
350                             "selectedvalue": "video_noSource",
351                             "isYoutube": false,
352                             "isSakaiVideoPlayer": false
353                         }
354                     };
355                     videoBatchData.push(item);
356                 }
357             }
358         }
359 
360         // Associate embedded items with the group
361         associatedEmbeddedItemsWithGroup(selectedItems);
362 
363         registerVideo(videoBatchData);
364 
365         if ($embedcontent_metadata_container.is(":visible")) {
366             var isValid = $embedcontent_metadata.valid();
367             if (isValid) {
368                 saveWidgetData(objectData);
369             }
370         } else {
371             saveWidgetData(objectData);
372         }
373 
374     };
375 
376     var saveWidgetData = function(data) {
377         sakai.api.Widgets.saveWidgetData(tuid, data, function() {
378             sakai.api.Widgets.Container.informFinish(tuid, "embedcontent");
379         });
380     };
381 
382     var getWidgetData = function(callback) {
383         sakai.api.Widgets.loadWidgetData(tuid, function(success, data) {
384             if (success) {
385                 widgetData = data;
386             }
387 
388             if ($.isFunction(callback)) {
389                 callback();
390             }
391         });
392     };

428     $(window).unbind("sakai-pickeradvanced-ready");

431             var pickerConfig = {

434             if (embedConfig.limit) {
435                 pickerConfig.limit = embedConfig.limit;
436             }
437             $(window).trigger("sakai-pickeradvanced-init", {"config": pickerConfig});
438         });

441     $embedcontent_metadata.validate();
442 
443     var doInit = function() {
444         getWidgetData(function() {
445             if (showSettings) {
446                 embedConfig.name = sakai.sitespages.site_info._pages[sakai.sitespages.selectedpage]["pageTitle"];
447                 renderSettings();
448                 $embedcontent_settings.show();
449             } else {
450                 $embedcontent_main_container.show();
451                 renderWidget();
452             }
453         });
454     };
455 
456     doInit();
