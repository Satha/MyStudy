 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.filerevisions
 25  *
 26  * @param {String} tuid Unique id of the widget
 27  * @param {Boolean} showSettings Show the settings of the widget or not
 28  */
 29 sakai.filerevisions = function(tuid, showSettings){
 30 
 31 
 32     ///////////////
 33     // Variables //
 34     ///////////////
 35 
 36     var $fileRevisionsDialog = $("#filerevisions_dialog");
 37     var baseFileData = {};
 38 
 39     // Templates
 40     var filerevisionsTemplate = "#filerevisions_template";
 41 
 42     // Containers
 43     var filerevisionsTemplateContainer = "#filerevisions_template_container";
 44 
 45     // Buttons
 46     var filerevisionsCloseButton = "#filerevisions_close";
 47 
 48     /**
 49      * Render the template that displays all revision information
 50      */
 51     var renderRevisionData = function(){
 52         var data = [];
 53         baseFileData.created = sakai.api.l10n.transformDate(new Date(baseFileData.data["jcr:created"]));
 54         data.data = baseFileData;
 55         data.linkrevision = $("#content_profile_details_view_revisions").hasClass("link_revision");
 56 
 57         var renderedTemplate = $.TemplateRenderer(filerevisionsTemplate, data);
 58         $(filerevisionsTemplateContainer).html(renderedTemplate);
 59 
 60         renderedTemplate = $.TemplateRenderer("#filerevision_header_text_template", data);
 61         $("#filerevision_header_text").html(renderedTemplate);
 62     };
 63 
 64     /**
 65      * Get userprofile with the userid provided
 66      * @param {Object} userid
 67      */
 68     var getUserProfile = function(userid){
 69         $.ajax({
 70             url: "/~" + userid + "/public/authprofile.infinity.json",
 71             success: function(profile){
 72                 baseFileData["sakai:savedByFull"] = sakai.api.User.getDisplayName(profile);
 73                 renderRevisionData();
 74             },
 75             error: function(xhr, textStatus, thrownError){
 76                 sakai.api.Util.notification.show("Failed loading profile", "Failed to load file profile information");
 77             }
 78         });
 79     };
 80 
 81     /**
 82      * Get detailed information on the files in the revision history
 83      * This data contains path, mimetype, description,...
 84      */
 85     var getRevisionInformationDetails = function(){
 86         var revisionInformationDetails = [];
 87         for (var i in baseFileData.revisions) {
 88             if (baseFileData.revisions.hasOwnProperty(i)) {
 89                 var item = {
 90                     "url": baseFileData.path + ".version.," + baseFileData.revisions[i]["jcr:name"] + ",.json",
 91                     "method": "GET"
 92                 };
 93                 revisionInformationDetails[revisionInformationDetails.length] = item;
 94             }
 95         }
 96         baseFileData.numberOfRevisions = revisionInformationDetails.length;
 97         // Do the Batch request
 98         $.ajax({
 99             url: sakai.config.URL.BATCH,
100 
101             type : "POST",
102             dataType: 'json',
103             cache: false,
104             data: {
105                 requests: $.toJSON(revisionInformationDetails)
106             },
107             success: function(data){
108                 // Get file information out of results
109                 var revisionFileDetails = [];
110                 for (var i in data.results){
111                     if (data.results.hasOwnProperty(i)) {
112                         revisionFileDetails.push($.parseJSON(data.results[i].body));
113                     }
114                 }
115                 baseFileData.revisionFileDetails = revisionFileDetails;
116                 if (baseFileData.data["sakai:savedBy"]) {
117                     getUserProfile(baseFileData.data["sakai:savedBy"]);
118                 } else {
119                     getUserProfile(baseFileData.data["sakai:pool-content-created-for"]);
120                 }
121             },
122             error: function(xhr, textStatus, thrownError){
123 
124             }
125         });
126     };
127 
128     /**
129      * Get the revision information on the specified file
130      */
131     var getRevisionInformation = function(){
132         $.ajax({
133             url: baseFileData.path + ".versions.json",
134             type : "GET",
135             success: function(data){
136                 var versions = [];
137                 for (var i in data.versions){
138                     if (data.versions.hasOwnProperty(i)) {
139                         var splitDate = data.versions[i]["jcr:created"].split("T")[0].split("-");
140                         data.versions[i]["jcr:created"] = sakai.api.l10n.transformDate(new Date(splitDate[0],splitDate[1]-1,splitDate[2]));
141                         versions.push(data.versions[i]);
142                     }
143                 }
144 
145                 baseFileData.revisions = versions.reverse();
146                 getRevisionInformationDetails();
147             },
148             error: function(){
149                 sakai.api.Util.notification.show("Revision information not retrieved", "The revision information for the file could not be retrieved");
150             }
151         });
152     };
153 
154     /**
155      * Load the file revisions widget with the appropriate data
156      * This function can be called from anywhere within Sakai
157      * @param {Object} data Contains data needed to call all revisions for the file
158      */
159     sakai.filerevisions.initialise = function(data){
160         baseFileData = data;
161         getRevisionInformation();
162         $fileRevisionsDialog.jqmShow();
163     };
164 
165     $fileRevisionsDialog.jqm({
166         modal: true,
167         overlay: 20,
168         toTop: true
169     });
170 
171     $(filerevisionsCloseButton).live("click", function(){
172         $(filerevisionsTemplateContainer).html("");
173         $fileRevisionsDialog.jqmHide();
174     });
175 
176 };
177 sakai.api.Widgets.widgetLoader.informOnLoad("filerevisions");
