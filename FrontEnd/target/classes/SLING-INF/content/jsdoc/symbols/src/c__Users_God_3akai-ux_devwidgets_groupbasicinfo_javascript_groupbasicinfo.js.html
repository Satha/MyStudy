 89     /**
 90      * Get a list of nodes representing the directory structure to be rendered
 91      */
 92     var getDirectoryStructure = function(){
 93         // Get directory structure from config file
 94         for(var i in sakai.config.Directory){
 95             if (sakai.config.Directory.hasOwnProperty(i)) {
 96                 // Create first level of content
 97                 var temp = {};
 98                 temp.name = i;
 99 
100                 // Create second level of content
101                 temp.secondlevels = [];
102                 for (var j in sakai.config.Directory[i]) {
103                     if (sakai.config.Directory[i].hasOwnProperty(j)) {
104                         var secondlevel = {};
105                         secondlevel.name = j;
106 
107                         // Create third level of content
108                         secondlevel.thirdlevels = [];
109                         for (var k in sakai.config.Directory[i][j]) {
110                             if (sakai.config.Directory[i][j].hasOwnProperty(k)) {
111                                 var thirdlevel = {};
112                                 thirdlevel.name = sakai.config.Directory[i][j][k];
113                                 secondlevel.thirdlevels.push(thirdlevel);
114                             }
115                         }
116 
117                         temp.secondlevels.push(secondlevel);
118                     }
119                 }
120                 directoryJSON.push(temp);
121             }
122         }
123     };

175     /**
176      * Render the template for group basic info
177      */
178     var renderTemplateBasicInfo = function(){
179 
180         var mode = '';
181         // Show in Edit mode
182         if (showSettings) {
183             mode = 'edit';
184         }
185         // Extract tags that start with "directory:"
186         var directory = [];
187         var tags = [];
188         $(sakai.currentgroup.data.authprofile["sakai:tags"]).each(function(i){
189             var splitDir = sakai.currentgroup.data.authprofile["sakai:tags"][i].split("/");
190             if (splitDir[0] === "directory") {
191                 var item = [];
192                 for (var j in splitDir) {
193                     if (splitDir.hasOwnProperty(j)) {
194                         if (splitDir[j] !== "directory") {
195                             item.push(splitDir[j]);
196                         }
197                     }
198                 }
199                 directory.push(item);
200             } else {
201                 tags.push(sakai.currentgroup.data.authprofile["sakai:tags"][i]);
202             }
203         });
204         // Get the group information out of the global group info object
205         json = {
206             "groupid" : sakai.currentgroup.id,
207             "url" : document.location.protocol + "//" + document.location.host + "/~" + sakai.currentgroup.id,
208             "data" : sakai.currentgroup.data.authprofile,
209             "mode" : mode,
210             "tags" : tags,
211             "directory" : directoryJSON,
212             "saveddirectory" : directory,
213             /* the following perSectionPermissions switch is used to turn off
214                the "Who can view or search this?" permissions dropdown for now.
215                The dropdown will need to be enabled and fully implemented later
216                on and the following switch can be removed. */
217             "perSectionPermissions" : false
218         };
219 
220         $groupbasicinfo_generalinfo.html($.TemplateRenderer("#groupbasicinfo_default_template", json));
221 
222         if (mode === "edit") {
223             addBinding();
224         }
225 
226         // If this widget is not shown on the group profile (i.e. when rendered inside a page)
227         // we show the widget's own update button
228         if (!sakai.currentgroup.profileView){
229             $(groupbasicinfo_buttons, $rootel).show();
230         }
231 
232     };

254     /**
255      * Update group data
256      */
257     var updateGroup = function(){
258 
259         // --- validate data ---
260 
261         // group title (cannot be blank)
262         var groupTitle = $.trim($(groupBasicInfoGroupTitle, $rootel).val());
263         if (!groupTitle) {
264             groupTitle = sakai.currentgroup.data.authprofile["sakai:group-title"];
265         }
266 
267         // group kind (should be preset values)
268         var groupKind = $(groupBasicInfoGroupKind, $rootel).val();
269 
270         // group tags
271         var currentTags = [];
272         if (sakai.currentgroup.data.authprofile["sakai:tags"]) {
273             currentTags = sakai.currentgroup.data.authprofile["sakai:tags"].slice(0);
274         }
275         var enteredTags = $.trim($(groupBasicInfoGroupTags).val()).split(",");
276         if (enteredTags !== currentTags) {
277             // user has changed tags
278             sakai.currentgroup.data.authprofile["sakai:tags"] = [];
279             $(enteredTags).each(function(i, tag) {
280                 tag = $.trim(tag).replace(/#/g,"");
281                 if (sakai.api.Security.escapeHTML(tag) === tag && tag.replace(/\\/g,"").length) {
282                     if ($.inArray(tag, sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
283                         sakai.currentgroup.data.authprofile["sakai:tags"].push(tag.replace(/\\/g,""));
284                     }
285                 }
286             });
287         }
288 
289         // Create tags for the directory structure
290         // For every groupbasicinfo_added_directory we create tags
291         $(".groupbasicinfo_added_directory").each(function(){
292             if ($(this).find(groupBasicInfoDirectoryLvlOne).selected().val() !== "no_value") {
293                 var directoryString = "directory/";
294                 if ($.inArray($(this).find(groupBasicInfoDirectoryLvlOne).selected().val().replace(/,/g, ""), sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
295                     sakai.currentgroup.data.authprofile["sakai:tags"].push($(this).find(groupBasicInfoDirectoryLvlOne).selected().val().replace(/,/g, ""));
296                 }
297                 directoryString += $(this).find(groupBasicInfoDirectoryLvlOne).selected().val().replace(/,/g, "");
298 
299                 if ($(this).find(groupBasicInfoDirectoryLvlTwo).selected().val() !== "no_value") {
300                     if ($.inArray($(this).find(groupBasicInfoDirectoryLvlTwo).selected().val().replace(/,/g, ""), sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
301                         sakai.currentgroup.data.authprofile["sakai:tags"].push($(this).find(groupBasicInfoDirectoryLvlTwo).selected().val().replace(/,/g, ""));
302                     }
303                     directoryString += "/" + $(this).find(groupBasicInfoDirectoryLvlTwo).selected().val().replace(/,/g, "");
304 
305                     if ($(this).find(groupBasicInfoDirectoryLvlThree).selected().val() !== "no_value") {
306                         if ($.inArray($(this).find(groupBasicInfoDirectoryLvlThree).selected().val().replace(/,/g, ""), sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
307                             sakai.currentgroup.data.authprofile["sakai:tags"].push($(this).find(groupBasicInfoDirectoryLvlThree).selected().val().replace(/,/g, ""));
308                         }
309                         directoryString += "/" + $(this).find(groupBasicInfoDirectoryLvlThree).selected().val().replace(/,/g, "");
310                     }
311 
312                 }
313 
314                 // Add string for all levels to tag array
315                 if ($.inArray(directoryString, sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
316                     sakai.currentgroup.data.authprofile["sakai:tags"].push(directoryString);
317                 }
318             }
319         });
320 
321         // Add the directory tags to the array that were already saved
322         $(groupBasicInfoSavedInfo + " li").each(function(){
323             var splitValues = this.className.split(",");
324             var savedDirString = "directory/" + splitValues[0];
325             if(splitValues.length > 1){
326                 savedDirString += "/" + splitValues[1];
327                 if(splitValues.length > 2){
328                     savedDirString += "/" + splitValues[2];
329                 }
330             }
331             if ($.inArray(savedDirString, sakai.currentgroup.data.authprofile["sakai:tags"]) < 0) {
332                 sakai.currentgroup.data.authprofile["sakai:tags"].push(savedDirString);
333             }
334         });
335 
336         // group description (can be blank)
337         var groupDesc = $.trim($(groupBasicInfoGroupDesc, $rootel).val());
338 
339         // group permissions settings
340         var joinable = $(groupBasicInfoGroupJoinable).val();
341         var visible = $(groupBasicInfoGroupVisible).val();
342         if(joinable !== sakai.currentgroup.data.authprofile["sakai:group-joinable"] ||
343             visible !== sakai.currentgroup.data.authprofile["sakai:group-visible"]) {
344             // only POST if user has changed values
345             sakai.currentgroup.data.authprofile["sakai:group-joinable"] = joinable;
346             sakai.currentgroup.data.authprofile["sakai:group-visible"] = visible;
347             sakai.api.Groups.setPermissions(sakai.currentgroup.id, joinable, visible);
348         }
349 
350         // Update the group object
351         sakai.currentgroup.data.authprofile["sakai:group-title"] = sakai.api.Security.escapeHTML(groupTitle);
352         sakai.currentgroup.data.authprofile["sakai:group-kind"] = groupKind;
353         sakai.currentgroup.data.authprofile["sakai:group-description"] = sakai.api.Security.escapeHTML(groupDesc);
354         groupProfileURL = "/~" + sakai.currentgroup.id + "/public/authprofile";
355 
356         $.ajax({
357             url: groupProfileURL,
358             data: {
359                 "_charset_":"utf-8",
360                 "sakai:group-title" : groupTitle,
361                 "sakai:group-kind" : groupKind,
362                 "sakai:group-description" : groupDesc
363             },
364             type: "POST",
365             success: function(data, textStatus){
366                 sakai.api.Util.tagEntity(groupProfileURL, sakai.currentgroup.data.authprofile["sakai:tags"], currentTags, function() {
367                     sakai.currentgroup.data.authprofile["sakai:tags"] = currentTags;
368                 });
369             },
370             error: function(xhr, textStatus, thrownError){
371                 debug.error("ERROR: groupbasicinfo.js/updateGroup() unable to set group information. Status: " + xhr.status + " - " + xhr.statusText);
372             },
373             complete: function() {
374                 sakai.api.Widgets.Container.informFinish(tuid, "groupbasicinfo");
375                 $(window).trigger("sakai.groupbasicinfo.updateFinished");
376             }
377         });
378     };

390     /**
391      * Update the select boxes on the stage
392      * @param {String} select Containing ID to check which box value has been changed
393      * @param {String} changedboxvalue Containing selected value
394      * @param {String} firstlevelvalue Containing value of first select box
395      */
396     var updateDirectoryDisplay = function(select, changedboxvalue, firstlevelvalue){
397         var obj = {
398             "firstlevelvalue":firstlevelvalue.selected().val(),
399             "changedboxvalue" : changedboxvalue.selected().val(),
400             "directory": directoryJSON
401         };
402         if(select === groupBasicInfoDirectoryLvlTwo){
403             $(firstlevelvalue.parent().children("#groupbasicinfo_secondlevel_template_container")).html($.TemplateRenderer(groupBasicInfoSecondLevelTemplate, obj));
404         }else{
405             $(firstlevelvalue.parent().children("#groupbasicinfo_thirdlevel_template_container")).html($.TemplateRenderer(groupBasicInfoThirdLevelTemplate, obj));
406         }
407     };

409     var removeDirectoryLocation = function(clickedParent){
410         // Look for group profile path
411         groupProfileURL = "/~" + sakai.currentgroup.id + "/public/authprofile";
412         // Extract tags from clickedParent
413         var tags = [];
414         tags = clickedParent[0].className.split(",");
415         tags.push("directory/" + tags.toString().replace(/,/g,"/"));
416 
417         var tagsAfterDeletion = sakai.currentgroup.data.authprofile["sakai:tags"].slice(0);
418         for (var tag in tags){
419             if (tags.hasOwnProperty(tag)) {
420                 if($.inArray(tags[tag],tagsAfterDeletion) > -1) {
421                     tagsAfterDeletion.splice($.inArray(tags[tag],tagsAfterDeletion), 1);
422                 }
423             }
424         }
425 
426         sakai.api.Util.tagEntity(groupProfileURL, tagsAfterDeletion, sakai.currentgroup.data.authprofile["sakai:tags"], function(){
427             sakai.currentgroup.data.authprofile["sakai:tags"].splice(tags);
428             //sakai.api.Widgets.Container.informFinish(tuid, "groupbasicinfo");
429             $(window).trigger("sakai.groupbasicinfo.updateFinished");
430         });
431

439     /**

449         // disable all buttons

451     };

454      * Enable all Group Basic Info input elements (i.e. after processing an update)
455      */

457         // enable all input elements
458         $("#groupbasicinfo_generalinfo input", $rootel).removeAttr("disabled");
459         // enable all textarea elements
460         $("#groupbasicinfo_generalinfo textarea", $rootel).removeAttr("disabled");
461         // enable all select dropdowns
462         $("#groupbasicinfo_generalinfo select", $rootel).removeAttr("disabled");
463         // enable all buttons
464         $("#groupbasicinfo_generalinfo button", $rootel).removeAttr("disabled");
465     };
