 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.helloworld
 25  *
 26  * @class helloworld
 27  *
 28  * @description
 29  * Initialize the helloworld widget
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.helloworld = function(tuid,showSettings){
 36 
 37 
 38     /////////////////////////////
 39     // Configuration variables //
 40     /////////////////////////////
 41 
 42     var defaultColor = "#000000";
 43 
 44     // Dom identifiers
 45     var rootel = $("#" + tuid);
 46     var settingsContainer = "#helloworld_settings";
 47     var viewContainer = "#helloworld_main";
 48     var colorPicker = "#helloworld_color";
 49     var usernameContainer = "#helloworld_username";
 50     var seaveHelloworld = "#helloworld_save";
 51 
 52 
 53     ////////////////////
 54     // Main functions //
 55     ////////////////////
 56 
 57     /**
 58      * Shows the Hello world string in the right color
 59      * @param {Object} color
 60      */
 61     var showHelloWorld = function(color){
 62         $(viewContainer + " p", rootel).css("color",color.color);
 63         $(viewContainer, rootel).show();
 64     };
 65 
 66 
 67     ////////////////////////
 68     // Settings functions //
 69     ////////////////////////
 70 
 71     /**
 72      * Selects the current color form the combobox
 73      * @param {Object} color
 74      */
 75     var selectCurrentColor = function(color){
 76         $(colorPicker,rootel).val(color.color);
 77     };
 78 
 79 
 80     ////////////////////
 81     // Event Handlers //
 82     ////////////////////
 83 
 84     /** Binds the helloworld save button*/
 85     $("#helloworld_save").bind("click", function(ev){
 86         var select = $(colorPicker, rootel).get(0);
 87         var selected = select.options[select.selectedIndex].value;
 88         sakai.api.Widgets.saveWidgetData(tuid, {color:selected}, function(success, data){
 89             sakai.api.Widgets.Container.informFinish(tuid, "helloworld");
 90         });
 91     });
 92 
 93 
 94     /////////////////////////////
 95     // Initialisation function //
 96     /////////////////////////////
 97 
 98     /**
 99      * Retrieves the prefered color from JCR
100      * @param {Object} callback
101      */
102     var getPreferedColor = function(callback){
103         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
104             if (success) {
105                 callback(data);
106             } else {
107                 callback({
108                     "color": defaultColor
109                 });
110             }
111 
112         });
113 
114     };
115 
116     var doInit = function(){
117         if (showSettings) {
118             getPreferedColor(selectCurrentColor);
119             $(settingsContainer, rootel).show();
120         } else {
121             var me = sakai.data.me;
122             $(usernameContainer, rootel).text(sakai.api.Security.saneHTML(sakai.api.User.getProfileBasicElementValue(me.profile, "firstName")));
123             getPreferedColor(showHelloWorld);
124         }
125     };
126     doInit();
127 
128 };
129 
130 sakai.api.Widgets.widgetLoader.informOnLoad("helloworld");
