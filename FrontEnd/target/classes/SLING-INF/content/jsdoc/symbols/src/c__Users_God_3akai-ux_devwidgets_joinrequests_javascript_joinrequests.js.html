 18 /*global $ */
 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.joinrequests
 24  *
 25  * @class joinrequests
 26  *
 27  * @description
 28  * The Join Requests widget manages join requests for a specific group. Its
 29  * primary placement is at the top of group_edit.html
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.joinrequests = function (tuid, showSettings) {
 36 
 37     /////////////////////////////
 38     // Configuration variables //
 39     /////////////////////////////
 40 
 41     var numJoinrequests = 0;  // keeps track of the total number of requests
 42 
 43     // DOM elements
 44     var $rootel = $("#" + tuid);
 45     var $joinrequestsWidget = $(".joinrequests_widget", $rootel);
 46     var $joinrequests = $("#joinrequests_joinrequests", $rootel);
 47     var $joinrequestsTitle = $("#joinrequests_title", $rootel);
 48     var $joinrequestsError = $("#joinrequests_error", $rootel);
 49     var $joinrequestsSuccess = $("#joinrequests_success", $rootel);
 50     var $joinrequestsTemplate = $("#joinrequests_template", $rootel);
 51     var $addLink = $("a.joinrequests_add_link", $rootel);
 52     var $ignoreLink = $("a.joinrequests_ignore_link", $rootel);
 53 
 54 
 55     /**
 56      * Renders the joinrequests widget with the given joinrequests array
 57      *
 58      * @param {Array} joinrequests Array of joinrequest Objects to display,
 59      * formatted for the UI.  Each joinrequest Object should contain:
 60      * {
 61      *    "userid": <userid>,
 62      *    "firstName": <user's first name>,
 63      *    "lastName": <user's last name>,
 64      *    "request_age": <request create-date, JS Date.toLocaleString() value>
 65      * }
 66      */
 67     var renderJoinRequests = function (joinrequests) {
 68         if (joinrequests) {
 69             // populate template with data
 70             var json = {
 71                 "joinrequests": joinrequests
 72             };
 73             $joinrequests.html($.TemplateRenderer($joinrequestsTemplate, json));
 74             // set images for users that have a profile picture
 75             for (var i in joinrequests) {
 76                 if (joinrequests.hasOwnProperty(i)) {
 77                     var pic_src = "/dev/images/default_profile_picture_64.png";
 78                     if (joinrequests[i].pic_src) {
 79                         var pic_src_json = $.parseJSON(joinrequests[i].pic_src);
 80                         pic_src = "/~" + joinrequests[i].userid +
 81                             "/public/profile/" + pic_src_json.name;
 82                     }
 83                     $("#joinrequests_userpicture_" + joinrequests[i].userid).attr(
 84                         "src", pic_src
 85                     );
 86                 }
 87             }
 88             // show the widget
 89             $joinrequestsWidget.show();
 90         }
 91     };
 92 
 93 
 94     /**
 95      * Returns a human readable date and time based on the given jcr:created
 96      * timestamp. WARNING--THIS FUNCTION IS NOT SAFE TO USE IN ALL BROWSERS
 97      *
 98      * @param {String} jcr_created jcr:created timestamp to convert
 99      * @return {String} human readable date
100      */
101     var getReadableRequestAge = function (jcr_created) {
102         if (jcr_created && typeof(jcr_created) === "string") {
103             var date = new Date(jcr_created);
104             if (date) {
105                 return date.toLocaleString();
106             } else {
107                 return jcr_created;
108             }
109         } else {
110             // not sure what this is - just send it back
111             return jcr_created;
112         }
113     };
114 
115 
116     /**
117      * Fetches join requests from the server
118      */
119     var getJoinRequestsData = function () {
120         // get join requests from server
121         sakai.api.Groups.getJoinRequests(sakai.currentgroup.id, function (success, data) {
122             if (success) {
123                 // process joinrequest data for UI
124                 if (data && data.total && data.total > 0) {
125                     numJoinrequests = data.total;
126                     var joinrequests = [];
127                     for (var i in data.results) {
128                         if (data.results.hasOwnProperty(i)) {
129                             var jr = data.results[i];
130                             joinrequests.push({
131                                 "userid": jr.userid,
132                                 "firstName": jr.basic.elements.firstName.value,
133                                 "lastName": jr.basic.elements.lastName.value,
134                                 "request_age": $.timeago(jr["jcr:created"]),
135                                 "pic_src": jr.picture
136                             });
137                         }
138                     }
139                     renderJoinRequests(joinrequests);
140                 }
141             }
142         });
143     };
144 
145 
146     /**
147      * Adds a user to the current group
148      *
149      * @param {String} userid The ID of the user to add to the current group
150      */
151     var addUser = function (userid) {
152         // add user to group
153         sakai.api.Groups.addUsersToGroup(sakai.currentgroup.id, "members", [userid], function (success) {
154             if (success) {
155                 // show notification
156                 sakai.api.Util.notification.show($joinrequestsTitle.html(),
157                     $("#joinrequests_username_link_" + userid).html() +
158                     " "+$joinrequestsSuccess.html());
159 
160                 // trigger the member list on group_edit.html to refresh
161                 $(window).trigger("sakai-listpeople-ready", "members");
162 
163                 // remove join request from UI and server
164                 removeJoinRequest(userid);
165             } else {
166                 sakai.api.Util.notification.show($joinrequestsTitle.html(), $joinrequestsError.html());
167             }
168         });
169     };
170 
171 
172     /**
173      * Removes a join request from the UI and the server
174      *
175      * @param {String} userid The ID of the user whose join request to remove
176      */
177     var removeJoinRequest = function (userid) {
178         // remove join request from server
179         sakai.api.Groups.removeJoinRequest(userid, sakai.currentgroup.id,
180         function (success) {
181             if (success) {
182                 // remove the UI joinrequest element
183                 $("#joinrequests_joinrequest_" + userid, $rootel).remove();
184                 if (--numJoinrequests === 0) {
185                     $joinrequestsWidget.hide();
186                 }
187             } else {
188                 sakai.api.Util.notification.show($joinrequestsTitle.html(), $joinrequestsError.html());
189                 hideSpinner(userid);
190             }
191         });
192     };
193 
194 
195     /**
196      * Shows an AJAX loading spinner while the user's join request is processed
197      *
198      * @param {String} userid The ID of the user whose join request is being processed
199      */
200     var showSpinner = function (userid) {
201         $("#joinrequests_actions_" + userid).hide();
202         $("#joinrequests_loading_" + userid).show();
203     };
204 
205 
206     /**
207      * Hides the AJAX loading spinner once the user's join request is processed
208      *
209      * @param {String} userid The ID of the user whose join request has been processed
210      */
211     var hideSpinner = function (userid) {
212         $("#joinrequests_loading_" + userid).hide();
213         $("#joinrequests_actions_" + userid).show();
214     };
215 
216 
217     /////////////////////////////
218     // Event Bindings          //
219     /////////////////////////////
220 
221     // Add the specific user when the 'Add as a member' link is clicked
222     $addLink.live("click", function () {
223         var userid = this.id.split("_")[2];
224         showSpinner(userid);
225         addUser(userid);
226     });
227 
228     // Ignore the specific user when the 'Ignore' link is clicked
229     $ignoreLink.live("click", function () {
230         var userid = this.id.split("_")[2];
231         showSpinner(userid);
232         removeJoinRequest(userid);
233     });
234 
235 
236     /////////////////////////////
237     // Initialization          //
238     /////////////////////////////
239 
240     /**
241      * Initialization function run when the widget loads
242      */
243     var init = function () {
244         // get join request data
245         getJoinRequestsData();
246     };
247 
248     init();
249 };
250 
251 sakai.api.Widgets.widgetLoader.informOnLoad("joinrequests");
