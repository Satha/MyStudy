 96     /**
 97      * Reset
 98      * Resets the people picker to a default state
 99      * @returns void

101     var reset = function() {
102         $pickeradvanced_content_search.html("");
103         $pickeradvanced_content_search.unbind("scroll");
104         pickerData.selected = {};
105         pickerData.currentElementCount = 0;
106         pickerData.selectCount = 0;

109     /**
110      * Render
111      * Renders the people picker
112      * @param iConfig {String} Config element for the widget
113      * @returns void
114      */
115     var render = function(iConfig) {
116         pickerData["searchIn"] = sakai.config.URL.SEARCH_USERS_ACCEPTED + "?page=0&items=12&_=&q=";
117 
118         // Merge user defined config with default
119         for (var element in iConfig) {
120             if (iConfig.hasOwnProperty(element) && pickerData.hasOwnProperty(element)) {
121                 pickerData[element] = iConfig[element];
122             }
123         }
124 
125         // display the groups list, bind elements and submit a search
126         $pickeradvanced_search_titles.hide();
127         $(".pickeradvanced_selected_list").removeClass("pickeradvanced_selected_list"); // deselect anything that was selected
128         if (pickerData["type"] === "people") {
129             $("#pickeradvanced_search_contacts").parent("li").addClass("pickeradvanced_selected_list");
130             getGroups();
131             $pickeradvanced_sort_on.show();
132             $pickeradvanced_search_people.show();
133         } else if (pickerData["type"] === "content") {
134             $("#pickeradvanced_search_files_mine").parent("li").addClass("pickeradvanced_selected_list");
135             $pickeradvanced_sort_on.hide();
136             $pickeradvanced_search_files.show();
137             pickerData["searchIn"] = sakai.config.URL.SEARCH_ALL_FILES.replace(".json", ".infinity.json") + "?page=0&items=12&_=&q=";
138         }
139         $("ul.pickeradvanced_search_" + pickerData["type"]).show();
140         $pickeradvanced_search_query.focus();
141         $pickeradvanced_search_button.click(submitSearch);
142         $pickeradvanced_content_search_form.submit(submitSearch);
143         $pickeradvanced_add_button.unbind("click");
144         $pickeradvanced_add_button.bind("click", function(){
145             addPeople();
146         });
147         submitSearch();
148     };

154         $pickeradvanced_group_specific_filters.html('');
155         $(sakai.data.me.groups).each(function(i,val) {
156             var groupItem = $.TemplateRenderer($pickeradvanced_group_search_template, {"data":val});
157             $pickeradvanced_group_specific_filters.append(groupItem);
158         });

161     /**
162      * submitSearch
163      * Perform the search based on the value in the $pickeradvanced_search_query text input box
164      * @returns void
165      */
166     var submitSearch = function(){
167         reset();
168         var searchQuery = $pickeradvanced_search_query.val();
169         if (!searchQuery) {
170             searchQuery = "*";
171         } else {
172             searchQuery = $.trim(searchQuery);
173             searchQuery = searchQuery.replace(/\s+/g, "* OR *");
174             searchQuery = "*" + searchQuery + "*";
175         }
176 
177         var pl_query = pickerData["searchIn"] + searchQuery + "&page=0&items=12&_=" + (Math.random() * 100000000000000000);
178         renderSearch(pl_query);
179     };
180 
181     /**
182      * RenderSearch
183      * Renders the people picker with a specified set of data. The function uses
184      * a search query initially, then does the paginating and subsequent requests
185      * for data automatically
186      * @param iSearchQuery {String} A Sakai search query
187      * @returns void
188      */
189     var renderSearch = function(iSearchQuery) {
190         // Parse search query
191         var searchQuery = {};
192         var main_parts = iSearchQuery.split("?");
193         searchQuery.url = main_parts[0];
194         var args = main_parts[1].split("&");
195         for (var i=0, il = args.length; i < il; i++) {
196             var kv_pair = args[i].split("=");
197             searchQuery[kv_pair[0]] = kv_pair[1];
198         }
199 
200         // Alter search query according to config
201         searchQuery.items = pickerData.items;
202 
203         // Add hash to search query in case it's not there to prevent caching
204         if (!searchQuery["_"]) {
205             searchQuery["_"] = (Math.random() * 100000000000000000);
206         }
207 
208         // Render the first page of results
209         addPage(0, searchQuery);
210 
211     };
212 
213     /**
214      * addPage
215      * Adds another page of search result to the picker's result list
216      * @pageNumber {Int} The page we want to load
217      * @searchQuery {Object} An object containing the search query elements
218      * @returns void
219      */
220     var addPage = function(pageNumber, searchQuery) {
221         // Create new container for the bit we load. This is then appended to the
222         // main container
223         var $pl_pageContainer = $("<ul id=\"pickeradvanced_page_" + pageNumber + "\" class=\"pickeradvanced_page pickeradvanced_page_list loadinganim\"></ul>");
224 
225         // Aadd relevant config elements to the search query
226         searchQuery.page = pageNumber;
227         if ((pickerData["type"] === "people" && $pickeradvanced_sort_on.is(":visible")) || firstTime) {
228             searchQuery.sortOn = pickerData["sortOn"];
229             searchQuery.sortOrder = pickerData["sortOrder"];
230             if (firstTime) { firstTime = false; }
231         }
232 
233         // Construct search query
234         var sq = searchQuery.url + "?";
235         for (var e in searchQuery) {
236             if (searchQuery.hasOwnProperty(e) && e !== "url") {
237                 sq += e + "=" + searchQuery[e] + "&";
238             }
239         }
240 
241         // Display empty new container with loading anim
242         $pickeradvanced_content_search.append($pl_pageContainer);
243 
244         // Make the request
245         $.ajax({
246             url: sq,
247             type: "GET",
248             dataType: "json",
249             success: function(rawData) {
250                 if (!rawData.results) {
251                     rawData.results = rawData;
252                 }
253                 // Eval profile data for now and extend it with additional info
254                 var doAdd = true;
255                 var newData = [];
256                 for (var i = 0, il = rawData.results.length; i < il; i++) {
257                     doAdd = true;
258                     if (rawData.results[i].profile) {
259                         rawData.results[i] = rawData.results[i].profile;
260                     }
261                     if (pickerlist && pickerlist.length) {
262                         if ($.inArray(rawData.results[i].user, pickerlist) !== -1) {
263                             doAdd = false;
264                         }
265                     }
266                     if (rawData.results[i]['rep:userId'] && (rawData.results[i]['rep:userId'] === "admin" || rawData.results[i]['rep:userId'] === "anonymous")) {
267                         doAdd = false;
268                     }
269                     if (doAdd) {
270                         newData.push(rawData.results[i]);
271                     }
272                 }
273                 rawData.results = newData;
274                 rawData.total = newData.length;
275 
276                 // Render the results data template
277                 var pageHTML = $.TemplateRenderer($pickeradvanced_content_search_pagetemplate, rawData);
278 
279                 // Remove loading animation
280                 $pl_pageContainer = $(".pickeradvanced_page.pickeradvanced_page_list.loadinganim");
281                 $pl_pageContainer.removeClass("loadinganim");
282 
283                 // Inject results into DOM
284                 $pl_pageContainer.html(pageHTML);
285                 // Wire loading the next page when user scrolls to the bottom of the list
286                 if ((rawData.total > searchQuery.items) || (rawData.total === -1)) {
287                     $pickeradvanced_content_search.bind("scroll", function(e){
288 
289                         if ((e.target.scrollHeight - e.target.scrollTop - $(e.target).height() ) === 0) {
290                             $pickeradvanced_content_search.unbind("scroll");
291                             addPage(tuid, (pageNumber + 1), searchQuery);
292                         }
293                     });
294                 }
295 
296                 // Wire item selection
297                 if (pickerData.selectable) {
298                     $pickeradvanced_select_all_button.unbind("click");
299                     $pickeradvanced_select_all_button.click(function(){
300                         pickerData.selectCount = 0;
301                         $('#pickeradvanced_content_search ul li').each(function(i) {
302                             $(this).addClass("pickeradvanced_selected_user");
303                             pickerData.selectCount += 1;
304                             pickerData["selected"][$(this).attr("id")] = rawData.results[i];
305                             if (rawData.results[i]['rep:userId']) {
306                                 pickerData["selected"][$(this).attr("id")].entityType = "user";
307                             } else if (rawData.results[i]['sakai:group-id']) {
308                                 pickerData["selected"][$(this).attr("id")].entityType = "group";
309                             } else if (rawData.results[i]['jcr:name']) {
310                                 pickerData["selected"][$(this).attr("id")].entityType = "file";
311                             }
312                             if ($pickeradvanced_add_button.is(":disabled")) {
313                                 $pickeradvanced_add_button.removeAttr("disabled");
314                             }
315                         });
316                     });
317                     $("#pickeradvanced_page_" + pageNumber + " li").unbind("click");
318                     $("#pickeradvanced_page_" + pageNumber + " li").bind("click", function(e){
319                         // Check if user click on top of a link
320                         if (e.target.tagName.toLowerCase() !== "a") {
321                             // Remove from selected list
322                             if ($(this).hasClass("pickeradvanced_selected_user")) {
323                                 $(this).removeClass("pickeradvanced_selected_user");
324                                 delete pickerData["selected"][$(this).attr("id")];
325                                 pickerData.selectCount -= 1;
326                                 if (pickerData.selectCount < 1) {
327                                     $pickeradvanced_add_button.attr("disabled", "disabled");
328                                 }
329                             } else {
330                                 if ((pickerData.limit && pickerData.selectCount < pickerData.limit) || !pickerData.limit) {
331                                     // Add to selected list
332                                     $(this).addClass("pickeradvanced_selected_user");
333                                     for (var j = 0; j < rawData.results.length; j++) {
334                                         if (rawData.results[j]['rep:userId'] && rawData.results[j]['rep:userId'] == [$(this).attr("id")]) {
335                                             pickerData.selectCount += 1;
336                                             pickerData["selected"][$(this).attr("id")] = rawData.results[j];
337                                             pickerData["selected"][$(this).attr("id")].entityType = "user";
338                                         } else if (rawData.results[j]['sakai:group-id'] && rawData.results[j]['sakai:group-id'] == [$(this).attr("id")]) {
339                                             pickerData.selectCount += 1;
340                                             pickerData["selected"][$(this).attr("id")] = rawData.results[j];
341                                             pickerData["selected"][$(this).attr("id")].entityType = "group";
342                                         } else if (rawData.results[j]['jcr:name'] && rawData.results[j]['jcr:name'] == [$(this).attr("id")]) {
343                                             pickerData.selectCount += 1;
344                                             pickerData["selected"][$(this).attr("id")] = rawData.results[j];
345                                             pickerData["selected"][$(this).attr("id")].entityType = "file";
346                                         }
347                                     }
348                                 }
349                                 if ($pickeradvanced_add_button.is(":disabled")) {
350                                     $pickeradvanced_add_button.attr("disabled", "");
351                                 }
352                             }
353                         }
354                     });
355                 }
356 
357                 //Update known total amount of displayed elements
358                 pickerData.currentElementCount += rawData.results.length;
359                 $pickeradvanced_count_people.hide();
360                 $pickeradvanced_count_person.hide();
361                 //Set search result count
362                 if ((rawData.total === -1) || (rawData.total > 1000)) {
363                     // If we don't know the total display what we know
364                     $pickeradvanced_count.html(pickerData.currentElementCount);
365                     $pickeradvanced_count.show();
366                     $pickeradvanced_count_people.show();
367                     $pickeradvanced_count_of.show();
368                     $pickeradvanced_count_thousands.show();
369 
370                 } else {
371                     // If we know the exact total display it
372                     $pickeradvanced_count.html(rawData.total);
373                     $pickeradvanced_count.show();
374                     if (rawData.total === 1) {
375                         $pickeradvanced_count_person.show();
376                     } else {
377                         $pickeradvanced_count_people.show();
378                     }
379                 }
380                 // Wire sorting select dropdown
381                 $pickeradvanced_sort_on.unbind("change");
382                 $pickeradvanced_sort_on.bind("change", function(e){
383                     // Reset everything
384                     reset();
385 
386                     // Set config to new sort key
387                     pickerData["sortOn"] = $(this).val().split("_")[0];
388                     pickerData["sortOrder"] = $(this).val().split("_")[1];
389                     // Start from scratch
390                     addPage(0, searchQuery);
391 
392                 });
393             },
394             error: function(xhr, status, thrown) {
395 
396                 // If it's likely to be a genuine server error
397                 if ($pl_pageContainer.length === 0) {
398                     $pickeradvanced_content_search.html($pickeradvanced_error_template.html());
399                 } else {
400                     // Probably it's the last page of the result set
401                     $pl_pageContainer.last().remove();
402                     $pickeradvanced_count_of.hide();
403                     $pickeradvanced_count_thousands.hide();
404                 }
405             }
406         });
407     };

416     var addPeople = function() {
417       // this value is a comma-delimited list
418       // split it and get rid of any empty values in the array
419       $pickeradvanced_container.jqmHide();
420       $(window).trigger("sakai-pickeradvanced-finished", {"toAdd":pickerData["selected"]});
421     };

428     $(window).unbind("sakai-pickeradvanced-init");
429     $(window).bind("sakai-pickeradvanced-init", function(e, config) {
430         // position dialog box at users scroll position
431         var htmlScrollPos = $("html").scrollTop();
432         var docScrollPos = $(document).scrollTop();
433         if (htmlScrollPos > 0) {
434             $pickeradvanced_container.css({"top": htmlScrollPos + 130 + "px"});
435         } else if (docScrollPos > 0) {
436             $pickeradvanced_container.css({"top": docScrollPos + 130 + "px"});
437         }
438 
439         firstTime = true;
440         render(config.config);
441         $pickeradvanced_container.jqmShow();
442         pickerlist = config.list;
443     });

455     $pickeradvanced_search_filter.die("click");
456     $pickeradvanced_search_filter.live("click", function() {
457        var searchType = $(this).attr("id").split("pickeradvanced_search_")[1];
458        $(".pickeradvanced_selected_list").removeClass("pickeradvanced_selected_list");
459        $(this).parent("li").addClass("pickeradvanced_selected_list");
460        var searchURL = false;
461        var searchingInGroup = false;
462        switch (searchType) {
463            case "contacts":
464                searchURL = sakai.config.URL.SEARCH_USERS_ACCEPTED;
465                $pickeradvanced_sort_on.show();
466                break;
467            case "users":
468                searchURL = sakai.config.URL.SEARCH_USERS;
469                $pickeradvanced_sort_on.show();
470                break;
471            case "groups":
472                searchURL = sakai.config.URL.SEARCH_GROUPS;
473                $pickeradvanced_sort_on.hide();
474                break;
475            case "groups_member":
476                searchURL = sakai.config.URL.SEARCH_GROUPS;
477                $pickeradvanced_sort_on.hide();
478                break;
479            case "groups_manager":
480                searchURL = sakai.config.URL.GROUPS_MANAGER;
481                $pickeradvanced_sort_on.hide();
482                break;
483            case "files_all":
484                searchURL = sakai.config.URL.SEARCH_ALL_FILES.replace(".json", ".infinity.json");
485                $pickeradvanced_sort_on.hide();
486                break;
487            case "files_mine":
488                searchURL = sakai.config.URL.POOLED_CONTENT_MANAGER.replace(".json", ".infinity.json");
489                $pickeradvanced_sort_on.hide();
490                break;
491            case "files_view":
492                searchURL = sakai.config.URL.POOLED_CONTENT_VIEWER.replace(".json", ".infinity.json");
493                $pickeradvanced_sort_on.hide();
494                break;
495            default: // should be any group specific search
496                searchURL = sakai.config.URL.SEARCH_GROUP_MEMBERS.replace(".json", ".3.json");
497                $pickeradvanced_sort_on.hide();
498                searchingInGroup = true;
499                break;
500        }
501        if (!searchingInGroup) {
502            pickerData["searchIn"] = searchURL + "?page=0&items=12&_=&q=";
503        } else {
504            pickerData["searchIn"] = searchURL + "?group=" + searchType.split("groups_")[1] + "&q=";
505        }
506        submitSearch();
507     });
508 
509     // Reset to defaults
510     reset();

516     $(window).trigger("sakai-pickeradvanced-ready");
