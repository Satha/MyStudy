102         /**
103          * Adds the configured user to the recipient list in the 'To' field.
104          * @param {String} name The display name (first and last name) of the user
105          * @param {String} uid The user his uuid.
106          */
107         var addRecipient = function(name, uid) {
108             // add an autoSuggest selection to the DOM
109             $(messageTo).text(name);
110         };

112         /**
113          * This method will check if there are any required fields that are not filled in.
114          * If a field is not filled in the invalidClass will be added to that field.
115          * @return true = no errors, false = error
116          */
117         var checkFieldsForErrors = function(recipients) {
118             var subjectEl = $(messageFieldSubject);
119             var bodyEl = $(messageFieldBody);
120 
121             var valid = true;
122             var subject = subjectEl.val();
123             var body = bodyEl.val();
124 
125             subjectEl.removeClass(invalidClass);
126             bodyEl.removeClass(invalidClass);
127 
128             if (!subject) {
129                 valid = false;
130                 subjectEl.addClass(invalidClass);
131             }
132             if (!body) {
133                 valid = false;
134                 bodyEl.addClass(invalidClass);
135             }
136             // check if there are recipients
137             if((recipients.length === 0 && !toUser) ||
138                 recipients.length === 1 && recipients[0] === "") {
139                 // no recipients are selected
140                 valid = false;
141                 // in the event allowOthers is false, the following will not be seen
142                 $(autoSuggestContainer).addClass(invalidClass);
143                 $(autoSuggestInput).addClass(invalidClass);
144             }
145 
146             return valid;
147         };

150          * Fade the message and close it.

173                 $(autoSuggestContainer).remove();
174                 $(autoSuggestResults).remove();
175                 contactsGroups = [];

177                 $("<input/>", {
178                     "id": "sendmessage_to_autoSuggest",
179                     "type": "text"
180                 }).insertAfter(messageMultipleLabel);
181             }
182 
183             // Remove error status styling classes
184             $(messageFieldSubject).removeClass(invalidClass);
185             $(messageFieldBody).removeClass(invalidClass);
186 
187             // Show the form and hide any previous messages.
188             $(messageDone).hide();
189             $(messageForm).show();
190         };

210         /**
211          * This function will show a message in either a red or a green square.
212          * @param {String} idToAppend The id of the element you wish to set this message in.
213          * @param {String} msg The message you want to display.
214          * @param {Boolean} isError true for error (red block)/false for normal message(green block).
215          * @param {Number} timeout The amount of milliseconds you want the message to be displayed, 0 = always (till the next message)
216          */
217         var showGeneralMessage = function(idToAppend, msg, isError, timeout) {
218             $(idToAppend).html(sakai.api.Security.saneHTML(msg));
219             if (isError) {
220                 $(idToAppend).addClass(errorClass);
221                 $(idToAppend).removeClass(normalClass);
222             }
223             else {
224                 $(idToAppend).removeClass(errorClass);
225                 $(idToAppend).addClass(normalClass);
226             }
227 
228             $(idToAppend).show();
229             if (typeof timeout === "undefined" || timeout !== 0) {
230                 $(idToAppend).fadeOut(generalMessageFadeOutTime);
231             }
232         };

234         /**
235          * Called when the request to the server has been answered
236          * @param {Boolean} succes    If the request failed or succeeded.
237          */
238         var showMessageSent = function(success) {
239             // Clear the subject and body input fields
240             $(messageFieldSubject).val("");
241             $(messageFieldBody).val("");
242 
243             // show a message
244             // We hide the formfield and show a message depending on the succes parameter.
245             $(messageForm).hide();
246             $(messageDone).show();
247 
248             // Remove any previous classes that their might be.
249             $(messageDone).removeClass(errorClass);
250             $(messageDone).removeClass(normalClass);
251 
252             // Depending on success we add the correct class and show the appropriate message.
253             if (success) {
254                 $(messageDone).addClass(normalClass);
255                 $(messageDone).text(sakai.api.Security.saneHTML($(messageOK).text()));
256                 var successMsg = $(notificationSuccess).text();
257                 sakai.api.Util.notification.show("", successMsg, sakai.api.Util.notification.type.INFORMATION);
258             }
259             else {
260                 $(messageDone).addClass(errorClass);
261                 $(messageDone).text(sakai.api.Security.saneHTML($(messageError).text()));
262                 var errorMsg = $(notificationError).text();
263                 sakai.api.Util.notification.show("", errorMsg, sakai.api.Util.notification.type.ERROR);
264             }
265 
266             // If we have a valid callback function we call that
267             // and dont show the message
268             // If we dont have a callback we show a default message and fade out the layover.
269             if (success && callbackWhenDone !== null) {
270                 callbackWhenDone({"response" : "OK"});
271             }
272             else {
273                 fadeMessage();
274             }
275         };

282         /**
283          * Initiates the 'To' field autoSuggest plugin with contact and group data
284          * @param {Object} contactsGroupsList Array containing JSON objects for
285          * each contact and group that should appear in the autoSuggest 'To'
286          * field.
287          * @return None
288          */
289         var autoSuggestContactsGroups = function(contactsGroupsList) {
290             $("#sendmessage_to_autoSuggest").autoSuggest("", {
291                 asHtmlID: "sendmessage_to_autoSuggest",
292                 startText: "Enter contact or group names",
293                 searchObjProps: "name",
294                 selectedItemProp: "name",
295                 keyDelay: "200",
296                 retrieveLimit: 10,
297                 formatList: function(data, elem) {
298                     // formats each line to be presented in autosuggest list
299                     // add the correct image, wrap name in a class
300                     var imgSrc = "/dev/images/user_avatar_icon_32x32.png";
301                     if(data.type === "group") {
302                         imgSrc = "/dev/images/group_avatar_icon_32x32.png";
303                     }
304                     var line_item = elem.html(
305                         '<img class="sm_suggestion_img" src="' + imgSrc + '" />' +
306                         '<span class="sm_suggestion_name">' + data.name + '</span>');
307                     return line_item;
308                 },
309                 source: function(query, add) {
310                     var searchUrl = sakai.config.URL.SEARCH_USERS_GROUPS;
311                     sakai.api.Server.loadJSON(searchUrl.replace(".json", ""), function(success, data){
312                         if (success) {
313                             var suggestions = [];
314                             $.each(data.results, function(i) {
315                                 if (data.results[i]["rep:userId"] &&
316                                 !(toUser && toUser.uuid == data.results[i]["rep:userId"])) {
317                                     // && data.results[i]["rep:userId"] !== sakai.data.me.user.userid) { // add this to ignore the user sending the message
318                                     suggestions.push({"value": data.results[i]["rep:userId"], "name": sakai.api.Security.saneHTML(sakai.api.User.getDisplayName(data.results[i])), "type": "user"});
319                                 } else if (data.results[i]["sakai:group-id"]) {
320                                     suggestions.push({"value": data.results[i]["sakai:group-id"], "name": data.results[i]["sakai:group-title"], "type": "group"});
321                                 }
322                             });
323                             add(suggestions);
324                         }
325                     }, {"q": "*" + query.replace(/\s+/g, "* OR *") + "*", "page": 0, "items": 15});
326                 }
327             });
328         };

331         /**
332          * Shows the lightbox and fills in the from and to field.
333          * @param {Object} hash The jqModal hash.
334          * @return None
335          */
336         var loadMessageDialog = function(hash) {
337             // Fill in the 'From' text with the current user's name
338             $(messageFieldFrom).text(sakai.api.User.getDisplayName(me.profile));
339 
340             // Depending on the allowOthers variable we show the appropriate input
341             if (allowOthers) {
342                 // Enable multiple recipients
343                 $(messageMultipleToContainer).show();
344 
345                 // Check if a recipient has already been set
346                 if (toUser !== false) {
347                     $(messageSingleToContainer).show();
348                     addRecipient(toUser.firstName + " " + toUser.lastName, toUser.uuid);
349                 } else {
350                     $(messageMultipleLabel).text("To");
351                 }
352 
353                 // fetch user's contacts and associated groups and set up
354                 // autoSuggest field with this content
355                 autoSuggestContactsGroups();
356             }
357             else {
358                 // We send this to a specific user.
359                 // Show the correct input box and fill in the name.
360                 $(messageMultipleToContainer).hide();
361                 $(messageSingleToContainer).show();
362 
363                 // check for null
364                 if (toUser !== null) {
365                     // Fill in the username
366                     $(messageFieldToSingle).text(sakai.api.Security.saneHTML(
367                         toUser.firstName + " " + toUser.lastName));
368                 }
369             }
370 
371             // If this is a layover we show the popup.
372             if (layover) {
373                 hash.w.show();
374             }
375         };

382         /**
383          * Initializes the sendmessage widget, optionally preloading the message
384          * with a recipient, subject and body. By default, the widget appears as
385          * a modal dialog. This function can be called from other widgets or pages.
386          * @param {Object} userObj The user object containing the nescecary information {uuid:  "user1", firstName: "foo", lastName: "bar"}, or a user profile
387          * @param {Boolean} allowOtherReceivers If the user can add other users, default = false
388          * @param {String} insertInId Insert the HTML into another element instead of showing it as a popup
389          * @param {Object} callback When the message is sent this function will be called. If no callback is provided a standard message will be shown that fades out.
390          */
391         sakai.sendmessage.initialise = function(userObj, allowOtherReceivers, insertInId, callback, subject, body) {
392 
393             // Make sure that everything is standard.
394             resetView();
395 
396             // The user we are sending a message to.
397             if (userObj && userObj.firstName) {
398                 toUser = userObj;
399             } else if (userObj) {
400                 toUser = {};
401                 toUser.firstName = sakai.api.User.getProfileBasicElementValue(userObj, "firstName");
402                 toUser.lastName = sakai.api.User.getProfileBasicElementValue(userObj, "lastName");
403                 toUser.uuid = userObj["rep:userId"];
404             } else {
405                 toUser = false;
406             }
407 
408             // Maybe this message can be sent to multiple people.
409             allowOthers = false;
410             if (allowOtherReceivers) {
411                 allowOthers = allowOtherReceivers;
412             }
413 
414             // Putting the subject and body which have been send in the textboxes
415             if(body) {
416                 $(messageFieldBody).val(sakai.api.Security.saneHTML(body));
417             }
418             if(subject) {
419                 $(messageFieldSubject).val(sakai.api.Security.saneHTML(subject));
420             }
421 
422             // Maybe we dont want to display a popup but instead want to add it in another div.
423             if (insertInId) {
424 
425                 // Make sure this id exists.
426                 if ($(insertInId).length > 0) {
427                     // The id exists!
428                     layover = false;
429                     putContentInId = insertInId;
430 
431                     // Remove the dialog stuff.
432                     $(dialogHeaderClass).remove();
433                     $(messageDialogContainer).removeClass(dialogClass.replace(/\./,''));
434                     $(dialogBoxContainer).removeClass(dialogBoxClass);
435                     $(dialogFooterContainer).removeClass(dialogFooterClass);
436 
437                     // Altough this isnt strictly nescecary it is cleaner.
438                     rootel = $(insertInId);
439                     rootel.append($(messageDialogContainer));
440                 }
441 
442             }
443 
444             // Store the callback
445             if (callback) {
446                 callbackWhenDone = callback;
447             }
448 
449             // show popup
450             if (layover) {
451                 $(messageDialogContainer).jqmShow();
452             }
453             else {
454                 // We want to add this in another element.
455                 loadMessageDialog();
456             }
457 
458             var o = {
459                 "setSubject" : setSubject,
460                 "setBody" : setBody
461             };
462 
463             return o;
464         };

471         /**
472          * Callback function called after a call to send a message has completed
473          * @param {Boolean} success Status of the 'sendMessage' AJAX call
474          * @param {Object} data Data returned from the 'sendMessage' AJAX call
475          * @return None
476          */
477         var handleSentMessage = function(success, data) {
478             if(success) {
479                 showMessageSent(success);
480             } else {
481                 alert("Your message failed to be delivered.");
482             }
483         };

485         /**
486          * Event handler to respond when the Send Message button is clicked.
487          * Checks that user input is valid and initiates a 'sendMessage' AJAX
488          * call to the server for the selected recipients.
489          */
490         $(buttonSendMessage).bind("click", function(ev) {
491             var recipients = [];
492             if (allowOthers) {
493                 // fetch list of selected recipients
494                 var recipientsString = $(autoSuggestValues).val();
495                 // autoSuggest adds unnecessary commas to the beginning and end
496                 // of the values string; remove them
497                 if(recipientsString[0] === ",") {
498                     recipientsString = recipientsString.slice(1);
499                 }
500                 if(recipientsString[recipientsString.length - 1] === ",") {
501                     recipientsString = recipientsString.slice(0, -1);
502                 }
503                 recipients = recipientsString.split(",");
504             }
505 
506             // Check if toUser has been defined
507             if(toUser) {
508                 recipients.push(toUser.uuid);
509             }
510 
511             // Check the fields if there are any required fields that are not filled in.
512             if(checkFieldsForErrors(recipients)) {
513                 sakai.api.Communication.sendMessage(recipients,
514                     $(messageFieldSubject).val(), $(messageFieldBody).val(),
515                     "message", null, handleSentMessage);
516             } else {
517                 alert("All fields are required.");
518             }
519         });
