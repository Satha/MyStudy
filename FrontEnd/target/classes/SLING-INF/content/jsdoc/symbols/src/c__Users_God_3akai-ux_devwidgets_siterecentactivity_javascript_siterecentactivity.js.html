 22 /**
 23  * @name sakai.siterecentactivity
 24  *
 25  * @class siterecentactivity
 26  *
 27  * @description
 28  * Initialize the siterecentactivity widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  * @param {Boolean} showSettings Show the settings of the widget or not
 33  */
 34 sakai.siterecentactivity = function(tuid, showSettings){
 35 
 36     var count = 5;
 37     var rootel = $("#" + tuid);
 38 
 39     var siterecentactivityContainer = "#siterecentactivity_container";
 40     var siterecentactivityContainerTemplate = "siterecentactivity_container_template";
 41 
 42 
 43     /////////////////////
 44     // RECENT ACTIVITY //
 45     /////////////////////
 46 
 47     /**
 48      * Render the sakai recentactivity
 49      */
 50     sakai.siterecentactivity.render = function(){
 51         if(sakai.siterecentactivity.recentactivity){
 52             for (i = 0, j = sakai.siterecentactivity.recentactivity.items.length; i < j; i++) {
 53 
 54                 sakai.siterecentactivity.recentactivity.items[i].page_url = "/sites/" + sakai.siterecentactivity.recentactivity.items[i].site_id + "#page=" + sakai.siterecentactivity.recentactivity.items[i].page_id;
 55 
 56                 sakai.siterecentactivity.recentactivity.items[i].date_parsed = humane_date(sakai.siterecentactivity.recentactivity.items[i].date+ "Z");
 57 
 58                 //if (sakai.site.site_info._pages[sakai.siterecentactivity.recentactivity.items[i].page_id]) {
 59                 //    sakai.siterecentactivity.recentactivity.items[i].page_title = sakai.site.site_info._pages[sakai.siterecentactivity.recentactivity.items[i].page_id]["pageTitle"];
 60                 //} else {
 61                 //     sakai.siterecentactivity.recentactivity.items[i].page_title = sakai.siterecentactivity.recentactivity.items[i].page_id;
 62                 //}
 63 
 64             }
 65 
 66             // We reverse the array to see the most recent item on top of the list
 67             var reversedItems = {
 68                 "items" : sakai.siterecentactivity.recentactivity.items.reverse()
 69             };
 70 
 71             // Render the recent sites
 72             $(siterecentactivityContainer).html($.TemplateRenderer(siterecentactivityContainerTemplate, reversedItems));
 73         }else{
 74             sakai.siterecentactivity.getRecentActivity(sakai.siterecentactivity.render);
 75         }
 76     };
 77 
 78     /**
 79      * Get the recent activity from a file
 80      * @param {Object} callback Callback function that will be executed
 81      */
 82     sakai.siterecentactivity.getRecentActivity = function(callback){
 83 
 84         sakai.api.Server.loadJSON("/~" + sakai.data.me.user.userid + "/private/recentactivity", function(success, data) {
 85             if (success) {
 86                 sakai.siterecentactivity.recentactivity = data;
 87             }
 88             else {
 89                 sakai.siterecentactivity.recentactivity = {
 90                     items: []
 91                 };
 92             }
 93             callback(success, data);
 94         });
 95     };
 96 
 97     /**
 98      * Save the recentactivity of a site to a file in JCR
 99      * and render the feed when is was successful
100      */
101     var saveRecentActivity = function(){
102 
103         // Save the recentactivity json file
104         sakai.api.Server.saveJSON("/~" + sakai.data.me.user.userid + "/private/recentactivity", sakai.siterecentactivity.recentactivity, sakai.siterecentactivity.render());
105     };
106 
107 
108     /**
109      * Save a item to the recent activities file
110      * @param {Object} activityitem A JSON object in the following format:
111      * {
112      *     "user_id" : "admin",
113      *     "date" : "2009-10-12T10:25:19Z",
114      *     "page_id" : "test",
115      *     "type" : "page_create"
116      * }
117      */
118     sakai.siterecentactivity.addRecentActivity = function(activityitem){
119 
120         // Check whether the activity item is actually an object
121         if($.isPlainObject(activityitem)){
122 
123             // Set the date of the activity
124             activityitem.date = sakai.api.Util.createSakaiDate();
125 
126             // Construct the callback function
127             var callback = function(){
128 
129                 // Add the current activity item to the array of activity items
130                 sakai.siterecentactivity.recentactivity.items.push(activityitem);
131 
132                 // If the number of recent activity items exceeds a specific number, remove the other ones
133                 if(sakai.siterecentactivity.recentactivity.items.length > count){
134                     var count_remove = sakai.siterecentactivity.recentactivity.items.length - count;
135 
136                     sakai.siterecentactivity.recentactivity.items.splice(0, count_remove);
137                 }
138 
139                 // Save the recent activity to a file in JCR
140                 saveRecentActivity();
141             };
142 
143             // Get the recent activity
144             sakai.siterecentactivity.getRecentActivity(callback);
145 
146         }
147 
148     };
149 
150     ///////////////////////
151     // Initial functions //
152     ///////////////////////
153 
154     $("#siterecentactivity_settings_submit",rootel).click(function(){
155         sakai.api.Widgets.Container.informFinish(tuid, "siterecentactivity");
156     });
157     $("#siterecentactivity_settings_cancel",rootel).click(function(){
158         sakai.api.Widgets.Container.informCancel(tuid, "siterecentactivity");
159     });
160 
161     // Hide or show the settings
162     if (showSettings){
163         $("#siterecentactivity_output",rootel).hide();
164         $("#siterecentactivity_settings",rootel).show();
165     } else {
166         $("#siterecentactivity_settings",rootel).hide();
167         $("#siterecentactivity_output",rootel).show();
168     }
169 
170     sakai.siterecentactivity.render();

174     /*
175      * Javascript Humane Dates
176      * Copyright (c) 2008 Dean Landolt (deanlandolt.com)
177      * Re-write by Zach Leatherman (zachleat.com)
178      *
179      * Adopted from the John Resig's pretty.js
180      * at http://ejohn.org/blog/javascript-pretty-date
181      * and henrah's proposed modification
182      * at http://ejohn.org/blog/javascript-pretty-date/#comment-297458
183      *

187     var humane_date = function(date_str){
188         var time_formats = [
189             [60, 'Just Now'],
190             [90, '1 Minute'], // 60*1.5
191             [3600, 'Minutes', 60], // 60*60, 60
192             [5400, '1 Hour'], // 60*60*1.5
193             [86400, 'Hours', 3600], // 60*60*24, 60*60
194             [129600, '1 Day'], // 60*60*24*1.5
195             [604800, 'Days', 86400], // 60*60*24*7, 60*60*24
196             [907200, '1 Week'], // 60*60*24*7*1.5
197             [2628000, 'Weeks', 604800], // 60*60*24*(365/12), 60*60*24*7
198             [3942000, '1 Month'], // 60*60*24*(365/12)*1.5
199             [31536000, 'Months', 2628000], // 60*60*24*365, 60*60*24*(365/12)
200             [47304000, '1 Year'], // 60*60*24*365*1.5
201             [3153600000, 'Years', 31536000], // 60*60*24*365*100, 60*60*24*365
202             [4730400000, '1 Century'] // 60*60*24*365*100*1.5
203         ];
204 
205         var time = ('' + date_str).replace(/-/g,"/").replace(/[TZ]/g," "),
206             dt = new Date(),
207             seconds = ((dt - new Date(time) + (dt.getTimezoneOffset() * 60000)) / 1000),
208             token = ' Ago',
209             i = 0,
210             format;
211 
212         if (seconds < 0) {
213             seconds = Math.abs(seconds);
214             token = '';
215         }
216         format = time_formats[0];
217         while (format) {
218             if (seconds < format[0]) {
219                 if (format.length == 2) {
220                     return format[1] + (i > 1 ? token : ''); // Conditional so we don't return Just Now Ago
221                 } else {
222                     return Math.round(seconds / format[2]) + ' ' + format[1] + (i > 1 ? token : '');
223                 }
224             }
225             format = time_formats[i++];
226         }
227 
228         // overflow for centuries
229         if(seconds > 4730400000) {
230             return Math.round(seconds / 4730400000) + ' Centuries' + token;
231         }
232 
233         return date_str;
234     };
235 
236     if(typeof $ != "undefined") {
237         $.fn.humane_dates = function(){
238             return this.each(function(){
239                 var date = humane_date(this.title);
240                 if(date && $(this).text() != date) {// don't modify the dom if we don't have to
241                     $(this).text(sakai.api.Security.saneHTML(date));
242                 }
243             });
244         };
245     }
