138     /**
139      * Function which (re)-loads the information available on a site (async)
140      * @param pageToOpen {String} URL safe title of a page which we want to open after the site info object refresh (optional)
141      * @return void
142      */
143     sakai.sitespages.refreshSiteInfo = function(pageToOpen, loadNav) {
144         var doLoadNav = true;
145         if (loadNav !== undefined) {
146             doLoadNav = loadNav;
147         }
148 
149         // Load site information
150         $.ajax({
151             url: sakai.config.URL.SEARCH_PAGES,
152             cache: false,
153             async: false,
154             data: {
155                 "path": config.fullpath + "_pages/",
156                 "items": 255
157             },
158             success: function(response) {
159 
160                 // Init
161                 response = response.results;
162 
163                 // Sort site objects by their path
164                 var compareURL = function(a,b) {
165                     return a.path>b.path ? 1 : a.path<b.path ? -1 : 0;
166                 };
167                 response.sort(compareURL);
168 
169                 // Create site_info object, the unique key being the partial path of the page from the root of the site
170                 // This will also keep the alphabetical order
171                 sakai.sitespages.site_info["_pages"] = {};
172                 for (var i=0, j=response.length; i<j; i++) {
173 
174                     if (typeof response[i] !== "undefined") {
175 
176                         // Save page data and add some helper attributes
177 
178                         // URL safe title
179                         var url_safe_title = "";
180                         var url_elements = response[i]["jcr:path"].split("/");
181                         url_safe_title = url_elements[url_elements.length - 1];
182                         response[i]["pageURLTitle"] = url_safe_title;
183 
184                         // URL safe name
185                         var url_safe_name = sakai.sitespages.createURLName(response[i]["jcr:path"]);
186                         response[i]["pageURLName"] = url_safe_name;
187 
188                         // Page depth
189                         response[i]["pageDepth"] = url_elements.length;
190 
191                         // Page base folder
192                         url_elements.pop();
193                         response[i]["pageFolder"] = url_elements.join("/");
194 
195                         // Main page data
196                         sakai.sitespages.site_info["_pages"][url_safe_name] = response[i];
197                     }
198                 }
199 
200                 // Create a helper function which returns the number of pages
201                 sakai.sitespages.site_info.number_of_pages = function() {
202                     var counter = 0;
203                     for (var i in sakai.sitespages.site_info._pages) {
204                         if (sakai.sitespages.site_info._pages.hasOwnProperty(i)) {
205                             counter++;
206                         }
207                     }
208                     return counter;
209                 };
210 
211                 // Open page if necessary
212                 if (pageToOpen && pageToOpen !== "") {
213                     sakai.sitespages.openPage(pageToOpen);
214                 }
215 
216                 // Load site navigation
217                 if (doLoadNav) {
218                     sakai.sitespages.loadSiteNavigation();
219                 }
220             },
221             error: function(xhr, textStatus, thrownError) {
222                 sakai.site.site_info = {};
223                 debug.error("site.js: Could not load site info. \n HTTP status code: " + xhr.status);
224 
225             }
226 
227         });

492     /**
493      * Cretes URL safe page title
494      * @param title {String} The title of a page
495      * @returns {String} URL safe title
496      */
497     sakai.sitespages.createURLSafeTitle = function(title) {
498         var url_safe_title = title.toLowerCase();
499         url_safe_title = url_safe_title.replace(/ /g,"-");
500         url_safe_title = url_safe_title.replace(/'/g,"");
501         url_safe_title = url_safe_title.replace(/"/g,"");
502         url_safe_title = url_safe_title.replace(/[:]/g,"");
503         url_safe_title = url_safe_title.replace(/[?]/g,"");
504         url_safe_title = url_safe_title.replace(/[=]/g,"");
505 
506         var regexp = new RegExp("[^a-z0-9_-]", "gi");
507         url_safe_title = url_safe_title.replace(regexp,"-");
508 
509         return url_safe_title;
510     };
