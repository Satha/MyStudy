178     /**
179      * Shows the settings screen
180      * @param {String} response
181      * @param {Boolean} exists
182      */
183     var showSettingsScreen = function(response, exists) {
184         if (exists) {
185             // Fill in video info
186             $(videoUrl, rootel).val(response.URL);
187             $(videoSource, rootel).val(response.source);
188         } else {
189             // Fill in video defaults
190             $(videoUrl, rootel).val("http://");
191             $(videoSource, rootel).val("");
192         }
193         $(videoOutput, rootel).hide();
194         $(videoSettings, rootel).show();
195     };
196 
197     /**
198      * returns a Json-object of the video-settings
199      */
200     var getVideoJson = function () {
201         var URL = $.trim($(videoUrl, rootel).val());
202         var source = $.trim($(videoSource, rootel).val());
203 
204         var video = {
205             "userid": sakai.data.me.user.userid,
206             "source": source,
207             "URL": URL
208         };
209 
210         return video;
211     };

331     /**
332      * Switch between main and settings page
333      * @param {Boolean} showSettings Show the settings of the widget or not
334      */
335     sakai.api.Widgets.loadWidgetData(tuid, function (success, data) {
336 
337         if (success) {
338             // we have a video set
339             ShowVideoSakaiPlayer(data, videoShowMain);
340             if (showSettings) {
341                 showSettingsScreen(data, true);
342             } else {
343                 $(videoSettings, rootel).hide();
344                 $(videoOutput, rootel).show();
345             }
346         } else {
347             // no video set
348             if (showSettings) {
349                 showSettingsScreen(data.status, false);
350             } else {
351                 $(videoSettings, rootel).hide();
352                 $(videoOutput, rootel).show();
353             }
354         }
355 
356     });
