 69 /**
 70  * window.debug, a console dot log wrapper
 71  * adapted from html5boilerplate.com's window.log and Ben Alman's window.debug
 72  *
 73  * Only logs information when sakai.config.displayDebugInfo is switched on
 74  *
 75  * debug.log, debug.error, debug.warn, debug.debug, debug.info
 76  * usage: debug.log("argument", {more:"arguments"})
 77  *
 78  * paulirish.com/2009/log-a-lightweight-wrapper-for-consolelog/
 79  * benalman.com/projects/javascript-debug-console-log/
 80  * https://gist.github.com/466188
 81  */
 82 window.debug = (function() {
 83     var that = {},
 84         methods = [ 'error', 'warn', 'info', 'debug', 'log' ],
 85         idx = methods.length;
 86 
 87     var createLogMethod = function(method) {
 88         that[method] = function() {
 89             if (!window.console || !sakai.config.displayDebugInfo) {
 90                 return;
 91             }
 92             if (console.firebug) {
 93                 console[method].apply(console, arguments);
 94             } else if (console[method]) {
 95                 console[method](Array.prototype.slice.call(arguments));
 96             } else {
 97                 console.log(Array.prototype.slice.call(arguments));
 98             }
 99         };
100     };
101 
102     while (--idx>=0) {
103         createLogMethod(methods[idx]);
104     }
105 
106     return that;
107 })();

125 /**
126  * Wrapper function for creating a Nakamura activity
127  *
128  * @param nodeUrl {String} The URL of the node we would like the activity to be
129  * stored on
130  * @param appID {String} The ID of the application/functionality creating the
131  * activity
132  * @param templateID {String} The ID of the activity template
133  * @param extraData {Object} Any extra data which will be stored on the activity
134  * node
135  * @param callback {Function} Callback function executed at the end of the
136  * operation
137  * @returns void
138  */
139 sakai.api.Activity.createActivity = function(nodeUrl, appID, templateID, extraData, callback) {
140 
141     // Check required params
142     if (typeof appID !== "string" || appID === "") {
143         debug.error("sakai.api.Activity.createActivity(): appID is required argument!");
144         return;
145     }
146     if (typeof templateID !== "string" || templateID === "") {
147         debug.error("sakai.api.Activity.createActivity(): templateID is required argument!");
148     }
149 
150     // Create event url with appropriate selector
151     var activityUrl = nodeUrl + ".activity.json";
152 
153     // Create data object to send
154     var dataToSend = {
155         "sakai:activity-appid": appID,
156         "sakai:activity-templateid": templateID
157     };
158     for (var i in extraData) {
159         if (extraData.hasOwnProperty(i)) {
160             dataToSend[i] = extraData[i];
161         }
162     }
163 
164     // Send request to create the activity
165     $.ajax({
166         url: activityUrl,
167         traditional: true,
168         type: "POST",
169         data: dataToSend,
170         success: function(data){
171 
172             if ($.isFunction(callback)) {
173                 callback(data, true);
174             }
175         },
176         error: function(xhr, textStatus, thrownError) {
177 
178             if ($.isFunction(callback)) {
179                 callback(xhr.status, false);
180             }
181         }
182     });
183 };

236  * Sanitizes HTML content. All untrusted (user) content should be run through
237  * this function before putting it into the DOM

239  * @param inputHTML {String} The content string we would like to sanitize
240  *
241  * @returns {String} Escaped and sanitized string
242  */
243 sakai.api.Security.saneHTML = function(inputHTML) {
244 
245     if (inputHTML === "") {
246         return "";
247     }
248 
249 
250     // Filter which runs through every url in inputHTML
251     var filterUrl = function(url) {
252 
253         return url;
254 
255     };
256 
257     // Filter which runs through every name id and class
258     var filterNameIdClass = function(nameIdClass) {
259 
260         return nameIdClass;
261 
262     };
263 
264     html4.ELEMENTS["video"] = 0;
265     html4.ATTRIBS["video::src"] = 0;
266     html4.ATTRIBS["video::class"] = 0;
267     html4.ATTRIBS["video::autoplay"] = 0;
268     html4.ELEMENTS["embed"] = 0;
269     html4.ELEMENTS["i"] = 0;
270     html4.ATTRIBS["embed::src"] = 0;
271     html4.ATTRIBS["embed::class"] = 0;
272     html4.ATTRIBS["embed::autostart"] = 0;
273     // A slightly modified version of Caja's sanitize_html function to allow style="display:none;"
274     var sakaiHtmlSanitize = function(htmlText, opt_urlPolicy, opt_nmTokenPolicy) {
275         var out = [];
276         html.makeHtmlSanitizer(
277             function sanitizeAttribs(tagName, attribs) {
278                 for (var i = 0; i < attribs.length; i += 2) {
279                     var attribName = attribs[i];
280                     var value = attribs[i + 1];
281                     var atype = null, attribKey;
282                     if (html4.ATTRIBS.hasOwnProperty(tagName + '::' + attribName)) {
283                         attribKey = tagName + '::' + attribName;
284                         atype = html4.ATTRIBS[attribKey];
285                     } else if (html4.ATTRIBS.hasOwnProperty('*::' + attribName)) {
286                         attribKey = '*::' + attribName;
287                         atype = html4.ATTRIBS[attribKey];
288                     }
289                     if (atype !== null) {
290                         switch (atype) {
291                             case html4.atype.SCRIPT:
292                             case html4.atype.STYLE:
293                                 var accept = ["color", "display", "background-color", "font-weight", "font-family",
294                                               "padding", "padding-left", "padding-right", "text-align", "font-style",
295                                               "text-decoration", "border"];
296                                 var sanitizedValue = "";
297                                 if (value){
298                                     var vals = value.split(";");
299                                     for (var attrid = 0; attrid < vals.length; attrid++){
300                                         var attrValue = $.trim(vals[attrid].split(":")[0]).toLowerCase();
301                                         if ($.inArray(attrValue, accept)){
302                                             sanitizedValue += vals[i];
303                                         }
304                                     }
305                                     if (!sanitizedValue) {
306                                         value = null;
307                                     }
308                                 } else {
309                                     value = sanitizedValue;
310                                 }
311                                 break;
312                             case html4.atype.IDREF:
313                             case html4.atype.IDREFS:
314                             case html4.atype.GLOBAL_NAME:
315                             case html4.atype.LOCAL_NAME:
316                             case html4.atype.CLASSES:
317                                 value = opt_nmTokenPolicy ? opt_nmTokenPolicy(value) : value;
318                                 break;
319                             case html4.atype.URI:
320                                 value = opt_urlPolicy && opt_urlPolicy(value);
321                                 break;
322                             case html4.atype.URI_FRAGMENT:
323                                 if (value && '#' === value.charAt(0)) {
324                                     value = opt_nmTokenPolicy ? opt_nmTokenPolicy(value) : value;
325                                     if (value) {
326                                         value = '#' + value;
327                                     }
328                                 } else {
329                                     value = null;
330                                 }
331                                 break;
332                         }
333                     } else {
334                         value = null;
335                     }
336                     attribs[i + 1] = value;
337                 }
338                 return attribs;
339             })(htmlText, out);
340         return out.join('');
341     };
342 
343     // Call a slightly modified version of Caja's sanitizer
344     return sakaiHtmlSanitize(inputHTML, filterUrl, filterNameIdClass);

349 /**
350  * Checks whether the given value is valid as defined by the given
351  * permissionsProperty.
352  *
353  * @param {Object} permissionsProperty Permissions property object
354  *   (i.e. sakai.config.Permissions.Groups.joinable) with valid values to check
355  *   against
356  * @param {Object} value Value to investigate
357  * @return true if the value has a valid property value, false otherwise
358  */
359 sakai.api.Security.isValidPermissionsProperty = function(permissionsProperty, value) {
360     if(!value || value === "") {
361         // value is empty - not valid
362         return false;
363     }
364     for(var index in permissionsProperty) {
365         if(permissionsProperty.hasOwnProperty(index)) {
366             if(value === permissionsProperty[index]) {
367                 // value is valid
368                 return true;
369             }
370         }
371     }
372     // value is not valid
373     return false;
374 };

459 /**
460  * Saves a specified JSON object to a specified URL in JCR. The structure of JSON data will be re-created in JCR as a node hierarchy.
461  *
462  * @param {String} i_url The path to the preference where it needs to be
463  * saved
464  * @param {Object} i_data A JSON object whic we would like to save
465  * (max 200 child object of each object)
466  * @param {Function} callback A callback function which is executed at the
467  * end of the operation
468  *
469  * @returns {Void}
470  */
471 sakai.api.Server.saveJSON = function(i_url, i_data, callback) {
472 
473     // Argument check
474     if (!i_url || !i_data) {
475 
476         // Log the error message
477         debug.warn("sakai.api.Server.saveJSON: Not enough or empty arguments!");
478 
479         // Still invoke the callback function
480         if ($.isFunction(callback)) {
481             callback(false, "The supplied arguments were incorrect.");
482         }
483 
484         // Make sure none of the other code in this function is executed
485         return;
486     }
487 
488     /**
489      * <p>Convert all the arrays in an object to an object with a unique key.<br />
490      * Mixed arrays (arrays with multiple types) are not supported.
491      * </p>
492      * <code>
493      * {
494      *     "boolean": true,
495      *     "array_object": [{ "key1": "value1", "key2": "value2"}, { "key1": "value1", "key2": "value2"}]
496      * }
497      * </code>
498      * to
499      * <code>
500      * {
501      *     "boolean": true,
502      *     "array_object": {
503      *         "__array__0__": { "key1": "value1", "key2": "value2"},
504      *         "__array__1__": { "key1": "value1", "key2": "value2"}
505      *     }
506      * }
507      * </code>
508      * @param {Object} obj The Object that you want to use to convert all the arrays to objects
509      * @return {Object} An object where all the arrays are converted into objects
510      */
511     var convertArrayToObject = function(obj) {
512 
513         // Since the native createTree method doesn't support an array of objects natively,
514         // we need to write extra functionality for this.
515         for(var i in obj){
516 
517             // Check if the element is an array, whether it is empty and if it contains any elements
518             if (obj.hasOwnProperty(i) && $.isArray(obj[i]) && obj[i].length > 0) {
519 
520                 // Deep copy the array
521                 var arrayCopy = $.extend(true, [], obj[i]);
522 
523                 // Set the original array to an empty object
524                 obj[i] = {};
525 
526                 // Add all the elements that were in the original array to the object with a unique id
527                 for (var j = 0, jl = arrayCopy.length; j < jl; j++) {
528 
529                     // Copy each object from the array and add it to the object
530                     obj[i]["__array__" + j + "__"] = arrayCopy[j];
531 
532                     // Run recursively
533                     convertArrayToObject(arrayCopy[j]);
534                 }
535             }
536 
537             // If there are array elements inside
538             else if ($.isPlainObject(obj[i])) {
539                 convertArrayToObject(obj[i]);
540             }
541 
542         }
543 
544         return obj;
545     };
546 
547     // Convert the array of objects to only objects
548     // We also need to deep copy the object so we don't modify the input parameter
549     i_data = convertArrayToObject($.extend(true, {}, i_data));
550 
551     // Send request
552     $.ajax({
553         url: i_url,
554         type: "POST",
555         data: {
556             ":operation": "import",
557             ":contentType": "json",
558             ":content": $.toJSON(i_data),
559             ":replace": true,
560             ":replaceProperties": true
561         },
562         dataType: "json",
563 
564         success: function(data){
565 
566             // If a callback function is specified in argument, call it
567             if ($.isFunction(callback)) {
568                 callback(true, data);
569             }
570         },
571 
572         error: function(xhr, status, e){
573 
574             // Log error
575             debug.error("sakai.api.Server.saveJSON: There was an error saving JSON data to: " + this.url);
576 
577             // If a callback function is specified in argument, call it
578             if ($.isFunction(callback)) {
579                 callback(false, xhr);
580             }
581         }
582     });
583 
584 };

648 /**
649  * <p>Convert all the objects with format __array__?__ in an object to an array</p>
650  * <code>
651  * {
652  *     "boolean": true,
653  *     "array_object": {
654  *         "__array__0__": {
655  *             "key1": "value1",
656  *             "key2": "value2"
657  *         }
658  *     }
659  * }
660  * </code>
661  * to
662  * <code>
663  * {
664  *     "boolean": true,
665  *     "array_object": [
666  *         {
667  *             "key1": "value1",
668  *             "key2": "value2"
669  *        }
670  *     ]
671  * }
672  * </code>
673  * @param {Object} specficObj The Object that you want to use to convert all the objects with the special format to arrays
674  * @param {Object} [globalObj] The parent object, we need this to run over the elements recursively
675  * @param {Object} [objIndex] The index of the parent object
676  * @return {Object} An object where all the objects with the special format are converted into arrays
677  */
678 sakai.api.Server.loadJSON.convertObjectToArray = function(specficObj, globalObj, objIndex){
679 
680     // Run over all the items in the object
681     for (var i in specficObj) {
682 
683         // If exists and it's an object recurse
684         if (specficObj.hasOwnProperty(i)) {
685 
686             // If it's a non-empty array-object it will have a first element with the key "__array__0__"
687             if (i === "__array__0__") {
688 
689                 // We need to get the number of items in the object
690                 var arr = [];
691                 var count = 0;
692                 for (var j in specficObj) {
693                     if (specficObj.hasOwnProperty(j)) {
694                         count++;
695                     }
696                 }
697 
698                 // Construct array of objects
699                 for(var k = 0, kl = count; k < kl; k ++){
700                     arr.push(specficObj["__array__"+k+"__"]);
701                 }
702 
703                 globalObj[objIndex] = arr;
704             }
705 
706             if ($.isPlainObject(specficObj[i])) {
707                 sakai.api.Server.loadJSON.convertObjectToArray(specficObj[i], specficObj, i);
708             }
709         }
710 
711     }
712     return specficObj;
713 };

812 /*
813  * Functionality that allows you to create HTML Templates and give that template
814  * a JSON object. That template will then be rendered and all of the values from
815  * the JSON object can be used to insert values into the rendered HTML. More information
816  * and examples can be found over here:
817  *
818  * http://code.google.com/p/trimpath/wiki/JavaScriptTemplates
819  *
820  * Template should be defined like this:
821  *  <div><!--
822  *   // Template here
823  *  --></div>
824  *
825  *  IMPORTANT: There should be no line breaks in between the div and the <!-- declarations,
826  *  because that line break will be recognized as a node and the template won't show up, as
827  *  it's expecting the comments tag as the first one.
828  *
829  *  We do this because otherwise a template wouldn't validate in an HTML validator and
830  *  also so that our template isn't visible in our page.
831  */
832 (function($){
833 
834     /**
835      * A cache that will keep a copy of every template we have parsed so far. Like this,
836      * we avoid having to parse the same template over and over again.
837      */
838     var templateCache = [];
839 
840     /**
841     * Trimpath Template Renderer: Renders the template with the given JSON object, inserts it into a certain HTML
842     * element if required, and returns the rendered HTML string
843     * @function
844     * @param {String|Object} templateElement The name of the template HTML ID or a jQuery selection object.
845     * @param {Object} templateData JSON object containing the template data
846     * @param {Object} outputElement (Optional) jQuery element in which the template needs to be rendered
847     * @param {Boolean} doSanitize (Optional) perform html sanitization. Defaults to true
848     */
849     $.TemplateRenderer = function (templateElement, templateData, outputElement, doSanitize) {
850 
851         var templateName;
852         var sanitize = true;
853         if (doSanitize !== undefined) {
854             sanitize = doSanitize;
855         }
856 
857         // The template name and the context object should be defined
858         if(!templateElement || !templateData){
859             throw "$.TemplateRenderer: the template name or the templateData is not defined";
860         }
861 
862         if(templateElement instanceof jQuery && templateElement[0]){
863             templateName = templateElement[0].id;
864         }
865         else if (typeof templateElement === "string"){
866             templateName = templateElement.replace("#", "");
867             templateElement = $("#" + templateName);
868         }
869         else {
870             throw "$.TemplateRenderer: The templateElement '" + templateElement + "' is not in a valid format or the template couldn't be found.";
871         }
872 
873         if (!templateCache[templateName]) {
874             var templateNode = templateElement.get(0);
875             if (templateNode) {
876                 var firstNode = templateNode.firstChild;
877                 var template = null;
878                 // Check whether the template is wrapped in <!-- -->
879                 if (firstNode && (firstNode.nodeType === 8 || firstNode.nodeType === 4)) {
880                     template = firstNode.data.toString();
881                 }
882                 else {
883                     template = templateNode.innerHTML.toString();
884                 }
885                 // Parse the template through TrimPath and add the parsed template to the template cache
886                 templateCache[templateName] = TrimPath.parseTemplate(template, templateName);
887 
888             }
889             else {
890                 throw "$.TemplateRenderer: The template '" + templateName + "' could not be found";
891             }
892         }
893 
894         // Run the template and feed it the given JSON object
895         var render = templateCache[templateName].process(templateData);
896 
897         // Run the rendered html through the sanitizer
898         if (sanitize) {
899             render = sakai.api.Security.saneHTML(render);
900         }
901 
902         // Check it there was an output element defined
903         // If so, put the rendered template in there
904         if (outputElement) {
905             outputElement.html(render);
906         }
907 
908         return render;
909 
910     };

912 })(jQuery);

920 /*
921  * We override the standard $.ajax error function, which is being executed when
922  * a request fails. We will check whether the request has failed due to an authorization
923  * required error, by checking the response code and then doing a request to the me service
924  * to find out whether we are no longer logged in. If we are no longer logged in, and the
925  * sendToLoginOnFail variable has been set in the options of the request, we will redirect
926  * to the login page with the current URL encoded in the url. This will cause the system to
927  * redirect to the page we used to be on once logged in.
928  */
929 (function($){
930 
931     /**
932     * Override default jQuery error behavior
933     * @function
934     * @param {String} s description
935     * @param {Object} xhr xhr object
936     * @param {String} status Status message
937     * @param {Object} e Thrown error
938     */
939     $.handleError = function (s, xhr, status, e) {
940 
941         var requestStatus = xhr.status;
942 
943         // Sometimes jQuery comes back with a parse-error, although the request
944         // was completely successful. In order to prevent the error method to be called
945         // in this case, we need this if clause.
946         if (requestStatus === 200) {
947             if (s.success) {
948                 s.success(xhr.responseText);
949             }
950         }
951         else {
952             // if the sendToLoginOnFail hasn't been set, we assume that we want to redirect when
953             // a 403 comes back
954             s.sendToLoginOnFail = s.sendToLoginOnFail || true;
955             if (requestStatus === 403 && s.sendToLoginOnFail) {
956 
957                 var decideLoggedIn = function(response, exists){
958                     var originalURL = document.location;
959                     originalURL = $.URLEncode(originalURL.pathname + originalURL.search + originalURL.hash);
960                     var redirecturl = sakai.config.URL.GATEWAY_URL + "?url=" + originalURL;
961                     if (exists && response.preferences && (response.preferences.uuid === "anonymous" || !response.preferences.uuid)) {
962                         document.location = redirecturl;
963                     }
964                 };
965 
966                 $.ajax({
967                     url: sakai.config.URL.ME_SERVICE,
968                     cache: false,
969                     success: function(data){
970                         decideLoggedIn(data, true);
971                     }
972                 });
973 
974             }
975 
976         // Handle HTTP conflicts thrown back by K2 (409) (for example when somebody tries to write to the same node at the very same time)
977         // We do this by re-sending the original request with the data transparently, behind the curtains, until it succeeds.
978         // This still does not eliminate a possibility of another conflict, but greatly reduces
979         // the chance and works in the background until the request is successful (ie jQuery will try to re-send the initial request until the response is not 409
980         // WARNING: This does not solve the locking/overwriting problem entirely, it merely takes care of high volume request related issues. Users
981         // should be notified in advance by the UI when somebody else is editing a piece of content, and should actively try reduce the possibility of
982         // overwriting.
983 /*        if (requestStatus === 409) {
984             // Retry initial post
985             $.ajax(s);
986         }*/
987 
988         // Call original error handler, but not in the case of 409 as we want that to be transparent for users
989         if ((s.error) && (requestStatus !== 409)) {
990           s.error(xhr, status, e);
991         }
992 
993         if (s.global) {
994           $.event.trigger("ajaxError", [xhr, status, e]);
995         }
996           }
997 
998     };
