 21 /**
 22  * @class i18n
 23  *
 24  * @description
 25  * <p>Internationalisation related functions for general page content, widget
 26  * content and UI elements.</p>
 27  * <p>This should only hold functions which are used across
 28  * multiple pages, and does not constitute functionality related to a single area/page.</p>
 29  *
 30  * @namespace
 31  * Internationalisation
 32  */
 33 sakai.api.i18n = sakai.api.i18n || {};
 34 
 35 /**
 36  * <p>Main i18n process</p>
 37  * <p>This function will be executed once the entire DOM has been loaded. The function will take
 38  * the HTML that's inside the body of the page, and load the default and user specific language
 39  * bundle. We then look for i18n keys inside the HTML, and replace with the values from the
 40  * language bundles. We always first check whether the key is available in the user specific
 41  * bundle and only if that one doesn't exist, we will take the value out of the default bundle.</p>
 42  */
 43 sakai.api.i18n.init = function(){
 44 
 45 
 46     /////////////////////////////
 47     // CONFIGURATION VARIABLES //
 48     /////////////////////////////
 49 
 50     // Initialise the sakai.data.18n variable
 51     /**
 52      * @name sakai.data.i18n
 53      */
 54     sakai.data.i18n = {};
 55 
 56     // Will contain all of the values for the default bundle
 57     sakai.data.i18n.defaultBundle = false;
 58 
 59     // Will contain all of the values for the bundle of the language chosen by the user
 60     sakai.data.i18n.localBundle = false;
 61 
 62     // Will contain all the i18n for widgets
 63     sakai.data.i18n.widgets = sakai.data.i18n.widgets || {};
 64 
 65     // Will contain all the i18n for widgets
 66     sakai.data.i18n.changeToJSON = sakai.data.i18n.changeToJSON || {};
 67 
 68 
 69     ////////////////////
 70     // HELP VARIABLES //
 71     ////////////////////
 72 
 73     /*
 74      * Cache the jQuery i18nable element. This makes sure that only pages with
 75      * the class i18nable on the body element get i18n translations and won't do
 76      * it on other pages.
 77      * <body class="i18nable">
 78      */
 79     var $i18nable = $("body.i18nable");
 80 
 81     /*
 82      * We take the HTML that is inside the body tag. We will use this to find string we want to
 83      * translate into the language chosen by the user
 84      */
 85     var tostring = $i18nable.html();
 86 
 87 
 88     ////////////////////////////
 89     // LANGUAGE BUNDLE LOADER //
 90     ////////////////////////////
 91 
 92     /**
 93      * Gets the site id if the user is currently on a site
 94      * if the user is on any other page then false is returned
 95      *
 96      * Proposed addin: Check to see if there is a siteid querystring parameter.
 97      * This will then i18n pages like site settings as well.
 98      */
 99     var getSiteId = function(){
100         var site = false;
101         var loc = ("" + document.location);
102         var siteid = loc.indexOf(sakai.config.URL.SITE_CONFIGFOLDER.replace(/__SITEID__/, ""));
103         if (siteid !== -1) {
104             var mark = (loc.indexOf("?") === -1) ? loc.length : loc.indexOf("?");
105             var uri = loc.substring(0, mark);
106             site = uri.substring(siteid, loc.length).replace(sakai.config.URL.SITE_CONFIGFOLDER.replace(/__SITEID__/, ""), "");
107             site = site.substring(0, site.indexOf("#"));
108         }
109         return site;
110     };
111 
112 
113     ////////////////////
114     // I18N FUNCTIONS //
115     ////////////////////
116 
117     /**
118      * Once all of the i18n strings have been replaced, we will finish the i18n step.
119      * The content of the body tag is hidden by default, in
120      * order not to show the non-translated string before they are translated. When i18n has
121      * finished, we can show the body again.
122      * We then tell the container that pre-loading of the page has finished and that widgets are
123      * now ready to be loaded. This will mostly mean that now the general page/container code
124      * will be executed.
125      * Finally, we will look for the definition of widgets inside the HTML code, which should look
126      * like this:
127      *  - Single instance: <div id="widget_WIDGETNAME" class="widget_inline"></div>
128      *  - Multiple instance support: <div id="widget_WIDGETNAME_UID_PLACEMENT" class="widget_inline"></div>
129      * and load them into the document
130      */
131     var finishI18N = function(){
132         var currentPage = window.location.pathname;
133         if (sakai.data.me.user.anon) {
134             if ($.inArray(currentPage, sakai.config.requireUser) > -1){
135                 sakai.api.Security.sendToLogin();
136                 return false;
137             }
138         } else {
139             if ($.inArray(currentPage, sakai.config.requireAnonymous) > -1){
140                 document.location = sakai.config.URL.MY_DASHBOARD_URL;
141                 return false;
142             }
143         }
144         if ($.inArray(currentPage, sakai.config.requireProcessing) === -1 && window.location.pathname.substring(0, 2) !== "/~"){
145             sakai.api.Security.showPage();
146         }
147         sakai.api.Widgets.Container.setReadyToLoad(true);
148         sakai.api.Widgets.widgetLoader.insertWidgets(null, false);
149         return true;
150     };
151 
152     /**
153      * This will give the body's HTML string, the local bundle (if present) and the default bundle to the
154      * general i18n function. This will come back with an HTML string where all of the i18n strings will
155      * be replaced. We then change the HTML of the body tag to this new HTML string.
156      * @param {Object} localjson
157      *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
158      * @param {Object} defaultjson
159      *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
160      *  in the default language
161      */
162     var doI18N = function(localjson, defaultjson){
163         var newstring = sakai.api.i18n.General.process(tostring, localjson, defaultjson);
164         // We actually use the old innerHTML function here because the $.html() function will
165         // try to reload all of the JavaScript files declared in the HTML, which we don't want as they
166         // will already be loaded
167         if($i18nable.length > 0){
168             $i18nable[0].innerHTML = newstring;
169         }
170         document.title = sakai.api.i18n.General.process(document.title, localjson, defaultjson);
171         finishI18N();
172     };
173 
174     /**
175      * This function will load the general language bundle specific to the language chosen by
176      * the user and will store it in a global variable. This language will either be the prefered
177      * user language or the prefered server language. The language will be available in the me feed
178      * and we'll use the global sakai.data.me object to extract it from. If there is no prefered langauge,
179      * we'll use the default bundle to translate everything.
180      */
181     var loadLocalBundle = function(langCode){
182         // globalization
183         var i10nCode = langCode.replace("_", "-");
184         if (Globalization.cultures) { // check if jquery.glob has been defined yet, should always be but just a sanity check
185             if (Globalization.cultures[i10nCode]) { // probably will never be true, but just in case, no need to get the script again
186                 Globalization.preferCulture(i10nCode);
187             } else {
188                 $.getScript(sakai.config.URL.I10N_BUNDLE_URL.replace("__CODE__", i10nCode), function(success, textStatus) {
189                     Globalization.preferCulture(i10nCode);
190                 });
191             }
192         }
193         // language bundles
194         $.ajax({
195             url: sakai.config.URL.I18N_BUNDLE_ROOT + langCode + ".properties",
196             success: function(data){
197                 data = sakai.data.i18n.changeToJSON(data);
198                 sakai.data.i18n.localBundle = data;
199                 doI18N(sakai.data.i18n.localBundle, sakai.data.i18n.defaultBundle);
200             },
201             error: function(xhr, textStatus, thrownError){
202                 // There is no language file for the language chosen by the user
203                 // We'll switch to using the default bundle only
204                 doI18N(null, sakai.data.i18n.defaultBundle);
205             }
206         });
207     };
208 
209     /**
210      * Load the language for a specific site
211      * @param {String} site The id of the site you want to load the language for
212      */
213     var loadSiteLanguage = function(site){
214         $.ajax({
215             url: sakai.config.URL.SITE_CONFIGFOLDER.replace("__SITEID__", site) + ".json",
216             cache: false,
217             success: function(data){
218                 var siteJSON = data;
219                 if (siteJSON.language && siteJSON.language !== "default_default") {
220                     loadLocalBundle(siteJSON.language);
221                 }
222                 else if (sakai.data.me.user.locale) {
223                     loadLocalBundle(sakai.data.me.user.locale.language + "_" + sakai.data.me.user.locale.country);
224                 }
225                 else {
226                     // There is no locale set for the current user. We'll switch to using the default bundle only
227                     doI18N(null, sakai.data.i18n.defaultBundle);
228                 }
229             },
230             error: function(xhr, textStatus, thrownError){
231                 loadLocalBundle();
232             }
233         });
234     };
235 
236     /**
237      * This change properties file into json object.
238      */
239     sakai.data.i18n.changeToJSON = function(input){
240        var json = {};
241         var inputLine = input.split(/\n/);
242         for (var i in inputLine) {
243             // IE 8 i has indexof as well which breaks the page.
244             if (inputLine.hasOwnProperty(i)) {
245                 var keyValuePair = inputLine[i].split(/ \=\s*/);
246                 var key = keyValuePair[0];
247                 var value = keyValuePair[1];
248                 json[key] = value;
249             }
250         }
251         return json;
252     };
253 
254     /**
255      * This will load the default language bundle and will store it in a global variable. This default bundle
256      * will be saved in a file called bundle/default.properties.
257      */
258     var loadDefaultBundle = function(){
259         $.ajax({
260             url: sakai.config.URL.I18N_BUNDLE_ROOT + "default.properties",
261             success: function(data){
262                 data = sakai.data.i18n.changeToJSON(data);
263                 sakai.data.i18n.defaultBundle = data;
264                 var site = getSiteId();
265                 if (!site) {
266                     if (sakai.data.me.user.locale) {
267                         loadLocalBundle(sakai.data.me.user.locale.language + "_" + sakai.data.me.user.locale.country);
268                     }
269                     else {
270                         // There is no locale set for the current user. We'll switch to using the default bundle only
271                         doI18N(null, sakai.data.i18n.defaultBundle);
272                     }
273                 }
274                 else {
275                     loadSiteLanguage(site);
276                 }
277             },
278             error: function(xhr, textStatus, thrownError){
279                 // There is no default bundle, so we'll just show the interface without doing any translations
280                 finishI18N();
281             }
282         });
283     };
284 
285 
286     /////////////////////////////
287     // INITIALIZATION FUNCTION //
288     /////////////////////////////
289 
290     loadDefaultBundle();
291 };

293 /**
294  * @class i18nGeneral
295  *
296  * @description
297  * Internationalisation related functions for general page content and UI elements
298  *
299  * @namespace
300  * General internationalisation functions
301  */
302 sakai.api.i18n.General = sakai.api.i18n.General || {};
303 
304 /**
305  * General process functions that will replace all the messages in a string with their corresponding translation.
306  * @example sakai.api.i18n.General.process(
307  *     "<h1>__MSG__CHANGE_LAYOUT__</h1>",
308  *     {"__MSG__CHANGE_LAYOUT__" : "verander layout"},
309  *     {"__MSG__CHANGE_LAYOUT__" : "change layout"}
310  * );
311  * @param {String} toprocess
312  *  HTML string in which we want to replace messages. Messages have the following
313  *  format: __MSG__KEY__
314  * @param {Object} localbundle
315  *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
316  * @param {Object} defaultbundle
317  *  JSON object where the keys are the keys we expect in the HTML and the values are the translated strings
318  *  in the default language
319  * @return {String} A processed string where all the messages are replaced with values from the language bundles
320  */
321 sakai.api.i18n.General.process = function(toprocess, localbundle, defaultbundle) {
322 
323     if(!toprocess){
324         return "";
325     }
326 
327     var expression = new RegExp(".{1}__MSG__(.*?)__", "gm"), processed = "", lastend = 0;
328 
329     while(expression.test(toprocess)) {
330         var replace = RegExp.lastMatch;
331         var lastParen = RegExp.lastParen;
332         var quotes = "";
333 
334         // need to add quotations marks if key is adjacent to an equals sign which means its probably missing quotes - IE
335         if (replace.substr(0,2) !== "__"){
336             if (replace.substr(0,1) === "="){
337                 quotes = '"';
338             }
339             replace = replace.substr(1, replace.length);
340         }
341         var toreplace;
342         // check for i18n debug
343         if (sakai.config.displayDebugInfo === true && sakai.data.me.user.locale && sakai.data.me.user.locale.language === "lu" && sakai.data.me.user.locale.country === "GB"){
344             toreplace = quotes + replace.substr(7, replace.length - 9) + quotes;
345             processed += toprocess.substring(lastend, expression.lastIndex - replace.length) + toreplace;
346             lastend = expression.lastIndex;
347         } else {
348             toreplace = quotes + sakai.api.i18n.General.getValueForKey(lastParen) + quotes;
349             processed += toprocess.substring(lastend, expression.lastIndex - replace.length) + toreplace;
350             lastend = expression.lastIndex;
351         }
352     }
353     processed += toprocess.substring(lastend);
354     return processed;
355 
356 };

358 /**
359  * Get the internationalised value for a specific key.
360  * We expose this function so you can do internationalisation within JavaScript.
361  * @example sakai.api.i18n.General.getValueForKey("CHANGE_LAYOUT");
362  * @param {String} key The key that will be used to get the internationalised value
363  * @return {String} The translated value for the provided key
364  */
365 sakai.api.i18n.General.getValueForKey = function(key) {
366     // First check if the key can be found in the locale bundle
367     if (sakai.data.i18n.localBundle[key]) {
368         return sakai.data.i18n.localBundle[key];
369     }
370     // If the key wasn't found in the localbundle, search in the default bundle
371     else if (sakai.data.i18n.defaultBundle[key]) {
372         return sakai.data.i18n.defaultBundle[key];
373     }
374     // If none of the about found something, log an error message
375     else {
376         debug.warn("sakai.api.i18n.General.getValueForKey: Not in local & default file. Key: " + key);
377         return false;
378     }
379 };

382 /**
383  * @class i18nWidgets
384  *
385  * @description
386  * Internationalisation in widgets
387  *
388  * @namespace
389  * Internationalisation in widgets
390  */
391 sakai.api.i18n.Widgets = sakai.api.i18n.Widgets || {};
392 
393 /**
394  * Get the value for a specific key in a specific widget.
395  * @example sakai.api.i18n.Widgets.getValueForKey("myprofile", "en_US", "PREVIEW_PROFILE");
396  * @param {String} widgetname The name of the widget
397  * @param {String} locale The locale for the getting the value
398  * @param {String} key The key which you want to be translated
399  * @return {String} The value you wanted to translate for a specific widget
400  */
401 sakai.api.i18n.Widgets.getValueForKey = function(widgetname, locale, key) {
402 
403     // Get a message key value in priority order: local widget language file -> widget default language file -> system local bundle -> system default bundle
404     if ((typeof sakai.data.i18n.widgets[widgetname][locale] === "object") && (typeof sakai.data.i18n.widgets[widgetname][locale][key] === "string")){
405         return sakai.data.i18n.widgets[widgetname][locale][key];
406 
407     } else if ((typeof sakai.data.i18n.widgets[widgetname]["default"][key] === "string") && (typeof sakai.data.i18n.widgets[widgetname]["default"] === "object")) {
408         return sakai.data.i18n.widgets[widgetname]["default"][key];
409 
410     } else if (sakai.data.i18n.localBundle[key]) {
411         return sakai.data.i18n.localBundle[key];
412 
413     } else if (sakai.data.i18n.defaultBundle[key]) {
414         return sakai.data.i18n.defaultBundle[key];
415 
416     }
