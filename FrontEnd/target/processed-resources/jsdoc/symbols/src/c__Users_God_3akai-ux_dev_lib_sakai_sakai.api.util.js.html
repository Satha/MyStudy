 35 /**
 36  * Parse a JavaScript date object to a JCR date string (2009-10-12T10:25:19)
 37  *
 38  * <p>
 39  *     Accepted values for the format [1-6]:
 40  *     <ol>
 41  *         <li>Year: YYYY (eg 1997)</li>
 42  *         <li>Year and month: YYYY-MM <br /> (eg 1997-07)</li>
 43  *         <li>Complete date: YYYY-MM-DD <br /> (eg 1997-07-16)</li>
 44  *         <li>Complete date plus hours and minutes: YYYY-MM-DDThh:mmTZD <br /> (eg 1997-07-16T19:20+01:00)</li>
 45  *         <li>Complete date plus hours, minutes and seconds: YYYY-MM-DDThh:mm:ssTZD <br /> (eg 1997-07-16T19:20:30+01:00)</li>
 46  *         <li>Complete date plus hours, minutes, seconds and a decimal fraction of a second YYYY-MM-DDThh:mm:ss.sTZD <br /> (eg 1997-07-16T19:20:30.45+01:00)</li>
 47  *     </ol>
 48  * </p>
 49  * <p>
 50  *     External links:
 51  *     <ul>
 52  *         <li><a href="http://www.w3.org/TR/NOTE-datetime">W3C datetime documentation</a></li>
 53  *         <li><a href="http://delete.me.uk/2005/03/iso8601.html">ISO8601 JavaScript function</a></li>
 54  *         <li><a href="http://confluence.sakaiproject.org/display/KERNDOC/KERN-643+Multiple+date+formats+in+the+back-end">Specification</a></li>
 55  *     </ul>
 56  * </p>
 57  * @param {Date} date
 58  *     JavaScript date object.
 59  *     If not set, the current date is used.
 60  * @param {Integer} format
 61  *     The format you want to put out
 62  * @param {String} offset
 63  *     Optional timezone offset +HH:MM or -HH:MM,
 64  *     if not set Z(ulu) or UTC is used
 65  * @return {String} a JCR date string
 66  */
 67 sakai.api.Util.createSakaiDate = function(date, format, offset) {
 68     if (!format) { format = 5; }
 69     if (!date) { date = new Date(); }
 70     if (!offset) {
 71         offset = 'Z';
 72     } else {
 73         var d = offset.match(/([\-+])([0-9]{2}):([0-9]{2})/);
 74         var offsetnum = (Number(d[2]) * 60) + Number(d[3]);
 75         offsetnum *= ((d[1] === '-') ? -1 : 1);
 76         date = new Date(Number(Number(date) + (offsetnum * 60000)));
 77     }
 78 
 79     var zeropad = function (num) { return ((num < 10) ? '0' : '') + num; };
 80 
 81     var str = "";
 82     str += date.getUTCFullYear();
 83     if (format > 1) { str += "-" + zeropad(date.getUTCMonth() + 1); }
 84     if (format > 2) { str += "-" + zeropad(date.getUTCDate()); }
 85     if (format > 3) {
 86         str += "T" + zeropad(date.getUTCHours()) +
 87                ":" + zeropad(date.getUTCMinutes());
 88     }
 89     if (format > 4) { str += ":" + zeropad(date.getUTCSeconds()); }
 90     if (format > 3) { str += offset; }
 91     if (format > 5) {
 92         str = date.getTime();
 93     }
 94     return str;
 95 };

 97 /**
 98  * Convert a file's size to a human readable size
 99  * example: 2301 = 2.301kB
100  *
101  * @param (Integer) filesize The file's size to convert
102  * @return (String) the file's size in human readable format
103  */
104 
105 sakai.api.Util.convertToHumanReadableFileSize = function(filesize) {
106     // Divide the length into its largest unit
107     var units = [[1024 * 1024 * 1024, 'GB'], [1024 * 1024, 'MB'], [1024, 'KB'], [1, 'bytes']];
108     var lengthunits;
109     for (var i = 0, j=units.length; i < j; i++) {
110 
111         var unitsize = units[i][0];
112         var unittext = units[i][1];
113 
114         if (filesize >= unitsize) {
115             filesize = filesize / unitsize;
116             // 1 decimal place
117             filesize = Math.ceil(filesize * 10) / 10;
118             lengthunits = unittext;
119             break;
120         }
121     }
122     // Return the human readable filesize (and localized)
123     return sakai.api.l10n.transformDecimal(filesize, 1) + " " + lengthunits;
124 };

126 /**
127  * Formats a comma separated string of text to an array of usable tags
128  * Filters out unwanted tags (eg empty tags)
129  * Returns the array of tags, if no tags were provided or none were valid an empty array is returned
130  *
131  * Example: inputTags = "tag1, tag2, , , tag3, , tag4" returns ["tag1","tag2","tag3","tag4"]
132  *
133  * @param {String} inputTags Unformatted, comma separated, string of tags put in by a user
134  * @return {Array} Array of formatted tags
135  */
136 sakai.api.Util.formatTags = function(inputTags){
137     if ($.trim(inputTags) !== "") {
138         var tags = [];
139         var splitTags = $(inputTags.split(","));
140         splitTags.each(function(index){
141             if ($.trim(splitTags[index]).length) {
142                 tags.push($.trim(splitTags[index]));
143             }
144         });
145         return tags;
146     }
147     else {
148         return [];
149     }
150 };

152 /**
153  * Add and delete tags from an entity
154  * The two arrays, newTags and currentTags, represent the state of tags on the entity
155  * newTags should be the tags that you want on the entity, the whole set
156  * currentTags should be the set of tags the entity had before the user modified it
157  * tagEntity will delete any tags in currentTags but not in newTags, and add any in
158  * newTags that aren't in currentTags
159  *
160  * @param (String) tagLocation the URL to the tag, ie. (~userid/public/authprofile)
161  * @param (Array) newTags The set of tags you wish to be on the entity
162  * @param (Array) currentTags The set of tags on the current entity
163  * @param (Function) callback The callback function
164  */
165 
166 sakai.api.Util.tagEntity = function(tagLocation, newTags, currentTags, callback) {
167     var setTags = function(tagLocation, tags, callback) {
168         if (tags.length) {
169             var requests = [];
170             $(tags).each(function(i, val){
171                 requests.push({
172                     "url": "/tags/" + val,
173                     "method": "POST",
174                     "parameters": {
175                         "sakai:tag-name": val,
176                         "sling:resourceType": "sakai/tag"
177                     }
178                 });
179             });
180             $.ajax({
181                 url: sakai.config.URL.BATCH,
182                 traditional: true,
183                 type: "POST",
184                 data: {
185                     requests: $.toJSON(requests)
186                 },
187                 success: function() {
188                     doSetTags(tags);
189                 },
190                 error: function(xhr, response){
191                     debug.error(val + " failed to be created");
192                     if ($.isFunction(callback)) {
193                         callback();
194                     }
195                 }
196             });
197         } else {
198             if ($.isFunction(callback)) {
199                 callback();
200             }
201         }
202 
203         // set the tag on the entity
204         var doSetTags = function(tags) {
205             var requests = [];
206             $(tags).each(function(i,val) {
207                 requests.push({
208                     "url": tagLocation,
209                     "method": "POST",
210                     "parameters": {
211                         "key": "/tags/" + val,
212                         ":operation": "tag"
213                     }
214                 });
215             });
216             $.ajax({
217                 url: sakai.config.URL.BATCH,
218                 traditional: true,
219                 type: "POST",
220                 data: {
221                     requests: $.toJSON(requests)
222                 },
223                 success: function() {
224                     if ($.isFunction(callback)) {
225                         callback();
226                     }
227                 },
228                 error: function(xhr, response){
229                     debug.error(tagLocation + " failed to be tagged as " + val);
230                 }
231             });
232         };
233     };
234 
235     /**
236      * Delete tags on a given node
237      *
238      * @param (String) tagLocation the URL to the tag, ie. (~userid/public/authprofile)
239      * @param (Array) tags Array of tags to be deleted from the entity
240      * @param (Function) callback The callback function
241      */
242 
243     var deleteTags = function(tagLocation, tags, callback) {
244         if (tags.length) {
245             var requests = [];
246             $(tags).each(function(i,val) {
247                 requests.push({
248                     "url": tagLocation,
249                     "method": "POST",
250                     "parameters": {
251                         "key": "/tags/" + val,
252                         ":operation": "deletetag"
253                     }
254                 });
255             });
256             $.ajax({
257                 url: sakai.config.URL.BATCH,
258                 traditional: true,
259                 type: "POST",
260                 data: {
261                     requests: $.toJSON(requests)
262                 },
263                 error: function(xhr, response) {
264                     debug.error(val + " tag failed to be removed from " + tagLocation);
265                 },
266                 complete: function() {
267                     if ($.isFunction(callback)) {
268                         callback();
269                     }
270                 }
271             });
272         } else {
273             if ($.isFunction(callback)) {
274                 callback();
275             }
276         }
277     };
278 
279     var tagsToAdd = [];
280     var tagsToDelete = [];
281     // determine which tags to add and which to delete
282     $(newTags).each(function(i,val) {
283         val = $.trim(val).replace(/#/g,"");
284         if (val && $.inArray(val,currentTags) == -1) {
285             if (sakai.api.Security.escapeHTML(val) === val && val.length) {
286                 if ($.inArray(val, tagsToAdd) < 0) {
287                     tagsToAdd.push(val);
288                 }
289             }
290         }
291     });
292     $(currentTags).each(function(i,val) {
293         val = $.trim(val).replace(/#/g,"");
294         if (val && $.inArray(val,newTags) == -1) {
295             if (sakai.api.Security.escapeHTML(val) === val && val.length) {
296                 if ($.inArray(val, tagsToDelete) < 0) {
297                     tagsToDelete.push(val);
298                 }
299             }
300         }
301     });
302     deleteTags(tagLocation, tagsToDelete, function() {
303         setTags(tagLocation, tagsToAdd, function() {
304             if ($.isFunction(callback)) {
305                 callback();
306             }
307         });
308     });
309 
310 };

404 /**
405  * Parse a ISO8601 date into a JavaScript date object.
406  *
407  * <p>
408  *     Supported date formats:
409  *     <ul>
410  *         <li>2010</li>
411  *         <li>2010-02</li>
412  *         <li>2010-02-18</li>
413  *         <li>2010-02-18T07:44Z</li>
414  *         <li>1997-07-16T19:20+01:00</li>
415  *         <li>1997-07-16T19:20:30+01:00</li>
416  *         <li>1269331220896</li>
417  *     </ul>
418  * </p>
419  *
420  * <p>
421  *     External links:
422  *     <ul>
423  *         <li><a href="http://www.w3.org/TR/NOTE-datetime">W3C datetime documentation</a></li>
424  *         <li><a href="http://delete.me.uk/2005/03/iso8601.html">ISO8601 JavaScript function</a></li>
425  *         <li><a href="http://confluence.sakaiproject.org/display/KERNDOC/KERN-643+Multiple+date+formats+in+the+back-end">Specification</a></li>
426  *     </ul>
427  * </p>
428  *
429  * @param {String|Integer} dateInput
430  *     The date that needs to be converted to a JavaScript date object.
431  *     If the format is in milliseconds, you need to provide an integer, otherwise a string
432  * @return {Date} JavaScript date object
433  */
434 sakai.api.Util.parseSakaiDate = function(dateInput) {
435 
436     // Define the regular expressions that look for the format of
437     // the dateInput field
438     var regexpInteger = /^\d+$/;
439     var regexpISO8601 = "([0-9]{4})(-([0-9]{2})(-([0-9]{2})" +
440         "(T([0-9]{2}):([0-9]{2})(:([0-9]{2})(\\.([0-9]+))?)?" +
441         "(Z|(([-+])([0-9]{2}):([0-9]{2})))?)?)?)?";
442 
443     // Test whether the format is in milliseconds
444     if(regexpInteger.test(dateInput) && typeof dateInput !== "string") {
445        return new Date(dateInput);
446     }
447 
448     // Test whether you get an ISO8601 format back
449     var d = dateInput.match(new RegExp(regexpISO8601));
450 
451     var offset = 0;
452     var date = new Date(d[1], 0, 1);
453     var dateOutput = new Date();
454 
455     if (d[3]) { date.setMonth(d[3] - 1); }
456     if (d[5]) { date.setDate(d[5]); }
457     if (d[7]) { date.setHours(d[7]); }
458     if (d[8]) { date.setMinutes(d[8]); }
459     if (d[10]) { date.setSeconds(d[10]); }
460     if (d[12]) { date.setMilliseconds(Number("0." + d[12]) * 1000); }
461     if (d[14]) {
462         offset = (Number(d[16]) * 60) + Number(d[17]);
463         offset *= ((d[15] === '-') ? 1 : -1);
464     }
465 
466     // Set the timezone for the date object
467     offset -= date.getTimezoneOffset();
468     var time = (Number(date) + (offset * 60 * 1000));
469     dateOutput.setTime(Number(time));
470 
471     // Return the date output
472     return dateOutput;
473 };

476 /**
477  * Parse an RFC822 date into a JavaScript date object.
478  * Examples of RFC822 dates:
479  * Wed, 02 Oct 2002 13:00:00 GMT
480  * Wed, 02 Oct 2002 13:00:00 +0200
481  *
482  * @param {String} dateString
483  * @return {Date}  a Javascript date object
484  */
485 sakai.api.Util.parseRFC822Date = function(dateString) {
486 
487     var dateOutput = new Date();
488 
489     var dateElements = dateString.split(" ");
490     var dateElementsTime = dateElements[4].split(":");
491 
492     var months = {"Jan":0,"Feb":1,"Mar":2,"Apr":3,"May":4,"Jun":5,"Jul":6,"Aug":7,"Sep":8,"Oct":9,"Nov":10,"Dec":11};
493     var zones = {
494         "UT": 0,
495         "GMT": 0,
496         "EST": -5,
497         "EDT": -4,
498         "CST": -6,
499         "CDT": -5,
500         "MST": -7,
501         "MDT": -6,
502         "PST": -8,
503         "PDT": -7,
504         "Z": 0,
505         "A":-1,
506         "M":-12,
507         "N": 1,
508         "Y": 12
509     };
510 
511     // Set day
512     if (dateElements[1]) { dateOutput.setDate(Number(dateElements[1])); }
513 
514     //Set month
515     if (dateElements[2]) { dateOutput.setMonth(months[dateElements[2]]); }
516 
517     // Set year
518     if (dateElements[3]) { dateOutput.setFullYear(Number(dateElements[3])); }
519 
520     // Set hour
521     if (dateElements[4] && dateElementsTime[0]) {
522 
523         // Take timezone offset into account
524         if (zones[dateElements[5]]) {
525             dateOutput.setHours(dateElementsTime[0] + zones[dateElements[5]]);
526         } else if (!isNaN(parseInt(dateElements[5], 10))) {
527             var zoneTimeHour = Number(dateElements[5].substring(0,2));
528             dateOutput.setHours(Number(dateElementsTime[0]) + zoneTimeHour);
529         }
530     }
531 
532     // Set minutes
533     if (dateElements[4] && dateElementsTime[1]) {
534 
535         if (!isNaN(parseInt(dateElements[5], 10))) {
536             var zoneTimeMinutes = Number(dateElements[5].substring(3,4));
537             dateOutput.setMinutes(Number(dateElementsTime[1]) + zoneTimeMinutes);
538         } else {
539             dateOutput.setMinutes(Number(dateElementsTime[1]));
540         }
541     }
542 
543     // Set seconds
544     if (dateElements[4] && dateElementsTime[2]) { dateOutput.setSeconds(dateElementsTime[2]); }
545 
546     return dateOutput;
547 
548 };

552  * Removes JCR or Sling properties from a JSON object
553  * @param {Object} i_object The JSON object you want to remove the JCR object from
554  * @returns void
555  */
556 sakai.api.Util.removeJCRObjects = function(i_object) {
557 
558     if (i_object["jcr:primaryType"]) {
559         delete i_object["jcr:primaryType"];
560     }
561 
562     if (i_object["jcr:created"]) {
563         delete i_object["jcr:created"];
564     }
565 
566     if (i_object["jcr:createdBy"]) {
567         delete i_object["jcr:createdBy"];
568     }
569 
570     if (i_object["jcr:mixinTypes"]) {
571         delete i_object["jcr:mixinTypes"];
572     }
573 
574     // Loop through keys and call itself recursively for the next level if an object is found
575     for (var i in i_object) {
576         if (i_object.hasOwnProperty(i) && $.isPlainObject(i_object[i])) {
577           sakai.api.Util.removeJCRObjects(i_object[i]);
578         }
579     }

619 (function($){
620     sakai.api.Util.include = {};
621     /**
622      * Generic function that will insert an HTML tag into the head of the document. This
623      * will be used to both insert CSS and JS files
624      * @param {Object} tagname
625      *  Name of the tag we want to insert. This is supposed to be "link" or "script".
626      * @param {Object} attributes
627      *  A JSON object that contains all of the attributes we want to attach to the tag we're
628      *  inserting. The keys in this object are the attribute names, the values in the object
629      *  are the attribute values
630      */
631     var insertTag = function(tagname, attributes) {
632         var tag = document.createElement(tagname);
633         var head = document.getElementsByTagName('head').item(0);
634         for (var a in attributes){
635             if(attributes.hasOwnProperty(a)){
636                 tag[a] = attributes[a];
637             }
638         }
639         head.appendChild(tag);
640     };
641 
642     /**
643      * Check to see if the tag+attributes combination currently exists in the DOM
644      *
645      * @param {Object} tagname
646      *  Name of the tag we want to insert. This is supposed to be "link" or "script".
647      * @param {Object} attributes
648      *  A JSON object that contains all of the attributes we want to attach to the tag we're
649      *  inserting. The keys in this object are the attribute names, the values in the object
650      *  are the attribute values
651      * @return {jQuery|Boolean} returns the selected objects if found, otherwise returns false
652      */
653     var checkForTag = function(tagname, attributes) {
654         var selector = tagname;
655         for (var i in attributes) {
656             if (i && attributes.hasOwnProperty(i)) {
657                 selector += "[" + i + "*=" + attributes[i] + "]";
658             }
659         }
660         if ($(selector).length) {
661             return $(selector);
662         }
663         else {
664             return false;
665         }
666     };
667 
668     /**
669      * Load a JavaScript file into the document
670      * @param {String} URL of the JavaScript file relative to the parent dom.
671      */
672     sakai.api.Util.include.js = function(url) {
673         var attributes = {"src": url, "type": "text/javascript"};
674         var existingScript = checkForTag("script", attributes);
675         if (existingScript) {
676             // Remove the existing script so we can place in a new one
677             // We need to do this otherwise the init functions that need to be called
678             // at the end of widgets never get called again
679             existingScript.remove();
680         }

702  * @description
703  * Sorting algorithms
704  *
705  * @namespace
706  * Sorting functions
707  */
708 sakai.api.Util.Sorting = {
709 
710     /**
711     * Natural sorting algorithm, for sorting file lists etc.
712     * @example sakai.api.Util.Sorting("z1", "z2", "z01");
713     * @param {String|Integer|Number} a The first element you want to sort
714     * @param {String|Integer|Number} b The second element you want to sort
715     * @return {Integer} [0 | 1 | -1]
716     *     <ul>
717     *         <li>-1: sort a so it has a lower index than b</li>
718     *         <li>0: a and b are equal</li>
719     *         <li>1: sort b so it has a lower index than a</li>
720     *     </ul>
721     */
722    naturalSort: function(a, b) {
723 
724         /*
725          * Natural Sort algorithm for Javascript
726          * Version 0.3
727          * Author: Jim Palmer (based on chunking idea from Dave Koelle)
728          *  optimizations and safari fix by Mike Grier (mgrier.com)
729          * Released under MIT license.
730          * http://code.google.com/p/js-naturalsort/source/browse/trunk/naturalSort.js
731          */
732 
733         // Setup temp-scope variables for comparison evalutation
734         var re = /(-?[0-9\.]+)/g,
735             x = a.toString().toLowerCase() || '',
736             y = b.toString().toLowerCase() || '',
737             nC = String.fromCharCode(0),
738             xN = x.replace( re, nC + '$1' + nC ).split(nC),
739             yN = y.replace( re, nC + '$1' + nC ).split(nC),
740             xD = (new Date(x)).getTime(),
741             yD = xD ? (new Date(y)).getTime() : null;
742         // Natural sorting of dates
743         if (yD) {
744             if (xD < yD) { return -1; }
745             else if (xD > yD) { return 1; }
746         }
747         // Natural sorting through split numeric strings and default strings
748         for( var cLoc = 0, numS = Math.max(xN.length, yN.length); cLoc < numS; cLoc++ ) {
749             var oFxNcL = parseFloat(xN[cLoc]) || xN[cLoc];
750             var oFyNcL = parseFloat(yN[cLoc]) || yN[cLoc];
751             if (oFxNcL < oFyNcL) { return -1; }
752             else if (oFxNcL > oFyNcL) { return 1; }
753         }
754         return 0;
755    }
