 23 var AIM = {
 24 
 25     frame : function(c) {
 26         var n = 'f' + Math.floor(Math.random() * 99999);
 27         var d = document.createElement('DIV');
 28         d.innerHTML = '<iframe style="display:none" src="about:blank" id="'+n+'" name="'+n+'" onload="AIM.loaded(\''+n+'\')"></iframe>';
 29         document.body.appendChild(d);
 30 
 31         var i = document.getElementById(n);
 32         if (c && typeof(c.onComplete) === 'function') {
 33             i.onComplete = c.onComplete;
 34         }
 35         return n;
 36     },
 37 
 38     form : function(f, name) {
 39         f.setAttribute('target', name);
 40     },
 41 
 42     submit : function(f, c) {
 43         AIM.form(f, AIM.frame(c));
 44         if (c && typeof(c.onStart) === 'function') {
 45             return c.onStart();
 46         } else {
 47             return true;
 48         }
 49     },
 50 
 51     loaded : function(id) {
 52         var i = document.getElementById(id);
 53         var d = null;
 54         if (i.contentDocument) {
 55             d = i.contentDocument;
 56         } else if (i.contentWindow) {
 57             d = i.contentWindow.document;
 58         } else {
 59             d = window.frames[id].document;
 60         }
 61         if (d.location.href === "about:blank") {
 62             return;
 63         }
 64 
 65         if (typeof(i.onComplete) === 'function') {
 66             i.onComplete(d.body.innerHTML);
 67         }
 68     }
 69 };

198     /**
199      * When the user has drawn a square this function will be called by imgAreaSelect.
200      * This will draw the thumbnail by modifying it's css values.
201      * @param {Object} img    The thumbnail
202      * @param {Object} selection The selection object from imgAreaSelect
203      */
204     function preview(img, selection){
205         // Save the user his selection in a global variable.
206         userSelection = selection;
207 
208         // How much has the user scaled down the image?
209         var scaleX = thumbnailWidth / (selection.width || 1);
210         var scaleY = thumbnailHeight / (selection.height || 1);
211 
212         // Change the thumbnail according to the user his selection via CSS.
213         $(thumbnail).css({
214             width: Math.round(scaleX * img.width) + 'px',
215             height: Math.round(scaleY * img.height) + 'px',
216             marginLeft: '-' + Math.round(scaleX * selection.x1) + 'px',
217             marginTop: '-' + Math.round(scaleY * selection.y1) + 'px'
218         });
219     }

237     /**
238      * On changepic form submit, check that a file has been selected
239      * and submit the form.

241     $(picForm).submit(function () {
242         // validate args
243         // file extension allow for image
244         var extensionArray = [".png", ".jpg", ".jpeg",".gif"];
245         // get file name
246         fileName = $(picInput).val();
247         // get extension from the file name.
248         var extension = fileName.slice(fileName.lastIndexOf(".")).toLowerCase();
249         var allowSubmit = false;
250 
251         for (var i = 0; i < extensionArray.length; i++) {
252              // extension is acceptable image format
253              if (extensionArray[i] === extension) {
254                 allowSubmit = true;
255                 break;
256              }
257         }
258         // if image format is acceptable
259         if(allowSubmit) {
260             $(picInputError).hide();
261             $(uploadNewButtons).hide();
262             $(uploadProcessing).show();
263             fileName = "tmp" + new Date().getTime() + ".jpg";
264             $(profilePicture).attr("name",fileName);
265             return AIM.submit(this, {
266                 'onStart' : sakai.changepic.startCallback,
267                 'onComplete' : sakai.changepic.completeCallback
268             });
269         } else {
270             // no input, show error
271             $(picInputError).show();
272             return false;
273         }

276     sakai.changepic.doInit = function(newpic){
277         picture = false;
278 
279         if (sakai.api.UI.changepic.mode && sakai.api.UI.changepic.id) {
280             id = sakai.api.UI.changepic.id;
281             mode = sakai.api.UI.changepic.mode;
282         } else {
283             id = sakai.data.me.user.userid;
284         }
285 
286         var json;
287 
288         if (mode === "group") {
289             // fetch group data to check if it has a picture
290             $.ajax({
291                 url: "/~" + id + "/public.infinity.json",
292                 async: false,
293                 success: function(data){
294                     json = data.authprofile;
295                 }
296             });
297         } else {
298             // Check whether there is a base picture at all
299             me = sakai.data.me;
300             //var json = me.profile;
301             json = me.profile;
302         }
303 
304         // If the image is freshly uploaded then reset the imageareaobject to reset all values on init
305         if (newpic) {
306             resetUploadField();
307             imageareaobject = null;
308             picture = {
309                 "_name": fileName,
310                 "selectedx1":0,
311                 "selectedy1":0,
312                 "selectedx2":64,
313                 "selectedy2":64
314             };
315         }
316         else if (json.picture) {
317             picture = $.parseJSON(json.picture);
318         }
319 
320         $(picForm).attr("action", "/~" + id + "/public/profile");
321 
322         // Get the preferred size for the thumbnail.
323         var prefThumbWidth = parseInt($(thumbnailContainer).css("width").replace(/px/gi,""), 10);
324         var prefThumbHeight = parseInt($(thumbnailContainer).css("height").replace(/px/gi,""), 10);
325 
326         // Make sure we don't have 0
327         thumbnailWidth  = (prefThumbWidth > 0) ? prefThumbWidth : thumbnailWidth;
328         thumbnailHeight  = (prefThumbHeight > 0) ? prefThumbHeight : thumbnailHeight;
329 
330         if (picture && picture._name) {
331             // The user has already uploaded a picture.
332             // Show the edit tab.
333             // Show tab in header
334             $(tabSelect).show();
335 
336             // Set the unvisible image to the full blown image. (make sure to filter the # out)
337             $(pictureMeasurer).html(sakai.api.Security.saneHTML("<img src='" + "/~" + id + "/public/profile/" + picture._name + "?sid=" + Math.random() + "' id='" + pictureMeasurerImage.replace(/#/gi, '') + "' />"));
338 
339             // Check the current picture's size
340             $(pictureMeasurerImage).bind("load", function(ev){
341 
342                 // save the image size in global var.
343                 realw = $(pictureMeasurerImage).width();
344                 realh = $(pictureMeasurerImage).height();
345 
346                 // Set the images
347                 $(fullPicture).attr("src", "/~" + id + "/public/profile/" + picture._name + "?sid=" + Math.random());
348                 $(thumbnail).attr("src", "/~" + id + "/public/profile/" + picture._name + "?sid=" + Math.random());
349 
350                 // Reset ratio
351                 ratio = 1;
352 
353                 // Width < 500 ; Height < 300 => set the original height and width
354                 if (realw < 500 && realh < 300){
355                     $(fullPicture).width(realw);
356                     $(fullPicture).height(realh);
357 
358                 // Width > 500 ; Height < 300 => Width = 500
359                 } else if (realw > 500 && (realh / (realw / 500) < 300)){
360                     ratio = realw / 500;
361                     $(fullPicture).width(500);
362                     $(fullPicture).height(Math.floor(realh / ratio));
363 
364                 // Width < 500 ; Height > 300 => Height = 300
365                 } else if (realh > 300 && (realw / (realh / 300) < 500)) {
366                     ratio = realh / 300;
367                     $(fullPicture).height(300);
368                     $(fullPicture).width(Math.floor(realw / ratio));
369 
370                 // Width > 500 ; Height > 300
371                 } else if (realh > 300 && (realw / (realh / 300) > 500)) {
372 
373                     var heightonchangedwidth = realh / (realw / 500);
374                     if (heightonchangedwidth > 300){
375                         ratio = realh / 300;
376                         $(fullPicture).height(300);
377                     } else {
378                         ratio = realw / 500;
379                         $(fullPicture).width(500);
380                     }
381                 }
382 
383                 var selectionObj = {
384                     width : picture.selectedx2 - picture.selectedx1,
385                     height :picture.selectedy2 - picture.selectedy1,
386                     x1 : picture.selectedx1,
387                     y1 : picture.selectedy1,
388                     x2 : picture.selectedx2,
389                     y2 : picture.selectedy2
390                 };
391 
392                 // Set the imgAreaSelect to a function so we can access it later on
393                 imageareaobject = $(fullPicture).imgAreaSelect({
394                     aspectRatio: "1:1",
395                     enable: true,
396                     show: true,
397                     instance: true,
398                     onInit: function(){
399                         // If the image gets loaded, make a first selection
400                         imageareaobject.setSelection(picture.selectedx1, picture.selectedy1, picture.selectedx2, picture.selectedy2);
401                         imageareaobject.setOptions({show: true, enable: true});
402                         imageareaobject.update();
403                         preview($("img" + fullPicture)[0], selectionObj);
404                     },
405                     onSelectChange: preview
406                 });
407             });
408             showSelectTab();
409 
410         }
411         else {
412             // The user hasn't uploaded a picture yet.
413             // Show the upload pic tab.
414             $(tabSelect).hide();
415             showNewTab();
416         }
417     };

424     // This is the function that will be called when a user has cut out a selection
425     // and saves it.
426     $(saveNewSelection).click(function(ev){
427 
428         if (!userSelection) {
429             userSelection = imageareaobject.getSelection();
430         }
431 
432         // The parameters for the cropit service.
433         var data = {
434             img: "/~" + id + "/public/profile/" + picture._name,
435             save: "/~" + id + "/public/profile",
436             x: Math.floor(userSelection.x1 * ratio),
437             y: Math.floor(userSelection.y1 * ratio),
438             width: Math.floor(userSelection.width * ratio),
439             height:Math.floor(userSelection.height * ratio),
440             dimensions: "256x256",
441             "_charset_":"utf-8"
442         };
443 
444         if(data.width === 0 || data.height === 0){
445             data.width = $(fullPicture).width();
446             data.height = $(fullPicture).height();
447             data.x = 0;
448             data.y = 0;
449         }
450 
451         // Post all of this to the server
452         $.ajax({
453             url: sakai.config.URL.IMAGE_SERVICE,
454             type: "POST",
455             data: data,
456             success: function(data){
457 
458                 var tosave = {
459                     "name": "256x256_" + picture._name,
460                     "_name": picture._name,
461                     "_charset_":"utf-8",
462                     "selectedx1" : userSelection.x1,
463                     "selectedy1" : userSelection.y1,
464                     "selectedx2" : userSelection.width + userSelection.x1,
465                     "selectedy2" : userSelection.height + userSelection.y1
466                 };
467 
468                 var stringtosave = $.toJSON(tosave);
469 
470                 sakai.data.me.profile.picture = stringtosave;
471 
472                 // Do a patch request to the profile info so that it gets updated with the new information.
473                 $.ajax({
474                     url: "/~" + id + "/public/authprofile.json",
475                     type : "POST",
476                     data : {
477                         "picture" : $.toJSON(tosave),
478                         "_charset_":"utf-8"
479                     },
480                     success : function(data) {
481                         // Change the picture in the page. (This is for my_sakai.html)
482                         // Math.random is for cache issues.
483                         for (var i = 0; i < imagesToChange.length;i++) {
484                             $(imagesToChange[i]).attr("src", "/~" + id + "/public/profile/" + tosave.name + "?sid=" + Math.random());
485                         }
486 
487                         // Hide the layover.
488                         $(container).jqmHide();
489 
490                     },
491                     error: function(xhr, textStatus, thrownError) {
492                         alert("An error has occured");
493                     }
494                 });
495 
496             },
497             error: function(xhr, textStatus, thrownError) {
498                 alert("An error has occured");
499             }
500         });
501 
502     });

526     };
