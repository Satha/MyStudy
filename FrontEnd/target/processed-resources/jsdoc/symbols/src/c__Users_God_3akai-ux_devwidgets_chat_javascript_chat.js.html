 80     /**
 81      * Load the list of your online contacts
 82      * make batch request for accepted friend and presence online
 83      * and close the chat window if contact has been deleted.
 84      * @param {Object} initial Whether or not this is the first time
 85      * the online friends are requested
 86      */
 87     var loadOnlineContacts = function(callback){
 88         var batchRequests = [];
 89         
 90         // accepted contacts
 91         var acceptedContacts = {
 92             "url":sakai.config.URL.CONTACTS_ACCEPTED + "?page=0&items=6",
 93             "method":"GET",
 94             "cache":false,
 95             "dataType":"json" 
 96         };
 97 
 98         // contacts online
 99         var contactsOnline = {
100             "url":sakai.config.URL.PRESENCE_CONTACTS_SERVICE,
101             "method":"GET",
102             "cache":false,
103             "dataType":"json" 
104         };
105         
106         batchRequests.push(acceptedContacts);
107         batchRequests.push(contactsOnline);
108 
109         $.ajax({
110         url: sakai.config.URL.BATCH,
111             type: "POST",
112             data: {
113                 requests: $.toJSON(batchRequests)
114             },
115             success: function(data){
116                 // contact list
117                 if (data.results.hasOwnProperty(0)) {
118                     acceptedContactList = $.parseJSON(data.results[0].body);
119                 }
120 
121                 // presence contacts
122                 if (data.results.hasOwnProperty(1)) {
123                     // load the list of contact onlines
124                     loadContactFinished($.parseJSON(data.results[1].body));
125                 }
126             },
127             complete: function(){
128                 if ($.isFunction(callback)) {
129                     callback();
130                 }
131             }
132         });
133 
134     };

153     /**
154      * Transform the JSON object received from the online contacts
155      * service into a trimmed down version that will be used to store
156      * and retrieve the chatwindows
157      * @param {Object} onlineContacts   Object received from the online
158      * contacts service
159      */
160     var transformContactsObject = function(onlineContacts){
161         var contactList = [];
162         for (var i = 0; i < onlineContacts.contacts.length; i++){
163             if (onlineContacts.contacts[i]["sakai:status"] !== "offline" && onlineContacts.contacts[i].profile.chatstatus !== "offline") {
164                 var contact = {};
165                 contact.profile = {
166                     "userid": onlineContacts.contacts[i].user,
167                     "name": sakai.api.User.getDisplayName(onlineContacts.contacts[i].profile),
168                     "status": onlineContacts.contacts[i].profile.status,
169                     "chatstatus": onlineContacts.contacts[i].profile.chatstatus || onlineContacts.contacts[i]["sakai:status"]
170                 };
171                 if (onlineContacts.contacts[i].profile.picture && $.parseJSON(onlineContacts.contacts[i].profile.picture).name) {
172                     contact.profile.picture = "/~" + onlineContacts.contacts[i].user + "/public/profile/" + $.parseJSON(onlineContacts.contacts[i].profile.picture).name;
173                 }
174                 contactList.push(contact);
175                 globalOnlineContactsLookup[contact.profile.userid] = contact.profile;
176             }
177         }
178         globalOnlineContacts = {"contacts": contactList, "sakaistatus": sakai.data.me.profile.chatstatus || onlineContacts["sakai:status"]};
179         return globalOnlineContacts;
180     };

324         $("#chat_with_" + userid + "_txt").focus();
325         // Scroll to the bottom
326         var chatwindow = $("#chat_with_" + userid + "_content");
327         chatwindow.attr("scrollTop", chatwindow.attr("scrollHeight"));
328         if (allMessages[userid] && allMessages[userid].length) {
329             var bulkRequests = [];
330             for (var i=0, j=allMessages[userid].length; i<j; i++) {
331                 var message = allMessages[userid][i];
332                 if (message["sakai:read"] === false) {
333                     bulkRequests.push(createBatchReadObject(message));
334                 }
335             }
336             for (var k=0, m=globalMessages.length; k<m; k++) {
337                 var msg = globalMessages[k];
338                 if (msg["userFrom"][0].userid === userid) {

360             globalChatWindows.push(window);
361             $("#chat_windows").append($.TemplateRenderer("chat_windows_template", window));
362             $("#chat_windows_container").append($.TemplateRenderer("chat_windows_windows_template", window));
363             // Position the chat window as the last chat window in the row
364             $("#chat_with_" + window.profile.userid).css("left", (window.index * 150) + "px");
365             if (openWindow) {
366                 closeAllChatWindows();
367                 openChatWindow(contactObject.profile.userid);
368             }
369         }

406     /**
407      * Update the chat windows so that the chat status and status
408      * message of their contacts is up to date. Also, disable all
409      * chat windows for which the contact is not online
410      */
411     var updateChatWindows = function(){
412         for (var i = 0; i < globalChatWindows.length; i++){
413             var contact = getOnlineContact(globalChatWindows[i].profile.userid);
414             var chatInputBox = $("#chat_with_" + globalChatWindows[i].profile.userid + "_txt");
415             // if contact has been deleted remove the chat window
416             if(!isContactExists(globalChatWindows[i].profile.userid)){
417                 removeChatWindow(globalChatWindows[i].profile.userid);
418             // The contact is offline. Disable the chat window and update the chat status
419             }else if (!contact || contact.profile.chatstatus === "offline"){
420                 // Set the chat status to offline
421                 setWindowChatStatus(globalChatWindows[i].profile.userid, "offline");
422                 // Disable the input box
423                 chatInputBox.attr("disabled", true);
424                 chatInputBox.val(globalChatWindows[i].profile.name + " is offline");
425             } else {
426                 // Check whether the chat status changed
427                 if (globalChatWindows[i].profile.chatstatus != contact.profile.chatstatus){
428                     // Make sure the input box is enabled
429                     if (globalChatWindows[i].profile.chatstatus === "offline") {
430                         chatInputBox.removeAttr("disabled").val("");
431                     }
432                     setWindowChatStatus(globalChatWindows[i].profile.userid, contact.profile.chatstatus);
433                 }
434                 // Check whether the statusmessage changed
435                 if (globalChatWindows[i].profile.status != contact.profile.status){
436                     setWindowStatusmessage(globalChatWindows[i].profile.userid, contact.profile.status);
437                 }
438             }
439         }
440     };

506      * Detect when a user wants to send a message to a user
507      * @param {Object} event
508      */
509     $(".chat_with_txt").live("keypress", function(event){
510         // Has the user pressed enter
511         if (event.keyCode == '13') {

535     /**
536      * Send a chat message to a given contact
537      * @param {Object} to    User id of the contact you are sending to
538      * @param {Object} messageText    Text of the message being sent
539      */
540     var sendMessage = function(to, messageText){
541         var date = new Date( );
542         var timestamp = date.getTime();
543         // Send a message to the other user
544         var message = {
545             "sakai:type": "chat",
546             "sakai:sendstate": "pending",
547             "sakai:messagebox": "outbox",
548             "sakai:to": "chat:" + to,
549             "sakai:from": sakai.data.me.user.userid,
550             "sakai:subject": "",
551             "sakai:body": messageText,
552             "sakai:category": "chat",
553             "sakai:timestamp": timestamp,
554             "_charset_": "utf-8"
555         };
556         $.ajax({
557             url: "/~" + sakai.data.me.user.userid + "/message.create.html",
558             data: message,
559             type: "POST"
560         });
561         // Append my message to the chat window
562         var sentDate = new Date();
563         // Localize date
564         sentDate.setTime(sentDate.getTime() + sentDate.getTimezoneOffset() * 60 * 1000);
565         sentDate.setTime(sentDate.getTime() + sakai.data.me.user.locale.timezone.GMT*60*60*1000);
566         appendMessage(sakai.data.me.user.userid, to, messageText, sentDate);
567 
568         // save the message data
569         message["userFrom"] = [];
570         message["userFrom"].push(sakai.data.me.profile);
571         message["userTo"] = [];
572         message["userTo"].push(globalOnlineContactsLookup[to]);
573         message["sentDate"] = sentDate;
574         globalMessages.push(message);
575     };

577     /**
578      * Append a message to a given chatwindow
579      * @param {Object} from    User id from the person sending the message
580      * @param {Object} window  User id of the user in which chat window the
581      * message should be appended
582      * @param {Object} messageText   Body text of the message
583      * @param {Object} sentDate      Date in which the message was sent (localized)
584      */
585     var appendMessage = function(from, window, messageText, sentDate){
586         var message = {};
587         if (from === sakai.data.me.user.userid){
588             message.name = "Me";
589         } else {
590             message.name = getChatWindow(window).profile.name;
591         }
592         message.time = sakai.api.l10n.transformTime(sentDate);
593         message.message = messageText;
594         var chatwindow = $("#chat_with_" + window + "_content");
595         chatwindow.append($.TemplateRenderer("chat_content_template", message));
596         // Scroll to the bottom
597         chatwindow.attr("scrollTop", chatwindow.attr("scrollHeight"));
598     };

601      * Function that will initiate the check for new messages at regular
602      * intervals
603      */
604     var checkNewMessages = function(){
605         // Only check for new messages when you have online contacts
606         if (globalOnlineContacts.contacts.length > 0){
607             loadNewMessages();
608         }
609         // Start polling regurarly to get new messages
610         if (!loadNewMessagesTimer) {
611 	        loadNewMessagesTimer = setInterval(checkNewMessages, loadNewMessagesInterval);
612         }
613     };

615     // Variable that keeps track of when we last checked for new messages
616     var lastCheckDate = new Date().getTime();
617 
618     /**
619      * Check whether there are new unread chat messages waiting
620      */
621     var loadNewMessages = function(){
622         $.ajax({
623             url: "/~" + sakai.data.me.user.userid + "/message.chatupdate.json",
624             cache: false,
625             data: {
626                 "t": lastCheckDate
627             },
628             success: function(resp1){
629                 // If there are new messages waiting, retrieve them
630                 if (resp1.update) {
631                     $.ajax({
632                         url: sakai.config.URL.CHAT_GET_SERVICE.replace(/__KIND__/, "unread"),
633                         data: {
634                             "_from": "*",
635                             "items": 1000,
636                             "t": resp1.pulltime,
637                             "sortOn": "sakai:created",
638                             "sortOrder": "descending"
639                         },
640                         cache: false,
641                         success: function(data){
642                             // Add the messages to the appropriate windows
643                             insertNewMessages(data);
644                             lastCheckDate = resp1.time;
645                         }
646                     });
647                 }
648             }
649         });
650     };

652     /**
653      * Callback function to sort messages based on timestamp
654      */
655     var sortMessages = function(a, b){
656         return a["sakai:timestamp"] > b["sakai:timestamp"] ? 1 : -1;
657     };

659     /**
660      * Once we know that there are new messages, we add them into
661      * the appropriate chat windows
662      * @param {Object} messages    List of new chat messages
663      */
664     var insertNewMessages = function(messages){
665         if (messages.results) {
666             // Sort messages based on timestamp
667             messages.results.sort(sortMessages);
668             var bulkRequests = [];
669             for (var i = 0; i < messages.results.length; i++) {
670                 var message = messages.results[i];
671                 var from = message.userFrom[0];
672                 // We don't add in our own messages as they are already in there.
673                 if (from.userid !== sakai.data.me.user.userid){
674                     // Check whether there is already a chatwindow for it
675                     var chatWindow = getChatWindow(from.userid);
676                     var sentDate = sakai.api.l10n.parseDateString(message["sakai:created"]);
677                     var messageText = message["sakai:body"];
678                     if (!chatWindow) {
679                         // If not, create a new chat window
680                         var contact = {};
681                         contact.profile = {
682                             "userid": from.userid,
683                             "name": sakai.api.User.getDisplayName(from),
684                             "status": from.status,
685                             "chatstatus": from.chatstatus || from["sakai:status"]
686                         };
687                         if (from.picture && $.parseJSON(from.picture).name) {
688                             from.picture = "/~" + from.userid + "/public/profile/" + $.parseJSON(from.picture).name;
689                         }
690                         appendChatWindow(contact, false);
691                     }
692                     // Append the message
693                     appendMessage(from.userid, from.userid, messageText, sentDate);
694                     // Pulse if it is not currently open
695                     chatWindow = getChatWindow(from.userid);
696                     if (!chatWindow.open && message["sakai:read"] === false) {
697                         $("#chat_online_button_" + from.userid).effect("pulsate", {times: 5}, 500);
698                     } else if (message["sakai:read"] !== true) {
699                         // the window is open, lets mark the message as read
700                         message["sakai:read"] = true;
701                         bulkRequests.push(createBatchReadObject(message));
702                     }
703                     allMessages[from.userid] = allMessages[from.userid] || [];
704                     allMessages[from.userid].push(message);
705                 }
706                 globalMessages.push(message);
707             }
708             // sent out the batch request saying the read messages are read
709             sendBatchReadRequests(bulkRequests);
710         }
711     };

713     /**
714      * Insert saved messages for this chat session, so they can persist across refreshes
715      * @param {Array} messages  the saved messages to insert into the chat windows
716      * @param {Array} validWindows  a list of windows to open if messages exist for them.
717      *                              if insertOwn === true and validWindows is present it'll only insert messages for the valid windows
718      */
719     var insertSavedMessages = function(messages, validWindows) {
720         for (var i = 0; i < messages.length; i++) {
721             var message = messages[i];
722             var from = message.userFrom[0];
723             if (from["rep:userId"] === sakai.data.me.user.userid) {
724                 from = message["userTo"][0];
725                 messageFrom = sakai.data.me.user.userid;
726             }
727             var chatWindow = getChatWindow(from.userid);
728             var sentDate;
729             if (message["sakai:created"]) {
730                 sentDate = sakai.api.l10n.parseDateString(message["sakai:created"]);
731             } else if (message["sentDate"]) {
732                 sentDate = new Date(message["sentDate"]);
733             }
734             if (validWindows && validWindows.length && $.inArray(from.userid, validWindows) > -1) {
735                 if (!chatWindow) {
736                     // If not, create a new chat window
737                     var contact = {};
738                     contact.profile = {
739                         "userid": from.userid,
740                         "name": from.name || sakai.api.User.getDisplayName(from),
741                         "status": from.status,
742                         "chatstatus": from.chatstatus || from["sakai:status"]
743                     };
744                     if (from.picture && $.parseJSON(from.picture).name) {
745                         from.picture = "/~" + from.userid + "/public/profile/" + $.parseJSON(from.picture).name;
746                     }
747                     appendChatWindow(contact, false);
748                 }
749                 // Append the message
750                 var messageText = message["sakai:body"];
751                 appendMessage(from.userid, from.userid, messageText, sentDate);
752                 // Pulse if it is not currently open
753                 chatWindow = getChatWindow(from.userid);
754                 if (!chatWindow.open && message["sakai:read"] === false) {
755                     $("#chat_online_button_" + from.userid).effect("pulsate", {times: 5}, 500);
756                 } else if (message["sakai:read"] !== true) {
757                     // the window is open, lets mark the message as read
758                     message["sakai:read"] = true;
759                 }
760                 globalMessages.push(message); // only push if this window is valid, so once they close it, its gone
761             }
762         }
763     };

786     /**
787      * Restore the state from the chat windows as specified in the persistent
788      * cookie
789      */
790     var restoreChatWindows = function(){
791         if ($.cookie('sakai_chat') && $.parseJSON($.cookie('sakai_chat'))) {
792             var storedState = $.parseJSON($.cookie('sakai_chat'));
793             var chatWindows = storedState.windows;
794             var validWindows = [];
795             for (var i = 0; i < chatWindows.length; i++) {
796 	            appendChatWindow({"profile": chatWindows[i].profile}, chatWindows[i].open);
797 	            validWindows.push(chatWindows[i].profile.userid);
798             }
799             // grab the session's saved chat messages if they exist
800             if (supportsSessionStorage) {
801                 var messages = $.parseJSON(sessionStorage.getItem('messages'));
802                 if (messages && messages.length) {
803                     insertSavedMessages(messages, validWindows);
804                 }
805             }
806         }
