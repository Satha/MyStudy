256     var sortCollectionByPosition = function() {
257         collectionData.collections.sort(function(a, b) {
258             var positionA = a.position;
259             var positionB = b.position;
260             if (positionA && positionB) {
261                 return positionA - positionB;
262             } else if (positionA) {
263                 return -1;
264             } else if (positionB) {
265                 return 1;
266             } else {
267                 return 0;
268             }
269         });

272     var parseState = function() {
273 
274         var collection = $.bbq.getState("collection");
275         var mode = $.bbq.getState("mode");
276         var pos = $.bbq.getState("pos");
277         var category = $.bbq.getState("category");
278         var item = $.bbq.getState("item");
279         var view = $.bbq.getState("view");
280         var fromShow = $.bbq.getState("fromShow");
281 
282         // set the state if its not set already
283         if (!view) {
284           if (settings.displayStyle == "mapView") {
285                 $.bbq.pushState({"view":"mapView"});
286                 return;
287             } else if (settings.displayStyle == "albumView") {
288                 $.bbq.pushState({"view":"albumView"});
289                 return;
290             }
291         }
292 
293         if (view == "albumView") {
294           if (item) {
295               selectedCollectionID = collection;
296               setCollectionData();
297               if (selectedCategoryID != category) {
298                   selectedCategoryID = category;
299                   viewCategory();
300               }
301               selectedItemID = item;
302               showItem();
303           } else if (category) {
304               hideEverything();
305               selectedCollectionID = collection;
306               setCollectionData();
307               selectedCategoryID = category;
308               viewCategory();
309           } else if (collection) {
310               hideEverything();
311               selectedCollectionID = collection;
312               viewAlbum();
313             } else {
314               hideEverything();
315               renderAlbumView();
316           }
317         } else if (view == "mapView") {
318               if (item) {
319                 hideEverything();
320                 selectedCollectionID = collection;
321                 setCollectionData();
322                 selectedItemID = item;
323                 if (mode === "edit") {
324                     editContent(item);
325                 } else {
326                     showContentItem();
327                 }
328               } else if (collection) {
329                 hideEverything();
330                 selectedCollectionID = collection;
331                 setCollectionData();
332                 if (mode === "edit") {
333                     editRoom(collection, fromShow, pos);
334                 } else {
335                     showRoom();
336                 }
337 
338               } else {
339                 hideEverything();
340                 renderMapView();
341             }
342         }
343 
344         // AES-256 security
345         // show any hidden options for editing if this person is a collaborator
346         if (sakai.show.canEdit()) {
347             $(".configure").show();
348         }
349     };

657                 if (item.mimeType && item.mimeType.split("/")[0] == "image") {
658                     $("<img/>")[0].src = item.url;

660                     if (!setFirstImage) {

663                         categoryImages[cat.id] = {};
664                         categoryImages[cat.id].currentImage = 0;
665                         categoryImages[cat.id].images = [];
666                     }
667                     categoryImages[cat.id].images.push(item.url);
668                 }
669             }
670             if (!categoryImages[cat.id]) {
671                 $("#category_" + cat.id + " div img").attr("src", "/dev/images/mimetypes/empty.png");
672             }
673         }
