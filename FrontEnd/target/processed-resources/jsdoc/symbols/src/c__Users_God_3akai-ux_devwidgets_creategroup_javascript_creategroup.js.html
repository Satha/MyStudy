 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.creategroup
 25  *
 26  * @class creategroup
 27  *
 28  * @description
 29  * Creategroup widget
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.creategroup = function(tuid, showSettings){
 36 
 37     /////////////////////////////
 38     // Configuration variables //
 39     /////////////////////////////
 40 
 41     var MAX_LENGTH = 30;
 42 
 43     // - ID
 44     var createGroup = "#creategroup";
 45 
 46     // Container
 47     var createGroupContainer = createGroup + "_container";
 48 
 49     // template containers
 50     var createGroupAddTemplate = "#noncourse_template_container";
 51 
 52     // Non course
 53     var createGroupAdd = createGroup + "_add";
 54     var createGroupAddCancel = createGroupAdd + "_cancel";
 55     var createGroupAddDescription = createGroupAdd + "_description";
 56     var createGroupAddId = createGroupAdd + "_id";
 57     var createGroupAddName = createGroupAdd + "_name";
 58     var createGroupAddProcess = createGroupAdd + "_process";
 59     var createGroupAddSave = createGroupAdd + "_save";
 60     var createGroupAddUrl = createGroupAdd + "_url";
 61     var createGroupAddUrlLength = createGroupAddUrl + "_length";
 62     var createGroupAddUrlMaxLength = createGroupAddUrl + "_max_length";
 63 
 64     // Error fields
 65     var createGroupAddNameEmpty = createGroupAddName + "_empty";
 66     var createGroupAddNameShort = createGroupAddName + "_short";
 67     var createGroupAddIdEmpty = createGroupAddId + "_empty";
 68     var createGroupAddIdTaken = createGroupAddId + "_taken";
 69     var createGroupAddIdShort = createGroupAddId + "_short";
 70     var errorFields = ".creategroup_error_msg";
 71 
 72     // CSS Classes
 73     var invalidFieldClass = "invalid";
 74 
 75     var changedURLManually = false;
 76 
 77     ///////////////////////
 78     // Utility functions //
 79     ///////////////////////
 80 
 81     /**
 82      * Public function that can be called from elsewhere
 83      * (e.g. chat and sites widget)
 84      * It initializes the creategroup widget and shows the jqmodal (ligthbox)
 85      */
 86     sakai.creategroup.initialise = function(){
 87         changedURLManually = false;
 88         $(".description_fields").show();
 89         $("#creategroup_add_container").show();
 90         $(createGroupAddTemplate).show();
 91         $(createGroupAddName).val("");
 92         $(createGroupAddDescription).val("");
 93         $(createGroupAddId).val("");
 94         $(createGroupContainer).jqmShow();
 95     };
 96 
 97     /**
 98      * Show or hide the process div and hide/shows the buttons
 99      * @param {Boolean} show
100      *     true: show the process div and hide the buttons
101      *     false: hide the process div and show the buttons
102      */
103     var showProcess = function(show){
104         if(show){
105             $(createGroupAddCancel).hide();
106             $(createGroupAddSave).hide();
107             $(createGroupAddProcess).show();
108         } else {
109             $(createGroupAddProcess).hide();
110             $(createGroupAddCancel).show();
111             $(createGroupAddSave).show();
112         }
113     };
114 
115     /**
116      * Replace or remove malicious characters from the string
117      * We use this function to modify the groupid
118      * String to test against: test :?=&;\/?@+$<>#%'"''{}|\\^[]'
119      * @param {Object} input The string where the characters need to be replaced
120      */
121     var replaceCharacters = function(input){
122 
123         input = $.trim(input); // Remove the spaces at the beginning and end of the id
124 
125         input = input.toLowerCase().replace(/ /g,"-");
126         input = input.toLowerCase().replace(/'/g,"");
127         input = input.toLowerCase().replace(/"/g,"");
128 
129         var regexp = new RegExp("[^a-z0-9]", "gi");
130         input = input.replace(regexp,"-");
131 
132         return input;
133     };
134 
135 
136     ////////////////////
137     // Error handling //
138     ////////////////////
139 
140     var resetErrorFields = function(){
141         $("input").removeClass(invalidFieldClass);
142         $("textarea").removeClass(invalidFieldClass);
143         $(errorFields).hide();
144     };
145 
146     /**
147      * Function that will visually mark a form field as an
148      * invalid field.
149      * @param String field
150      *  JQuery selector of the input box we want to show as invalid
151      * @param String errorField
152      *  JQuery selector of the error message that needs to be shown.
153      * @param boolean noReset
154      *  Parameter that specifies whether we need to make all of the
155      *  fiels valid again first
156      */
157     var setError = function(field,errorField, noReset){
158         if (!noReset) {
159             resetErrorFields();
160         }
161         $(field).addClass(invalidFieldClass);
162         $(errorField).show();
163     };
164 
165     var myClose = function(hash) {
166         resetErrorFields();
167         hash.o.remove();
168         hash.w.hide();
169     };
170 
171 
172     ///////////////////
173     // Create a group //
174     ///////////////////
175 
176     var saveGroup = function(){
177         // Get the values from the input text and radio fields
178         var grouptitle = $(createGroupAddName).val() || "";
179         var groupdescription = $(createGroupAddDescription).val() || "";
180         var groupid = replaceCharacters($(createGroupAddId).val());
181         var inputError = false;
182 
183         if($("#creategroup_form").valid()) {
184             // Hide the buttons and show the process status
185             showProcess(true);
186             sakai.api.Groups.createGroup(groupid, grouptitle, groupdescription, function(success, nameTaken) {
187                 if (success) {
188                     document.location = "/~" + groupid;
189                 } else if (!success) {
190                     if (nameTaken) {
191                         setError(createGroupAddId,createGroupAddIdTaken,true);
192                     }
193                     showProcess(false);
194                 }
195             });
196         }
197         else {
198             return;
199         }
200     };
201 
202     ////////////////////
203     // Event Handlers //
204     ////////////////////
205 
206     /*
207      * Add jqModal functionality to the container.
208      * This makes use of the jqModal (jQuery Modal) plugin that provides support
209      * for lightboxes
210      */
211     $(createGroupContainer).jqm({
212         modal: true,
213         overlay: 20,
214         toTop: true,
215         onHide: myClose
216     });
217 
218     /*
219      * Add binding to the save button (create the group when you click on it)
220      */
221     $(createGroupAddSave).click(function(){
222         var entered = replaceCharacters($(createGroupAddId).val());
223         $(createGroupAddId).val(entered);
224         saveGroup();
225     });
226 
227     /*
228      * When you change something in the name of the group, it first removes the bad characters
229      * and then it shows the edited url in the span
230      */
231     $(createGroupAddName).bind("keyup change", function(ev){
232         if (!changedURLManually) { // only change the url if the user hasn't changed it themselves
233             var entered = replaceCharacters($(this).val());
234             $(createGroupAddId).val(entered);
235             resetErrorFields();
236         }
237     });
238 
239     $(createGroupAddId).bind("keyup change", function(e) {
240         changedURLManually = true;
241         var entered = replaceCharacters($(this).val());
242         $(createGroupAddId).val(entered);
243         resetErrorFields();
244     });
245 
246     $("#creategroup_form").validate({
247         errorClass: "creategroup_error_msg",
248         errorElement: "span",
249         highlight: function(element, errorClass) {
250             $(element).addClass("invalid");
251         },
252         unhighlight: function(element, errorClass, validClass) {
253             $(element).removeClass("invalid");
254         },
255         onclick:false,
256         onkeyup:false,
257         onfocusout:false,
258         rules: {
259             creategroup_add_name: {
260                 required: true,
261                 minlength: 3
262             },
263             creategroup_add_id: {
264                 required: true,
265                 minlength: 3
266             }
267         },
268         messages: {
269             creategroup_add_name:  {
270                 required: $(createGroupAddNameEmpty).text(),
271                 minlength: $(createGroupAddNameShort).text()
272             },
273             creategroup_add_id: {
274                 required: $(createGroupAddIdEmpty).text(),
275                 minlength: $(createGroupAddIdShort).text()
276             }
277         }
278     });
279 
280 
281     /////////////////////////////
282     // Initialisation function //
283     /////////////////////////////
284 
285     var doInit = function(){
286         changedURLManually = false;
287         // Hide error fields at start
288         $(errorFields).hide();
289 
290         // Set the text of the span containing the url of the current group
291         // e.g. http://celestine.caret.local:8080/~
292         var url = document.location.protocol + "//" + document.location.host;
293         url += "/~";
294         // get max length value
295         var maxLength = parseInt(MAX_LENGTH,10);
296 
297         // get length
298 
299         // if url is too long greater than 30 character
300         // show only first 15 characters +...+ last 15 characters
301         // e.g.http://sakai3-demo.uits.indiana.edu:8080/~
302         // it will change to shorter form:
303         // http://sakai3-...diana.edu:8080/~
304         if (url.length > maxLength) {
305             url = url.substr(0,15)+ "..."+ url.substr(url.length-15,url.length-1);
306         }
307 
308         $(createGroupAddUrl).text(sakai.api.Security.saneHTML(url));
309     };
310 
311     doInit();
312 };
313 
314 sakai.api.Widgets.widgetLoader.informOnLoad("creategroup");
