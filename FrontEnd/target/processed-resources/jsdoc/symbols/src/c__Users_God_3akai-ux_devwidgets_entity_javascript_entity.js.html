236     /**
237      * Gets the number of members and managers in a group and determines users role
238      */
239     var getGroupMembersManagers = function(){
240         var requests = []; // Array used to contain all the information we need to send to the batch post
241         var groupid = entityconfig.data.profile["sakai:group-id"];
242         requests[0] = {
243             "url": "/system/userManager/group/" + groupid + ".members.json",
244             "method": "GET"
245         };
246         requests[1] = {
247             "url": "/system/userManager/group/" + groupid + "-managers.members.json",
248             "method": "GET"
249         };
250 
251         $.ajax({
252             url: sakai.config.URL.BATCH,
253             traditional: true,
254             type: "POST",
255             data: {
256                 requests: $.toJSON(requests)
257             },
258             async: false,
259             success: function (data) {
260                 var groupMembers = $.parseJSON(data.results[0].body);
261                 var groupManagers = $.parseJSON(data.results[1].body);
262 
263                 // set the number of members and managers in this group
264                 entityconfig.data.profile["memberCount"] = groupMembers.length;
265                 entityconfig.data.profile["managerCount"] = groupManagers.length;
266 
267                 // set whether the current user's role
268                 if (sakai.api.Groups.isCurrentUserAManager(groupid)) {
269                     // current user is a manager
270                     entityconfig.data.profile["role"] = "manager";
271                 } else if (sakai.api.Groups.isCurrentUserAMember(groupid)) {
272                     // current user must be a member and not a manager
273                     // because of the structure of the if/else if
274                     entityconfig.data.profile["role"] = "member";
275                 } else {
276                     // current user is either anonymous or a logged-in non-member
277                     if (sakai.data.me.user.anon) {
278                         entityconfig.data.profile["role"] = "anon";
279                     } else {
280                         entityconfig.data.profile["role"] = "non-member";
281                     }
282                 }
283             }
284         });

447     /**
448      * Check whether a user is already a contact, invited or pending
449      * @param {String} userid    the user's userid
450      */
451     var checkContact = function(userid){
452         // Do a batch request to get contacts, invited and pending
453         var reqs = [
454             {
455                 "url" : "/var/contacts/accepted.json?page=0&items=100",
456                 "method" : "GET"
457             },
458             {
459                 "url" : "/var/contacts/invited.json?page=0&items=100",
460                 "method" : "GET"
461             },
462             {
463                 "url" : "/var/contacts/pending.json?page=0&items=100",
464                 "method" : "GET"
465             },
466             {
467                 "url" : sakai.config.URL.PRESENCE_CONTACTS_SERVICE,
468                 "method" : "GET"
469             }
470         ];
471 
472         var handleChatUpdate = function(e) {
473             var contactProfile = sakai.chat.getOnlineContact(userid).profile;
474             if (contactProfile && contactProfile.chatstatus) {
475                 if (entityconfig.data.profile.chatstatus !== contactProfile.chatstatus) {
476                     $("#entity_contact_" + entityconfig.data.profile.chatstatus).hide();
477                     entityconfig.data.profile.chatstatus = contactProfile.chatstatus;
478                     $("#entity_contact_" + entityconfig.data.profile.chatstatus).show();
479                 }
480             } else {
481                 $("#entity_contact_" + entityconfig.data.profile.chatstatus).hide();
482                 entityconfig.data.profile.chatstatus = "offline";
483                 $("#entity_contact_" + entityconfig.data.profile.chatstatus).show();
484             }
485         };
486 
487         $.ajax({
488             url: "/system/batch",
489             type: "POST",
490             data: {
491                 "requests": $.toJSON(reqs)
492             },
493             success: function(data){
494                 var presence = $.parseJSON(data.results[3].body);
495                 for (var l in presence.contacts){
496                     if (presence.contacts[l].user === userid){
497                         if (presence.contacts[l].profile.chatstatus && presence.contacts[l]["sakai:status"] === "online") {
498                             entityconfig.data.profile.chatstatus = presence.contacts[l].profile.chatstatus;
499                         } else {
500                             entityconfig.data.profile.chatstatus = "offline";
501                         }
502                         $("#entity_contact_" + entityconfig.data.profile.chatstatus).show();
503 
504                         // Add binding to chat status updates for the contact
505                         $(window).bind("sakai-chat-update", handleChatUpdate);
506                     }
507                 }
508                 var contacts = $.parseJSON(data.results[0].body);
509                 for (var i in contacts.results){
510                     if (contacts.results[i].target === userid){
511                         $("#entity_contact_accepted").show();
512                         return true;
513                     }
514                 }
515                 var invited = $.parseJSON(data.results[1].body);
516                 for (var j in invited.results){
517                     if (invited.results[j].target === userid){
518                         $("#entity_contact_invited").show();
519                         return true;
520                     }
521                 }
522                 var pending = $.parseJSON(data.results[2].body);
523                 for (var k in pending.results){
524                     if (pending.results[k].target === userid){
525                         $("#entity_contact_pending").show();
526                         return true;
527                     }
528                 }
529                 $("#entity_add_to_contacts").show();
530                 return true;
531             }
532         });
533     };

554     /**
555      * Add binding to the profile status elements
556      */
557     var addBindingProfileStatus = function(){
558 
559         // We need to reinitialise the jQuery objects after the rendering
560         $entity_profile_status = $("#entity_profile_status", $rootel);
561         $entity_profile_status_input = $("#entity_profile_status_input", $rootel);
562         $entity_profile_status_input_dummy = $("#entity_profile_status_input_dummy", $rootel);
563         $entity_profile_status_input_saving = $("#entity_profile_status_input_saving", $rootel);
564         $entity_profile_status_input_saving_failed = $("#entity_profile_status_input_saving_failed", $rootel);
565 
566         // Add the focus event to the profile status status
567         $entity_profile_status_input.bind("focus", function(){
568 
569             // If we don't have the dummy status (e.g. What are you doing) variable set yet, set it now.
570             if (!profile_dummy_status) {
571                 profile_dummy_status = $entity_profile_status_input_dummy.text();
572             }
573 
574             // Check whether the status field has the dummy class
575             if ($entity_profile_status_input.hasClass(entity_profile_status_input_dummy)) {
576 
577                 // Clear the current value for the input box and remove the dummy class
578                 $entity_profile_status_input.val("");
579                 $entity_profile_status_input.removeClass(entity_profile_status_input_dummy);
580             }
581 
582         });
583 
584         // Add the blur event to the profile status
585         $entity_profile_status_input.bind("blur", function(){
586 
587             // Check if it still has a dummy
588             if (!$entity_profile_status_input.hasClass(entity_profile_status_input_dummy) && $.trim($entity_profile_status_input.val()) === "") {
589 
590                 // Add the dummy class
591                 $entity_profile_status_input.addClass(entity_profile_status_input_dummy);
592 
593                 // Set the input value to the dummy status (e.g. What are you doing)
594                 $entity_profile_status_input.val(profile_dummy_status);
595 
596             }
597         });
598 
599         // Add the submit event to the status form
600         $entity_profile_status.unbind("submit");
601         $entity_profile_status.bind("submit", function(){
602 
603             // Get the correct input value from the user
604             var inputValue = $entity_profile_status_input.hasClass(entity_profile_status_input_dummy) ? "" : $.trim($entity_profile_status_input.val());
605 
606             // Escape html
607             inputValue = sakai.api.Security.escapeHTML(inputValue);
608 
609             if (!profile_status_value || profile_status_value !== inputValue) {
610                 profile_status_value = inputValue;
611 
612                 if (sakai.data.me.profile.activity) {
613                     delete sakai.data.me.profile.activity;
614                 }
615 
616                 var profileData = $.extend(true, {}, sakai.data.me.profile, {"status": inputValue});
617 
618                 sakai.api.Server.filterJCRProperties(profileData);
619 
620                 var originalText = $("button span", $entity_profile_status).text();
621                 $("button span", $entity_profile_status).text(sakai.api.Security.saneHTML($entity_profile_status_input_saving.text()));
622 
623                 sakai.api.Server.saveJSON(authprofileURL, profileData, function(success, data) {
624                     if (success) {
625                         // Set the button back to it's original text
626                         $("button span", $entity_profile_status).text(sakai.api.Security.saneHTML(originalText));
627 
628                         // Create an activity item for the status update
629                         var nodeUrl = authprofileURL;
630                         var activityMsg = "Status: " + inputValue;
631 
632                         var activityData = {
633                             "sakai:activityMessage": activityMsg
634                         };
635 
636                         sakai.data.me.profile = $.extend(true, sakai.data.me.profile, {"status": inputValue});
637 
638                         //trigger chat_status_message_change to update the status message on chat widget.
639                         $(window).trigger("chat_status_message_change", inputValue);
640 
641                         sakai.api.Activity.createActivity(nodeUrl, "status", "default", activityData);
642                     } else {
643                         // Log an error message
644                         debug.error("Entity widget - the saving of the profile status failed");
645                         profile_status_value = "";
646 
647                         // Show the message about a saving that failed to the user
648                         $("button span", $entity_profile_status).text(sakai.api.Security.saneHTML($entity_profile_status_input_saving_failed.text()));
649 
650                         // Show the origin text after 5 min
651                         window.setTimeout(function(){
652                             $("button span", $entity_profile_status).text(sakai.api.Security.saneHTML(originalText));

719     /**
720      * Add binding to elements related to tag drop down
721      */
722     var addBindingGroup = function(){
723         // Add the click event to the leave group button
724         $(entityGroupLeave).bind("click", function () {
725             leaveGroup();
726         });
727 
728         // Add the click event to the join group button
729         $(entityGroupJoin).bind("click", function(){
730             joinGroup();
731         });
732 
733         // Add the click event to the join group request button
734         $(entityGroupJoinRequest).bind("click", function(){
735             requestJoinGroup();
736         });
737 
738         // determine which button to display to the current user
739         var groupid = entityconfig.data.profile["sakai:group-id"];
740         var joinability = entityconfig.data.profile["sakai:group-joinable"];
741         var role = entityconfig.data.profile.role;
742 
743         if (role === "member" || (role === "manager" && entityconfig.data.profile.managerCount > 1)) {
744             // we have either a group member or manager, but not the last group manager
745             showGroupMembershipButton("leave");
746         }
747         else if ((role === "manager" && entityconfig.data.profile.managerCount === 1) ||
748             (role === "non-member" && joinability ===
749                 sakai.config.Permissions.Groups.joinable.manager_add) || role === "anon") {
750             // we have either the last group manager or a non-member with
751             // joinability set to 'only managers can add' or an anonymous user
752             hideGroupMembershipButton();
753         }
754         else if (role === "non-member" &&
755             joinability === sakai.config.Permissions.Groups.joinable.user_direct) {
756             // we have a non-member with joinability set to 'users can join directly'
757             showGroupMembershipButton("join");
758         }
759         else if (role === "non-member" &&
760             joinability === sakai.config.Permissions.Groups.joinable.user_request) {
761             // we have a non-member with joinability set to 'users must request to join'
762 
763             // has this user already requested to join the group? Search the list of join requests
764             sakai.api.Groups.getJoinRequests(groupid, function (success, data) {
765                 if (success) {
766                     // search data
767                     var foundRequest = false;
768                     if (data && data.total && data.total > 0) {
769                         for (var i in data.results) {
770                             if (data.results.hasOwnProperty(i) &&
771                                 data.results[i].userid === sakai.data.me.user.userid) {
772                                 // this user has a pending join request for this group
773                                 foundRequest = true;
774                                 break;
775                             }
776                         }
777                     }
778                     if (foundRequest) {
779                         // user has a pending join request
780                         showGroupMembershipButton("pending");
781                     } else {
782                         // user has not requested to join
783                         showGroupMembershipButton("request");
784                     }
785                 } else {
786                     // not sure if this user has requested, show request button
787                     showGroupMembershipButton("request");
788                 }
789             },
790             false);  // this is an non-async call
791         }
792         else {
793             // unrecognized combination of user and joinability setting
794             hideGroupMembershipButton();
795         }
796     };

886     /**
887      * Set the data for the content object information
888      * @param {Object} data The data we need to parse
889      */
890     var setContentData = function(data){
891         if (!data) {
892             debug.warn("Entity widget - setContentData - the data parameter is invalid:'" + data + "'");
893             return;
894         }
895 
896         var filedata = data.data;
897         var jcr_content = filedata["jcr:content"];
898         var jcr_access = filedata["rep:policy"];
899 
900         entityconfig.data.profile = {};
901 
902         // Check whether there is a jcr:content variable
903         if (jcr_content) {
904 
905             // Set the person that last modified the resource
906             if (jcr_content["jcr:lastModifiedBy"]) {
907                 entityconfig.data.profile.lastmodifiedby = jcr_content["jcr:lastModifiedBy"];
908             }
909             // Set the last modified date
910             if (jcr_content["jcr:lastModified"]) {
911                 entityconfig.data.profile.lastmodified = $.timeago(new Date(jcr_content["jcr:lastModified"]));
912             }
913             // Set the size of the file
914             if (jcr_content[":jcr:data"]) {
915                 entityconfig.data.profile.filesize = sakai.api.Util.convertToHumanReadableFileSize(jcr_content[":jcr:data"]);
916             }
917             // Set the mimetype of the file
918             if (jcr_content["jcr:mimeType"]) {
919                 entityconfig.data.profile.mimetype = jcr_content["jcr:mimeType"];
920             }
921         }
922 
923         // Set file extension
924         entityconfig.data.profile.extension = filedata["sakai:fileextension"];
925 
926         // Check if user is a manager or viewer
927         entityconfig.data.profile["role"] = "viewer";
928         if (jcr_access) {
929             // check if user is a manager
930             $.each(jcr_access, function(index, resultObject){
931                 if (resultObject["rep:principalName"] === sakai.data.me.user.userid) {
932                     if ($.inArray("jcr:all", resultObject["rep:privileges"]) != 1) {
933                         entityconfig.data.profile["role"] = 'manager';
934                     }
935                 }
936             });
937         }
938 
939         // Set the created by and created (date) variables
940         if (filedata["jcr:createdBy"]) {
941             entityconfig.data.profile.createdby = filedata["jcr:createdBy"];
942         }
943 
944         if (filedata["jcr:created"]) {
945             entityconfig.data.profile.created = $.timeago(new Date(filedata["jcr:created"]));
946         }
947 
948         if (filedata["sakai:pooled-content-file-name"]) {
949             entityconfig.data.profile.name = filedata["sakai:pooled-content-file-name"];
950         }
951 
952         // If it's a URL then set the URL
953         if(filedata["sakai:pooled-content-url"]){
954             entityconfig.data.profile.url = filedata["sakai:pooled-content-url"];
955         }
956 
957         // If it's a URL then set the URL
958         if(filedata["sakai:pooled-content-url"]){
959             entityconfig.data.profile.revurl = filedata["sakai:pooled-content-revurl"];
960         }
961 
962         // Set the path of the resource
963         if(data.url){
964             entityconfig.data.profile.path = data.url;
965         }
966         // Set the contentpath of the resource
967         if(data.url){
968             entityconfig.data.profile.contentpath = data.contentpath;
969         }
970 
971         // Set the description of the resource
972         if (filedata["sakai:description"]) {
973             entityconfig.data.profile.description = filedata["sakai:description"];
974         }
975 
976         // Set the tags of the resource
977         if (filedata["sakai:tags"]) {
978             entityconfig.data.profile['sakai:tags'] = filedata["sakai:tags"].toString();
979         }
980 
981         // Set the copyright of the file
982         if (filedata["sakai:copyright"]) {
983             entityconfig.data.profile.copyright = filedata["sakai:copyright"];
984         }
985 
986         if (document.location.pathname === "/dev/content_profile.html"){
987             entityconfig.data["link_name"] = false;
988         } else {
989             entityconfig.data["link_name"] = true;
990         }
991     };

993     /**
994      * Get the data for a specific mode
995      * @param {String} mode The mode you want to get the data for
996      * @param {Object} [data] The data you received from the page that called this (can be undefined)
997      * @param {Function} [callback] A callback function that will be fired it is supplied
998      */
999     var getData = function(mode, data){
1000 
1001         switch (mode) {
1002             case "profile":
1003                 entityconfig.data.profile = $.extend(true, {}, data);
1004                 // Set the correct profile data
1005                 setProfileData();
1006                 break;
1007             case "myprofile":
1008                 // Set the profile for the entity widget to the personal profile information
1009                 // We need to clone the sakai.data.me.profile object so we don't interfere with it
1010                 entityconfig.data.profile = $.extend(true, {}, sakai.data.me.profile);
1011                 //get data from sakai.data.me object and set in the entityconfig
1012                 setData();
1013                 // Set the correct profile data
1014                 setProfileData();
1015                 break;
1016             case "group":
1017                 // Set the profile for the entity widget to the group authprofile
1018                 entityconfig.data.profile = $.extend(true, {}, data.authprofile);
1019                 // Set the correct group profile data
1020                 setGroupData();
1021                 break;
1022             case "content":
1023                 setContentData(data);
1024                 break;
1025         }
1026 
1027         if(entityconfig.mode ==="content" && !data){
1028             return;
1029         }
1030 
1031         // Render the main template
1032         renderTemplate();
1033 
1034         // Should we show the Add To Contacts button or not
1035         if (mode === "profile" && !sakai.data.me.user.anon){
1036             checkContact(entityconfig.data.profile["rep:userId"]);
1037         }
1038 
1039         // Add binding
1040         addBinding();
1041 
1042     };
