377     /**
378      * Set the description of the uploaded files
379      */
380     var batchSetDescriptionAndName = function(data){
381         // Batch link the files with the tags
382         var batchDescriptionData = [];
383         // Check if it's a link that's been uploaded
384         if (uploadedLink) {
385             if (newVersionIsLink) {
386                 var item = {
387                     "url": "/p/" + oldVersionPath,
388                     "method": "POST",
389                     "parameters": {
390                         "sakai:pooled-content-revurl": $fileUploadLinkBoxInput.val()
391                     }
392                 };
393                 batchDescriptionData[batchDescriptionData.length] = item;
394             }
395             else {
396                 var pathHash, url;
397                 for (var k in data) {
398                     if (data.hasOwnProperty(k)) {
399                         pathHash = data[k];
400                         url = k.substring(0,k.length-4); // remove .lnk from the end of it
401                         break;
402                     }
403                 }
404                 var item2 = {
405                     "url": "/p/" + pathHash,
406                     "method": "POST",
407                     "parameters": {
408                         "sakai:pooled-content-url": url,
409                         "sakai:pooled-content-revurl": url,
410                         "sakai:pooled-content-file-name": url,
411                         "sakai:directory": "default",
412                         "sakai:copyright": "creativecommons"
413                     }
414                 };
415                 batchDescriptionData[batchDescriptionData.length] = item2;
416             }
417         } else {
418             for (var i in uploadedFiles) {
419                 if (uploadedFiles.hasOwnProperty(i)) {
420                     var item3 = {
421                         "url": "/p/" + uploadedFiles[i].hashpath,
422                         "method": "POST",
423                         "parameters": {
424                             "sakai:description": $fileUploadAddDescription.val(),
425                             "sakai:fileextension": uploadedFiles[i].filename.substring(uploadedFiles[i].filename.lastIndexOf("."), uploadedFiles[i].filename.length),
426                             "sakai:pooled-content-file-name": uploadedFiles[i].name,
427                             "sakai:directory": "default",
428                             "sakai:permissions": $fileUploadPermissionsSelect.val(),
429                             "sakai:copyright": "creativecommons"
430                         }
431                     };
432                     batchDescriptionData[batchDescriptionData.length] = item3;
433                 }
434             }
435         }
436         dataResponse = batchDescriptionData;
437         // Do the Batch request
438         $.ajax({
439             url: sakai.config.URL.BATCH,
440             traditional: true,
441             type : "POST",
442             cache: false,
443             data: {
444                 requests: $.toJSON(batchDescriptionData)
445             },
446             success: function(data){
447                 resetFields();
448             }
449         });
450     };

453      * After uploading the link and if the context is 'group' the resource should be set as a group resource
454      * @param {Object} data response from creating the link on the server

477         data[data.length] = properties;

480             url: sakai.config.URL.BATCH,

482             type: "POST",
483             cache: false,
484             data: {
485                 requests: $.toJSON(data)

496     /**
497      * Upload the validated link to a file
498      * Execute checks to see if this link is a revision or not
499      */
500     var uploadLink = function(){
501         var body = "";
502         body = "--AAAAA\r\n";
503         body = body + "Content-Disposition: form-data; name=\"*\"; filename=\"" + $fileUploadLinkBoxInput.val() + ".lnk\"\r\n";
504         body = body + "Content-Type: x-sakai/link\r\n";
505         body = body + "sakai:pooled-content-url: " + $fileUploadLinkBoxInput.val() + "\r\n";
506         body = body + "Content-transfer-encoding: binary\r\n\r\n";
507         body = body + $fileUploadLinkBoxInput.val() + "\r\n";
508         body = body + "--AAAAA--\r\n";
509 
510         var path = "";
511         if (newVersionIsLink) {
512             path = oldVersionPath;
513         }
514         else {
515             path = uploadPath;
516         }
517         var url = path;
518         if (context === "new_version") {
519             url = "/p/" + path;
520         }
521         $.ajax({
522             url: url,
523             data: body,
524             type: "POST",
525             dataType: "json",
526             contentType: "multipart/form-data; boundary=AAAAA",
527             success: function(data){
528                 //loop over node to extract data
529                 var linkArray = [];
530                 for (var i in data) {
531                     if (data.hasOwnProperty(i)) {
532                         var obj = {};
533                         obj.filename = i;
534                         obj.hashpath = data[i];
535                         linkArray.push(obj);
536                     }
537                 }
538                 dataResponse = data;
539 
540                 uploadedLink = true;
541                 filesUploaded = false;
542 
543                 batchSetDescriptionAndName(data);
544 
545                 newVersionIsLink = false;
546 
547                 if (context === "group") {
548                     setLinkAsGroupResource(data);
549                 } else {
550                     // Set permissions on the files
551                     sakai.api.Content.setFilePermissions("public", linkArray, function(permissionsSet){
552                     });
553                 }
554 
555                 $("#fileupload_link_submit").removeAttr("disabled");
556                 $fileUploadAddLinkButton.removeAttr("disabled");
557                 $fileUploadLinkBoxInput.removeAttr("disabled");
558             },
559             error: function(err){
560                 $fileUploadAddLinkButton.removeAttr("disabled");
561                 $fileUploadLinkBoxInput.removeAttr("disabled");
562                 sakai.api.Util.notification.show($(fileUploadCheckURL).html(), $(fileUploadEnterValidURL).html());
563             }
564         });
565     };

646      *

648     var initialiseUploader = function(){
649         $multiFileUpload.MultiFile({
650             list: $multiFileList
651         });
652 
653         // Set the form action attribute
654         $newUploaderForm.attr("action", uploadPath);
655 
656         $multiFileForm.ajaxForm({
657             success: function(data){
658                 // reset some variables
659                 uploadedFiles = [];
660                 tags = [];
661 
662                 // Create DOM element to extract data from response
663                 // Use an object to keep track of the data
664                 var responseData = {};
665                 if (jQuery.browser.webkit) {
666                     responseData = $.parseJSON(data.split(">")[1].split("<")[0]);
667                 }
668                 else {
669                     responseData = $.parseJSON(data.replace(/<pre>/i, "").replace(/<\/pre>/i, ""));
670                 }
671                 var extractedData = [];
672 
673                 //loop over nodes to extract data
674                 for (var i in responseData) {
675                     if (responseData.hasOwnProperty(i)) {
676                         var obj = {};
677                         obj.filename = i;
678                         obj.hashpath = responseData[i];
679                         extractedData.push(obj);
680                     }
681                 }
682                 dataResponse = extractedData;
683 
684                 // Check if there were any files uploaded
685                 if (extractedData.length === 0) {
686                     noFilesUploaded();
687                 }
688                 else {
689                     // Files uploaded
690                     filesUploaded = true;
691 
692                     // Get the values out of the name boxes
693                     $multiFileList.find("input").each(function(index){
694                         for (var i in extractedData){
695                             if (extractedData.hasOwnProperty(i)) {
696                                 // Fix IE7 issue where full path is used as a name for an upload
697                                 // Cut off the extra path information with the following replace functions
698                                 var replacedData = extractedData[i].filename.replace(/\./g, "_");
699                                 if ($(this)[0].id === replacedData.substring(replacedData.lastIndexOf("\\") + 1, replacedData.length)) {
700                                     extractedData[i].name = $(this).val();
701                                     break;
702                                 }
703                             }
704                         }
705                     });
706 
707                     uploadedFiles = extractedData;
708 
709                     // Initiate the tagging process
710                     tags = sakai.api.Util.formatTags($fileUploadAddTags.val());
711                     for (var file in uploadedFiles) {
712                         if (uploadedFiles.hasOwnProperty(file)) {
713                             sakai.api.Util.tagEntity("/p/" + uploadedFiles[file].hashpath, tags, []);
714                         }
715                     }
716 
717                     // If the file is a new version set is as one
718                     // Else it is a new file and needs to have a description, permissions, tags, ...
719                     if (context !== "new_version") {
720                         // Set the description data on the completed uploads
721 
722                         // Set permissions on the files
723                         sakai.api.Content.setFilePermissions($fileUploadPermissionsSelect.val(), uploadedFiles, function(permissionsSet){
724                             batchSetDescriptionAndName();
725                         }, contextData.id);
726 
727                     } else {
728                         // Set the new file extension
729                         setFileExtension(uploadedFiles[0].filename.substring(uploadedFiles[0].filename.lastIndexOf("."), uploadedFiles[0].filename.length));
730                     }
731                 }
732             },
733             error : function(){
734                 noFilesUploaded();
735             }
736         });
737     };
