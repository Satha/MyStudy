 19 /*global $ */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.footer
 25  *
 26  * @class footer
 27  *
 28  * @description
 29  * Initialize the footer widget
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.footer = function(tuid,showSettings){
 36 
 37 
 38     /////////////////////////////
 39     // Configuration variables //
 40     /////////////////////////////
 41 
 42     var doc_name;
 43     var $back_to_top_link = $(".footer_main .back-top");
 44     var $footer_debug_info = $("#footer_debug_info");
 45     var $footer_date_end = $("#footer_date_end");
 46     var $footer_root = $(".footer_main");
 47     var $footer_logo = $("#footer_logo");
 48 
 49 
 50     //////////////////////
 51     // Helper functions //
 52     //////////////////////
 53 
 54     /**
 55      * This helper function will return the name of the current document (e.g. my_sakai.html)
 56      * @return {String} The name of the current document
 57      */
 58     var getDocName = function() {
 59         var url = document.URL;
 60         var slash = "/";
 61         if (url.match(/\\/)) {
 62             slash = "\\";
 63         }
 64         return url.substring(url.lastIndexOf(slash) + 1);
 65     };
 66 
 67     /**
 68      * Check whether this is the index page or not
 69      * @return {Boolean} True when it is the index page
 70      */
 71     var checkIndexPage = function(){
 72         return document.URL.match(/index.html[?a-zA-Z0-9=]*/);
 73     };
 74 
 75 
 76     ////////////////////
 77     // Main functions //
 78     ////////////////////
 79 
 80     /**
 81      * Render the debug info
 82      * @param {Object} container jQuery selector where you want the debug info to appear in
 83      */
 84     var renderDebugInfo = function(container) {
 85         
 86         $.ajax({
 87             url: "/var/scm-version.json",
 88             type: "GET",
 89             cache: false,
 90             dataType: "json",
 91             success: function(data){
 92                 // Construct debug info
 93                 var debug_text = "DEBUG:";
 94                 debug_text += " Nakamura Version: " + data["sakai:nakamura-version"];
 95                 getUxVersion(debug_text, container);
 96             }
 97         });
 98     };
 99 
100     var getUxVersion = function(debug_text, container) {
101         $.ajax({
102             url: "/var/ux-version/ux-version.json",
103             type: "GET",
104             cache: false,
105             dataType: "json",
106             success: function(data){
107                 debug_text += " | UX Version: " + data["sakai:ux-version"];
108                 debug_text += "<br/>DOC mod date: " + document.lastModified;
109                 debug_text += " | PLACE: " + (doc_name || "index.html");
110 
111                 // Put text into holding tag
112                 container.html(sakai.api.Security.saneHTML(debug_text));
113             }
114         });
115     };
116 
117     /**
118      * This event handler will make sure that the Top link
119      * that's available in every page footer will scroll back
120      * to the top of the page
121      */
122     $(".back-top").live("click", function(ev){
123         window.scrollTo(0,0);
124     });
125 
126 
127     /////////////////////////////
128     // Initialisation function //
129     /////////////////////////////
130 
131     /**
132      * Main initialization function for the footer widget
133      */
134     var doInit = function(){
135 
136         // Get the name of the current document
137         doc_name = getDocName();
138 
139         // Display debug info if set in config
140         if (sakai.config.displayDebugInfo === true) {
141 
142             // Add binding to the image
143             $footer_logo.toggle(function(){
144 
145                 // Render the debug info
146                 renderDebugInfo($footer_debug_info);
147 
148                 // Show the debug info
149                 $footer_debug_info.show();
150 
151             },function(){
152 
153                 // Hide the debug info
154                 $footer_debug_info.hide();
155 
156             }).addClass("footer_clickable");
157 
158         }
159 
160         // index.html mods
161         if (checkIndexPage() || doc_name === "") {
162             $back_to_top_link.hide();
163             $footer_root.addClass("footer_index");
164         }
165 
166         // Set the end year of the copyright notice
167         var d = new Date();
168         $footer_date_end.text(d.getFullYear());
169 
170     };
171 
172     doInit();
173 
174 };
175 
176 sakai.api.Widgets.widgetLoader.informOnLoad("footer");
