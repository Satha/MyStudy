 19 /*global $, Config, addBinding */
 20 
 21 var sakai = sakai || {};
 22 
 23 /**
 24  * @name sakai.googlemaps
 25  *
 26  * @class googlemaps
 27  *
 28  * @description
 29  * Initialize the googlemaps widget
 30  *
 31  * @version 0.0.1
 32  * @param {String} tuid Unique id of the widget
 33  * @param {Boolean} showSettings Show the settings of the widget or not
 34  */
 35 sakai.googlemaps = function(tuid, showSettings){
 36 
 37 
 38     /////////////////////////////
 39     // Configuration variables //
 40     /////////////////////////////
 41 
 42     var rootel = $("#" + tuid);
 43     var iframeContentWindow = {};
 44     var json = false;
 45 
 46     /**
 47      * To finish the widget
 48      */
 49     var finish = function() {
 50         sakai.api.Widgets.Container.informFinish(tuid, "googlemaps");
 51     };
 52 
 53     /**
 54      * This will saved the map data to backend server
 55      */
 56     var saveToJCR = function() {
 57         json = $("#googlemaps_iframe_map", rootel)[0].contentWindow.getJSON();
 58 
 59         // Set the value of mapsize according to the radio button selection status
 60         if ($("#googlemaps_radio_large", rootel).is(":checked")) {
 61             json.mapsize = "LARGE";
 62         }
 63         else {
 64             json.mapsize = "SMALL";
 65         }
 66 
 67         // Store the corresponded map data into backend server
 68         sakai.api.Widgets.saveWidgetData(tuid, json, finish);
 69     };
 70 
 71     /**
 72      * To set the map initial size in the show panel
 73      */
 74     var setMapSize = function(callback) {
 75         if(!showSettings) {
 76             $("#googlemaps_size_options").hide();
 77             if (json && json.mapsize == "SMALL") {
 78 
 79                 // Set the size of map according to the data stored on the backend server
 80                 $("#googlemaps_iframe_map", rootel).width("50%");
 81                 //$("#googlemaps_iframe_map", rootel).css({"float": "right"});
 82             }
 83             else {
 84                 $("#googlemaps_iframe_map", rootel).width("95%");
 85             }
 86         }
 87         else {
 88         
 89             $("#googlemaps_size_options").show();
 90     
 91             // Show the search input textfield and save, search, cancel buttons
 92             $("#googlemaps_form_search", rootel).show();
 93             $("#googlemaps_save_cancel_container", rootel).show();
 94 
 95             // Add a submit listener so that the search function can be executed
 96             $("#googlemaps_form_search", rootel).submit(function() {
 97                 var input = $("#googlemaps_input_text_location", rootel).val();
 98                 if (input) {
 99 
100                     // Quick hack so that searches are more local - this will need to be done via the Google API
101                     if (input.indexOf(",") === -1) {
102                         iframeContentWindow.search(input, sakai.data.me.user.locale.displayCountry);
103                     } else {
104                         iframeContentWindow.search(input, "");
105                     }
106                 }
107                 return false;
108             });
109 
110             // Add listener to save button
111             $("#googlemaps_save", rootel).bind("click", function(e, ui) {
112                 saveToJCR();
113             });
114 
115             // Add listerner to cancel button
116             $("#googlemaps_cancel", rootel).bind("click", function(e, ui) {
117                 sakai.api.Widgets.Container.informCancel(tuid, "googlemaps");
118             });
119 
120             if (json) {
121                 if (json.mapsize == "SMALL") {
122 
123                     // If the reserved mapsize is "SMALL", the "small" radio button should be checked
124                     $("#googlemaps_radio_small", rootel).attr("checked", "checked");
125                 }
126                 else if (json.mapsize == "LARGE") {
127 
128                     // If the reserved mapsize is "LARGE", the "large" radio button should be checked
129                     $("#googlemaps_radio_large", rootel).attr("checked", "checked");
130                 }
131             }
132         }
133 
134         callback();  // Set the map center, zoom and infowindow in the map iframe
135     };
136 
137     /**
138      * To set the map's center, zoom, size, info window content properties
139      */
140     var setMap = function() {
141         if (json) {
142             iframeContentWindow.setMap(json);
143         }
144     };
145 
146     /**
147      * This is to get map zoom and center properties from backend server
148      */
149     var getFromJCR = function() {
150 
151         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
152 
153             if (success) {
154 
155                 // Get data from the backend server
156                 json = data;
157 
158                 // Set the size of the map's iframe
159                 setMapSize(setMap);
160 
161                 // Set the initial value of search keyword input textbox
162                 $("#googlemaps_input_text_location", rootel).val(json.mapinput);
163                 
164             } else {
165                 if (showSettings) {
166                     // Show the search input textfield and save, search, cancel buttons
167                     $("#googlemaps_form_search", rootel).show();
168                     $("#googlemaps_save_cancel_container", rootel).show();
169 
170 
171                     // If the googlemaps is opened for the first time, the "large" radio button should be default checked
172                     $("#googlemaps_radio_large", rootel).attr("checked", "checked");
173 
174 
175                     // Add a submit listener so that the search function can be executed
176                     $("#googlemaps_form_search", rootel).submit(function() {
177                         var input = $("#googlemaps_input_text_location", rootel).val();
178                         if (input) {
179 
180                             // Quick hack so that searches are more local - this will need to be done via the Google API
181                             if (input.indexOf(",") === -1) {
182                                 $("#googlemaps_input_text_location", rootel).val(input + ", " + sakai.data.me.user.locale.displayCountry);
183                                 iframeContentWindow.search(input, sakai.data.me.user.locale.displayCountry);
184                             } else {
185                                 iframeContentWindow.search(input, "");
186                             }
187                         }
188                         return false;
189                     });
190 
191                     // Add listener to save button
192                     $("#googlemaps_save", rootel).bind("click", function(e, ui) {
193                         saveToJCR();
194                     });
195 
196                     // Add listerner to cancel button
197                     $("#googlemaps_cancel", rootel).bind("click", function(e, ui) {
198                         sakai.api.Widgets.Container.informCancel(tuid, "googlemaps");
199                     });
200                 }
201 
202             }
203             // show main div
204             $("#main-content-div").show();
205             // Set focus in location text box
206             $("#googlemaps_input_text_location").focus();
207         });
208     };
209 
210     /**
211      * This is to init the google map and other controls
212      */
213     var init = function() {
214         // hide main div to prevent showing default cambridge location while loading
215         $("#main-content-div").hide();
216         $("#googlemaps_iframe_map", rootel).load(function() {
217 
218             // To set the map iframe's content as the iframeContentWindow (var)
219             iframeContentWindow = $("#googlemaps_iframe_map", rootel)[0].contentWindow;
220 
221             // Get the map zoom and center property data from backend server
222             getFromJCR();
223         });
224     };
225 
226     init();
227 
228 };
229 
230 sakai.api.Widgets.widgetLoader.informOnLoad("googlemaps");
