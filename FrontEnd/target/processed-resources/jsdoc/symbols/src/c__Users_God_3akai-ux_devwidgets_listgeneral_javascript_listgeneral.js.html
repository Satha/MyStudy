 32 /**
 33  * @name sakai.listgeneral
 34  *
 35  * @class listgeneral
 36  *
 37  * @description
 38  * General Lister widget<br />
 39  * This is a general widget which aims to display an arbitriary number of
 40  * spaces, sites or content, loading dynamically if the list is very long.
 41  *
 42  * @version 0.0.1
 43  * @param {String} tuid Unique id of the widget
 44  * @param {Boolean} showSettings Show the settings of the widget or not
 45  */
 46 sakai.listgeneral = function(tuid, showSettings){
 47 
 48     // Config defaults
 49     var default_config = {
 50         "items": 25,
 51         "selectable": false
 52     };
 53 
 54     // Create a config object for this instance
 55     sakai.config.widgets.listgeneral = sakai.config.widgets.listgeneral || {};
 56     sakai.config.widgets.listgeneral[tuid] = default_config;
 57 
 58     // Create a data object for this instance
 59     sakai.data.listgeneral = sakai.data.listgeneral || {};
 60     sakai.data.listgeneral[tuid] = {};
 61     sakai.data.listgeneral[tuid].selected = 0;
 62 
 63     // Reset to defaults
 64     sakai.api.UI.listGeneral.reset(tuid);
 65 
 66     // Send out an event that says the widget is ready to
 67     // accept a search query to process and display. This event can be picked up
 68     // in a page JS code
 69     $(window).trigger("listgeneral_ready", [tuid]);
 70 };

 73 /**
 74  * Reset
 75  * Resets the general lister to a default state
 76  * @param tuid {String} Unique id of the widget
 77  * @returns void
 78  */
 79 sakai.api.UI.listGeneral.reset = function(tuid) {
 80 
 81     //Clear results from DOM
 82     $("#"+tuid+" .listgeneral_search_results").html("");
 83 
 84     // Reset selected items
 85     sakai.data.listgeneral[tuid].selected = {};
 86 
 87 };

 89 /**
 90  * Render
 91  * Renders the general lister with a specified set of data. The function uses
 92  * a search query initially, then does the paginating and subsequent requests
 93  * for data automatically
 94  * @param tuid {String} Unique id of the widget
 95  * @param iSearchQuery {Array} A Sakai search queries
 96  * @param iConfig {Object} Optional config overrides
 97  * @returns void
 98  */
 99 sakai.api.UI.listGeneral.render = function(tuid, iSearchQuery, iConfig) {
100 
101     // Merge user defined config with defaults
102     for (var element in iConfig) {
103         if (iConfig.hasOwnProperty(element)) {
104             sakai.config.widgets.listgeneral[tuid][element] = iConfig[element];
105         }
106     }
107 
108     // Parse search query
109     var searchQuery = [];
110     for (var i = 0, il = iSearchQuery.length; i < il; i++) {
111 
112         var searchQueryObject = {};
113         var main_parts = iSearchQuery[i].split("?");
114         searchQueryObject.url = main_parts[0];
115         var args = main_parts[1].split("&");
116         for (var j = 0, jl = args.length; j < jl; j++) {
117             var kv_pair = args[j].split("=");
118             searchQueryObject[kv_pair[0]] = kv_pair[1];
119         }
120 
121         // Alter search query according to config
122         searchQueryObject.items = sakai.config.widgets.listgeneral[tuid].items[i];
123 
124         // Add hash to search query in case it's not there to prevent caching
125         if (!searchQueryObject["_"]) {
126             searchQueryObject["_"] = (Math.random() * 1000000000000000000);
127         }
128 
129         // Push search query into the array of queries
130         searchQuery.push(searchQueryObject);
131 
132     }
133     // Set up sorting drop-down according to the type of list
134     $(".listgeneral_sortview").prepend($(".listgeneral_sorttemplate_" + sakai.config.widgets.listgeneral[tuid].type).html());
135 
136 
137     // Sorting
138     if (!sakai.config.widgets.listgeneral[tuid].sortOn) {
139         sakai.config.widgets.listgeneral[tuid].sortOn = $("#" + tuid + " .listgeneral_sorttemplate_" + sakai.config.widgets.listgeneral[tuid].type+" option[default='true']").attr("value");
140     }
141     if (!sakai.config.widgets.listgeneral[tuid].sortOrder) {
142         sakai.config.widgets.listgeneral[tuid].sortOrder = $("#" + tuid + " .listgeneral_sorttemplate_" + sakai.config.widgets.listgeneral[tuid].type+" option[default='true']").attr("data-sortorder");
143     }
144 
145     // View mode
146     sakai.config.widgets.listgeneral[tuid].viewMode = $("#" + tuid + " .listgeneral_sortview_view option[default='true']").attr("value");
147 
148 
149     // Wire up sort button
150     $("#" + tuid + " .listgeneral_sort_button").bind("click", function(e) {
151 
152         // Get sorting flag from the dropdown
153         sakai.config.widgets.listgeneral[tuid].sortOn = $("#" + tuid + " .listgeneral_sortview_sort option:selected").attr("value");
154         sakai.config.widgets.listgeneral[tuid].sortOrder = $("#" + tuid + " .listgeneral_sortview_sort option:selected").attr("data-sortorder");
155 
156         // Get view mode from the dropdown
157         sakai.config.widgets.listgeneral[tuid].viewMode =  $("#" + tuid + " .listgeneral_sortview_view option:selected").attr("value");
158 
159         // Reset everything
160         sakai.api.UI.listGeneral.reset(tuid);
161 
162         // Get new results with new sorting
163         sakai.api.UI.listGeneral.addPage(tuid, 0, searchQuery);
164     });
165 
166 
167     // Deal with non logged in users - remove favourite select options
168 
169 
170     // Render the first page of results
171     sakai.api.UI.listGeneral.addPage(tuid, 0, searchQuery);
172 
173 };

175 /**
176  * addPage
177  * Adds another page of search result to the People lister's result list
178  * @param tuid {String} The instance ID of a widget
179  * @pageNumber {Int} The page we want to load
180  * @searchQuery {Array} An array containing objects of the search query elements
181  * @returns void
182  */
183 sakai.api.UI.listGeneral.addPage = function(tuid, pageNumber, searchQuery) {
184 
185     // Store number of requests we need to process to make sure we process all at the end of the ajax requests
186     sakai.data.listgeneral[tuid].numOfQueriesToProcess = searchQuery.length;
187 
188     // Create storage for aggregate results
189     sakai.data.listgeneral[tuid].aggregateResults = {};
190     sakai.data.listgeneral[tuid].aggregateResults.results = [];
191 
192 
193     var handleSuccess = function(rawData) {
194         // Keep track of queries which needs to be processed
195         sakai.data.listgeneral[tuid].numOfQueriesToProcess--;
196 
197         for (var j = 0, jl = rawData.results.length; j < jl; j++) {
198 
199             // Make code more legible, namespace additional info
200             var result = rawData.results[j];
201             result.listgeneral = {};
202 
203             // Decorating / sanitising the result objects with convenience info for rendering
204 
205             // Sites, pages, spaces
206             if (result.type === "sakai/page" ||
207                 result.type === "sakai/pagecontent" ||
208                 result["sling:resourceType"] === "sakai/site" ||
209                 (result.site && result.site["sling:resourceType"] === "sakai/site") ||
210                 result["sling:resourceType"] === "sakai/space" ||
211                 (result.space && result.space["sling:resourceType"] === "sakai/space")
212                 ) {
213 
214                 // Add type property
215                 if (result.type === "sakai/space") {
216                     result.listgeneral.resultType = "spaces";
217                 } else {
218                     result.listgeneral.resultType = "sites";
219                 }
220 
221                 // Try to get name
222                 if (result.site && result.site.name) {
223                     result.listgeneral.name = result.site.name;
224                 } else if (result.name) {
225                     result.listgeneral.name = result.name;
226                 } else {
227                     result.listgeneral.name = result["jcr:name"];
228                 }
229 
230                 // Get description
231                 if (result.description) {
232                     result.listgeneral.description = result.description;
233                 } else if (result.site && result.site.description) {
234                     result.listgeneral.description = result.site.description;
235                 } else if (result.excerpt) {
236                     result.listgeneral.description = $("" + result.excerpt + "").text().replace(/<[^>]*>/g, "");
237                 } else {
238                     result.listgeneral.description = "";
239                 }
240 
241 
242                 // Create page path if the hit is a page content
243                 if (result.data && result.data["sling:resourceType"] === "sakai/pagecontent") {
244                     var pagePath = "";
245                     pagePath = result.data["sling:resourceType"].replace(/\/_pages/g, "");
246                     pagePath = pagePath.replace(/\/pageContent/g, "");
247                     pagePath = pagePath.replace(/\//g,"");
248                     pagePath = result.site["jcr:path"] + "#" + pagePath;
249                     result.listgeneral.pagePath = pagePath;
250                 } else {
251                     // Or just use site path if it's site root
252                     result.listgeneral.pagePath = result["jcr:path"];
253                 }
254 
255                 // Eval picture object if exists either in site or in main result. If not use a default one.
256                 if (result.picture) {
257                     result.picture = $.parseJSON(result.picture);
258                     result.listgeneral.avatar = result["jcr:path"] + "/" + result.picture.name;
259                 } else if (result.site && result.site.picture) {
260                     result.site.picture = $.parseJSON(result.site.picture);
261                     result.listgeneral.avatar = result.site["jcr:path"] + "/" + result.site.picture.name;
262                 } else if (result.space && result.space.picture) {
263                     result.space.picture = $.parseJSON(result.space.picture);
264                     result.listgeneral.avatar = result.space["jcr:path"] + "/" + result.space.picture.name;
265                 } else {
266                     result.listgeneral.avatar = "/dev/images/mimetypes/html.png";
267                 }
268 
269             }
270 
271             // Content
272             if (result["jcr:mimeType"] && result["jcr:primaryType"] === "nt:file") {
273 
274                 // Add type property
275                 result.listgeneral.resultType = "content";
276 
277                 // Get mimetype icon and human readable mimetype description
278                 if (sakai.config.MimeTypes[result["jcr:mimeType"]]) {
279                     result.listgeneral.avatar = sakai.config.MimeTypes[result["jcr:mimeType"]].URL;
280                     result.listgeneral.mimeTypeDescripton = sakai.config.MimeTypes[result["jcr:mimeType"]].description;
281                 } else {
282                     result.listgeneral.avatar = "/dev/images/mimetypes/empty.png";
283                     result.listgeneral.mimeTypeDescripton = sakai.config.MimeTypes.other.description;
284                 }
285 
286                 // Description
287                 if (result.description) {
288                     result.listgeneral.description = result.description;
289                 } else {
290                     result.listgeneral.description = "";
291                 }
292 
293             }
294 
295 
296             // People
297             if (result["sling:resourceType"] === "sakai/user-profile") {
298 
299                 // Add type property
300                 result.listgeneral.resultType = "people";
301 
302                 // Avatar
303                 if (result.picture) {
304                     result.picture = $.parseJSON(result.picture);
305                     result.listgeneral.avatar = "/~" + result["rep:userId"] + "/public/profile/" + result.picture.name;
306                 } else {
307                     result.listgeneral.avatar = sakai.config.URL.USER_DEFAULT_ICON_URL;
308                 }
309 
310                 // University role
311                 if (result.basic) {
312                     result.basic = $.parseJSON(result.basic);
313                     result.listgeneral.role = result.basic.unirole;
314                 } else {
315                     result.listgeneral.role = "";
316                 }
317 
318                 // Description
319                 if (result.aboutme) {
320                     result.aboutme = $.parseJSON(result.aboutme);
321                     result.listgeneral.description = result.aboutme.aboutme;
322                 } else {
323                     result.listgeneral.description = "";
324                 }
325 
326 
327             }
328 
329 
330             // Add result to the collection of aggregate results
331             sakai.data.listgeneral[tuid].aggregateResults.results.push(rawData.results[j]);
332         }
333 
334         // If this is the last search query to process
335         if (sakai.data.listgeneral[tuid].numOfQueriesToProcess === 0) {
336             sakai.api.UI.listGeneral.sortAndDisplay(tuid, pageNumber);
337         }
338 
339 
340     };
341 
342     var handleError = function(xhr, status, thrown) {
343         // Keep track of queries which needs to be processed
344         sakai.data.listgeneral[tuid].numOfQueriesToProcess--;
345 
346         if (sakai.data.listgeneral[tuid].numOfQueriesToProcess === 0) {
347             sakai.api.UI.listGeneral.sortAndDisplay(tuid, pageNumber);
348         }
349 
350         // Log error in console
351         debug.error("listgeneral.js: Error in processing a search request!");
352     };
353 
354     var handleMoreButtonClick = function(e) {
355         $(this).remove();
356         sakai.api.UI.listGeneral.addPage(tuid, pageNumber + 1, searchQuery);
357     };
358     // Create new container for the bit we load. This is then appended to the
359     // main container
360     for (var i=0, il = searchQuery.length; i < il; i++) {
361 
362         // Add relevant config elements to the search query
363         searchQuery[i].page = pageNumber;
364         searchQuery[i].sortOn = sakai.config.widgets.listgeneral[tuid].sortOn;
365         searchQuery[i].sortOrder = sakai.config.widgets.listgeneral[tuid].sortOrder;
366 
367         // Construct search query
368         var sq = searchQuery[i].url + "?";
369         for (var e in searchQuery[i]) {
370             if (searchQuery[i].hasOwnProperty(e) && e !== "url") {
371                 sq += e + "=" + searchQuery[i][e] + "&";
372             }
373         }
374         sq = sq.substring(0, sq.length-1);
375 
376 
377         // Fire search requests
378         $.ajax({
379             url: sq,
380             type: "GET",
381             success: handleSuccess,
382             error: handleError
383         });
384 
385         // Display empty new container with loading anim
386         $("#" + tuid + " .listgeneral_search_results").append($("<div class=\"listgeneral_resultpage loadinganim\" data-pagenumber=\"" + pageNumber + "\"></div>"+$("#" + tuid + " .listgeneral_more_results_template").html()));
387 
388         // Wire up more results button
389         $("#" + tuid + " .listgeneral_more_button").bind("click", handleMoreButtonClick);
390 
391     }
392 
393 };

396 /**
397  * sortAndDsiplay
398  * Displays the search results by rendering the appropriate template
399  * @param tuid {String} The instance ID of the widget
400  * @param pageNumber {Int} The page number we wuld like to render
401  */
402 sakai.api.UI.listGeneral.sortAndDisplay = function(tuid, pageNumber) {
403 
404     var $resultPage = $("#" + tuid + " .listgeneral_resultpage[data-pagenumber='"+pageNumber+"']");
405 
406     // No sorting of aggregate results for now...
407 
408     // If there are no search results, display message template
409     if (sakai.data.listgeneral[tuid].aggregateResults.results.length === 0) {
410 
411         $("#" + tuid + " .listgeneral_more_button").hide();
412         $resultPage.removeClass("loadinganim").html($("#"+tuid+" .listgeneral_noresults_template").html());
413 
414     } else {
415 
416         // Render the results data template
417         var newPageHTML = $.TemplateRenderer("#" + tuid + " .listgeneral_content_template", sakai.data.listgeneral[tuid].aggregateResults);
418 
419         // Add rendered result HTML to DOM and set viewmode class
420         $resultPage.removeClass("loadinganim").addClass("listgeneral_result_viewmode_" + sakai.config.widgets.listgeneral[tuid].viewMode).append(newPageHTML);
421     }
422 };
