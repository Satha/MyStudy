 25  * @name sakai.listpeople

 44 sakai.listpeople = function(tuid, showSettings) {
 45 
 46     var totalResult = 5; // total no. of items display in list
 47     // DOM Selectors
 48     var $rootel = $("#"+tuid);
 49     var $listpeople_container = $(".listpeople_content", $rootel);
 50     var $listpeople_content = $(".listpeople_content", $rootel);
 51     var $listpeople_count = $(".listpeople_count", $rootel);
 52     var $listpeople_count_items = $(".listpeople_count_items", $rootel);
 53     var $listpeople_count_of = $(".listpeople_count_of", $rootel);

 56     var $listpeople_sort_order = $("#listpeople_sort_order", $rootel);

 58     // Template Selectors
 59     var $listpeople_content_pagetemplate = $(".listpeople_content_pagetemplate", $rootel);

 62     var listType = "";

 64     /**
 65      * Reset
 66      * Resets the people lister to a default state
 67      * @returns void
 68      */
 69     var reset = function() {
 70         $listpeople_content.html("");
 71         $listpeople_count.html("0");
 72         $listpeople_count_items.show();
 73         $listpeople_count_of.hide();
 74         $listpeople_count_total.hide();
 75         $listpeople_count_selected.hide();
 76         $listpeople_content.unbind("scroll");
 77         $listpeople_sort_order.unbind("change");
 78 
 79         sakai.data.listpeople[listType].selected = {};
 80         sakai.data.listpeople[listType].currentElementCount = 0;
 81         sakai.data.listpeople[listType].selectCount = 0;
 82     };

 84     /**
 85      * Render
 86      * Renders the people lister with a specified set of data. The function uses
 87      * a search query initially, then does the paginating and subsequent requests
 88      * for data automatically
 89      * @param iConfig {Object} Optional sakai.config.listpeople[listType] overrides
 90      * @param url {String} URL to request data from
 91      * @param id {String} Unique id related to the data we're listing
 92      * @returns void
 93      */
 94     var render = function(iTuid, iConfig, url, id) {
 95         if (iTuid !== tuid) { return; }
 96         reset();
 97 
 98         // Merge user defined sakai.config.listpeople[tuid] with defaults
 99         for (var element in iConfig) {
100             if (iConfig.hasOwnProperty(element)) {
101                 sakai.config.listpeople[listType][element] = iConfig[element];
102             }
103         }
104 
105         if (sakai.config.listpeople[listType].items) {
106             if (url.indexOf("?", 0) > 0) {
107                 url = url + "&items=" + sakai.config.listpeople[listType].items;
108             } else {
109                 url = url + "?items=" + sakai.config.listpeople[listType].items;
110             }
111         }
112 
113         // get data
114         $.ajax({
115             url: url,
116             cache: false,
117             success: function(data){
118                 var json_data = {},
119                     itemCount = 0;
120                 if (typeof(data) === 'string') {
121                     data = $.parseJSON(data);
122 
123                     // Variable to check if the logged in user is a manager of the file or not
124                     // If not we handel him like an anonymous user (not giving any editting possibilities in the UX)
125                     if (typeof sakai.data.me.user.userid !== "undefined") {
126                         if (data.managers.length !== 0) {
127                             for (var i in data.managers) {
128                                 if (data.managers[i]["rep:userId"] === sakai.data.me.user.userid) {
129                                     sakai.config.listpeople[listType].anon = false;
130                                     break;
131                                 }
132                                 else {
133                                     sakai.config.listpeople[listType].anon = true;
134                                 }
135                             }
136                         }
137                         else {
138                             sakai.config.listpeople[listType].anon = true;
139                         }
140                     } else {
141                         sakai.config.listpeople[listType].anon = true;
142 
143                     }
144 
145                     // Check to set the buttons visible or invisible according to logged in user
146                     if (!sakai.config.listpeople[listType].anon) {
147                         $(".content_profile_list_buttons").show();
148                     }
149 
150                     json_data = {
151                         "results" : data,
152                         "total" : itemCount
153                     };
154                 } else if (!data.results) {
155                     json_data = {
156                         "results" : data,
157                         "total" : itemCount
158                     };
159                 } else {
160                     json_data = data;
161                 }
162 
163                 // if we're loading authorizables for a content profile we need to check down a level for either viewers or managers
164 
165                 if (json_data.results[listType]) {
166                     json_data.results = json_data.results[listType];
167                 }
168 
169                 if (json_data) {
170                     itemCount = 0;
171 
172                     // filter out the manager group if we're listing items for a group, the everyone group and the anonymous user
173                     json_data.results = $.grep(json_data.results, function(resultObject, index){
174                             if (resultObject['sakai:group-id'] !== id + '-managers' &&
175                                 resultObject['sakai:group-id'] !== 'everyone' &&
176                                 resultObject['rep:userId'] !== 'everyone' &&
177                                 resultObject['rep:userId'] !== 'anonymous') {
178 
179                                 itemCount++;
180                                 return true;
181                             }
182                             return false;
183                     });
184 
185                     json_data.total = itemCount;
186                     // Render list of objects
187                     renderList(0, json_data);
188                 }
189             }
190         });
191 
192         /*if(!$.isEmptyObject(objects)){
193             // Render list of objects
194             renderList(0, objects);
195         }*/
196     };

199     /**
200      * renderList
201      * Renders the list of objects the user can select from
202      * @pageNumber {Int} The page we want to load
203      * @objects {Object} An object containing the elements to list
204      * @returns void
205      */
206     var renderList = function(pageNumber, objects) {
207         sakai.data.listpeople[listType].userList = {};
208         sakai.data.listpeople[listType].total = 0;
209         var rawData = objects;
210 
211         // main container
212         var uniqueIdentifier = Math.round(Math.random() + 1000);
213         var $listpeople_page_container = $("<ul id=\"listpeople_page_" + pageNumber + "\" class=\"listpeople_page loadinganim " + uniqueIdentifier + "\"></ul>");
214 
215         // Display empty new container with loading anim
216         //$listpeople_container.append($listpeople_page_container);
217 
218         addToList(rawData);
219 
220         var json_data = {
221             "userList" : sakai.data.listpeople[listType].userList,
222             "selectable" : sakai.config.listpeople[listType].selectable
223         };
224 
225         // Override selectable property if needed
226         // This should be overridden when a user is not a manager of the content and set to false
227         if (sakai.config.listpeople[listType].anon){
228             json_data.selectable = false;
229         }
230         json_data.pageNumber = pageNumber;
231 
232         // Render the results data template
233         var pageHTML = $.TemplateRenderer($listpeople_content_pagetemplate, json_data);
234 
235         // Display count of items
236         $listpeople_count.html(sakai.data.listpeople[listType].total);
237 
238         // Remove loading animation
239         //$listpeople_page_container.removeClass("loadinganim");
240 
241         // Inject results into DOM
242         $listpeople_container.html(pageHTML);
243 
244         // Wire item selection
245         if (sakai.config.listpeople[listType].selectable) {
246 
247             $(".listpeople_page" + " li", $rootel).addClass("selectable");
248             $("#listpeople_page_" + pageNumber + " li", $rootel).bind("click", function(e){
249                 // Check if user click on top of a link
250                 if (e.target.tagName.toLowerCase() !== "a") {
251                     // Remove from selected list
252                     if ($(this).hasClass("listpeople_selected")) {
253                         $(this).removeClass("listpeople_selected");
254                         $(this).children("input").attr('checked', false);
255                         delete sakai.data.listpeople[listType]["selected"][$(this).attr("id")];
256                         sakai.data.listpeople[listType].selectCount -= 1;
257                     } else {
258                         // Add to selected list
259                         $(this).addClass("listpeople_selected");
260                         $(this).children("input").attr('checked', true);
261                         sakai.data.listpeople[listType].selectCount += 1;
262 
263                         for (var i = 0; i < sakai.data.listpeople[listType].total; i++) {
264                             if (sakai.data.listpeople[listType].userList[$(this).attr("id")]['rep:userId'] == [$(this).attr("id")] || sakai.data.listpeople[listType].userList[$(this).attr("id")]['userid'] == [$(this).attr("id")] || sakai.data.listpeople[listType].userList[$(this).attr("id")]['sakai:group-id'] == [$(this).attr("id")] || sakai.data.listpeople[listType].userList[$(this).attr("id")]['content_id'] == [$(this).attr("id")]) {
265                                 sakai.data.listpeople[listType]["selected"][$(this).attr("id")] = sakai.data.listpeople[listType].userList[$(this).attr("id")];
266                             }
267                         }
268                     }
269                     // trigger event when add/remove from the list
270                     $(window).trigger("list-people-selected-change",tuid);
271                 }
272                 $listpeople_count_items.hide();
273                 $listpeople_count_of.show();
274                 $listpeople_count_total.show();
275                 $listpeople_count_selected.show();
276                 $(".listpeople_page" + " li", $rootel).addClass("selectable");
277                 $listpeople_count.html(sakai.data.listpeople[listType].selectCount);
278             });
279         }
280 
281         //Update known total amount of displayed elements
282         sakai.data.listpeople[listType].currentElementCount += sakai.data.listpeople[listType].total;
283 
284         //Set search result count
285         // If we know the exact total display it
286         $listpeople_count_total.html(sakai.data.listpeople[listType].total);
287 
288         // Wire sorting select dropdown
289         $listpeople_sort_order.bind("change", function(e){
290             var sortOrder = $listpeople_sort_order.val();
291             sortList(pageNumber, sortOrder);
292         });
293 
294         // sort list
295         sortList(pageNumber, sakai.config.listpeople[listType].sortOrder);
296         pagerClickHandler(1);
297         // SAKIII-1714 - In IE7, the list didn't rerender properly. Adding in this
298         // resolved the problem
299         if ($.browser.msie) {
300             $listpeople_page_container.hide();
301             setTimeout("$(\"." + uniqueIdentifier + "\").show()", 20);
302         }
303         
304     };
305 
306     
307     var pagerClickHandler = function(clicked){
308        if (sakai.data.listpeople[listType].total > totalResult) {
309             var pageNumber = (parseInt(clicked, 10) - 1) * totalResult;
310             var listItems = $("#listpeople_page_0", $rootel).children('li');
311             
312             // hide all the list items.
313             $(listItems).hide();
314 
315             // show only 5 items at a time
316             for (var i = pageNumber; i < pageNumber+totalResult; i++) {
317                 $(listItems).eq(i).show();
318             }
319             
320             var totalNumberItems = sakai.data.listpeople[listType].total;
321             $('.jq_pager', $rootel).pager({
322                 pagenumber: pageNumber,
323                 pagecount: Math.ceil(totalNumberItems / totalResult),
324                 buttonClickCallback: pagerClickHandler
325             });
326             $('.jq_pager', $rootel).show();
327         }
328     };

330     /**
331      * sortList
332      * Returns an array of selected items
333      * @param pageNumber {String} Number of the page to sort
334      * @param sortOrder {String} Order to sort the list by
335      * @returns void
336      */
337     var sortList = function(pageNumber, sortOrder) {
338         var mylist = $("#listpeople_page_" + pageNumber, $rootel);
339         var listitems = mylist.children('li').get();
340         listitems.sort(function(a, b) {
341             var compA = $(a).text().toUpperCase();
342             var compB = $(b).text().toUpperCase();
343             if (sortOrder === 'ascending') {
344                 return (compA < compB) ? -1 : (compA > compB) ? 1 : 0;
345             }
346             return (compB < compA) ? -1 : (compB > compA) ? 1 : 0;
347         });
348         $.each(listitems, function(idx, itm) { mylist.append(itm); });
349     };

362     /**
363      * addToList
364      * Takes an object/array of users and adds them to the list to be rendered.
365      * @param object {Object} The object of users to add
366      * @returns viod
367      */
368     var addToList = function(object) {
369 
370         if(!$.isEmptyObject(object)){
371             // Get display name and subName for each user or group and add to the list object. This should also filter out the anonymous user
372             $.each(object.results, function(index, resultObject) {
373                 var iSubNameInfoGroup = sakai.config.listpeople[listType]["subNameInfoGroup"];
374                 var iSubNameInfoUser = sakai.config.listpeople[listType]["subNameInfoUser"];
375                 var iSubNameInfoContent = sakai.widgets.widgets.listpeople.subNameInfoContent;
376                 if (resultObject.userid) {
377                     // get user details
378                     sakai.data.listpeople[listType].userList[resultObject.userid] = resultObject;
379                     sakai.data.listpeople[listType].total += 1;
380                     if (sakai.api.User.getDisplayName(resultObject)) {
381                         sakai.data.listpeople[listType].userList[resultObject.userid]["displayName"] = sakai.api.User.getDisplayName(resultObject);
382                     } else if (resultObject["firstName"] && resultObject["lastName"]) {
383                         sakai.data.listpeople[listType].userList[resultObject.userid]["displayName"] = resultObject["firstName"] + ' ' + resultObject["lastName"];
384                     } else {
385                         sakai.data.listpeople[listType].userList[resultObject.userid]["displayName"] = resultObject.userid;
386                     }
387                     if (resultObject.picture && typeof(resultObject.picture) === 'string') {
388                         sakai.data.listpeople[listType].userList[resultObject.userid]["picture"] = $.parseJSON(resultObject.picture);
389                     }
390                     if (!sakai.data.listpeople[listType].userList[resultObject.userid]["subNameInfo"]) {
391                         sakai.data.listpeople[listType].userList[resultObject.userid]["subNameInfo"] = resultObject[iSubNameInfoUser];
392                     }
393                 } else if (resultObject.groupid) {
394                     // get group details
395                     sakai.data.listpeople[listType].userList[resultObject.groupid] = resultObject;
396                     sakai.data.listpeople[listType].total += 1;
397                     if (resultObject.picture && typeof(resultObject.picture) === 'string') {
398                         sakai.data.listpeople[listType].userList[resultObject.userid]["picture"] = $.parseJSON(resultObject.picture);
399                     }
400                     if (!sakai.data.listpeople[listType].userList[resultObject.groupid]["subNameInfo"]) {
401                         sakai.data.listpeople[listType].userList[resultObject.groupid]["subNameInfo"] = resultObject[iSubNameInfoGroup];
402                     }
403                 } else if (resultObject['sakai:group-id']) {
404                     // get group details
405                     sakai.data.listpeople[listType].userList[resultObject['sakai:group-id']] = resultObject;
406                     sakai.data.listpeople[listType].total += 1;
407                     if (resultObject.picture && typeof(resultObject.picture) === 'string') {
408                         sakai.data.listpeople[listType].userList[resultObject['sakai:group-id']]["picture"] = $.parseJSON(resultObject.picture);
409                     }
410                     if (!sakai.data.listpeople[listType].userList[resultObject['sakai:group-id']]["subNameInfo"]) {
411                         sakai.data.listpeople[listType].userList[resultObject['sakai:group-id']]["subNameInfo"] = resultObject[iSubNameInfoGroup];
412                     }
413                 } else if (sakai.api.User.getDisplayName(resultObject) && resultObject['rep:userId']) {
414                     // get user details
415                     sakai.data.listpeople[listType].userList[resultObject['rep:userId']] = resultObject;
416                     sakai.data.listpeople[listType].userList[resultObject['rep:userId']]["displayName"] = sakai.api.User.getDisplayName(resultObject);
417                     if (resultObject.picture && typeof(resultObject.picture) === 'string') {
418                         sakai.data.listpeople[listType].userList[resultObject['rep:userId']]["picture"] = $.parseJSON(resultObject.picture);
419                     }
420                     if (!sakai.data.listpeople[listType].userList[resultObject['rep:userId']]["subNameInfo"]) {
421                         sakai.data.listpeople[listType].userList[resultObject['rep:userId']]["subNameInfo"] = resultObject[iSubNameInfoUser];
422                     }
423                     sakai.data.listpeople[listType].total += 1;
424                 } else if (resultObject["jcr:primaryType"] === "sakai:pooled-content") {
425                     // get content details
426                     if (!resultObject["jcr:name"] && resultObject["content_id"]) {
427                         resultObject["jcr:name"] = resultObject["content_id"];
428                     }
429 
430                     var content_path = '/p/' + resultObject["jcr:name"];
431 
432                     $.ajax({
433                         url: sakai.config.SakaiDomain + content_path + ".2.json",
434                         async: false,
435                         success: function(data){
436 
437                             sakai.data.listpeople[listType].userList[resultObject["jcr:name"]] = data;
438                             sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]['content_id'] = resultObject["jcr:name"];
439                             sakai.data.listpeople[listType].total += 1;
440                             if (sakai.config.MimeTypes[data["jcr:content"]["jcr:mimeType"]]) {
441                                 sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]['avatar'] = sakai.config.MimeTypes[data["jcr:content"]["jcr:mimeType"]].URL;
442                                 sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]['mimeTypeDescripton'] = sakai.config.MimeTypes[data["jcr:content"]["jcr:mimeType"]].description;
443                             } else {
444                                 sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]['avatar'] = "/dev/images/mimetypes/empty.png";
445                                 sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]['mimeTypeDescripton'] = sakai.config.MimeTypes.other.description;
446                             }
447                             if (!sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]["subNameInfo"]) {
448                                 sakai.data.listpeople[listType].userList[resultObject["jcr:name"]]["subNameInfo"] = data[iSubNameInfoContent];
449                             }
450                         }
451                     });
452                 }
453             });
454         }
455     };

458     /**
459      * removeFromList
460      * Removes selected items from the list and re-renders list
461      * @returns viod
462      */
463     sakai.listpeople.removeFromList = function() {
464         $.each(sakai.data.listpeople[listType]["selected"], function(index, resultObject) {
465             delete sakai.data.listpeople[listType].userList[index];
466         });
467         var tempList = {};
468         tempList.results = sakai.data.listpeople[listType].userList;
469 
470         reset();
471 
472         if(!$.isEmptyObject(tempList)){
473             // Re-render list of objects
474             renderList(0, tempList);
475         }
476     };

478     var init = function() {
479         // Send out an event that says the widget is ready to
480         // accept a search query to process and display. This event can be picked up
481         // in a page JS code
482         $(window).bind("sakai-listpeople-render", function(e, data) {
483             if (data.tuid === tuid) {
484                 listType = data.listType;
485                 sakai.data.listpeople[listType] = {
486                     "selected": {},
487                     "currentElementCount": 0,
488                     "selectCount": 0,
489                     "total": 0,
490                     "userList": {}
491                 };
492                 sakai.config.listpeople[listType] = {
493                     "items": 1000,
494                     "selectable": false,
495                     "sortOn": "lastName",
496                     "sortOrder": "ascending",
497                     "function": "getSelection",
498                     "anon": false
499                 };
500                 render(data.tuid, data.pl_config, data.url, data.id);
501             }
502         });
503         $(window).trigger("sakai-listpeople-ready", tuid);
504         sakai.listpeople.isReady = true;
505         sakai.data.listpeople[tuid] = sakai.data.listpeople[tuid] || {};
506         sakai.data.listpeople[tuid].isReady = true;
507     };
508     init();
509 
510 };
