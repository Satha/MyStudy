 18 /*global $ */
 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.listpeoplewrappergroup
 24  *
 25  * @class listpeoplewrappergroup
 26  *
 27  * @description
 28  * Initialize the listpeoplewrappergroup widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  */
 33 sakai.listpeoplewrappergroup = function(tuid, showSettings){
 34 
 35     var rootel = "#" + tuid;
 36     var listType = "members";
 37 
 38     ///////////////////
 39     // CSS Selectors //
 40     ///////////////////
 41 
 42     var listpeoplewrappergroup = "#listpeoplewrappergroup";
 43     var listpeoplewrappergroupContainer = listpeoplewrappergroup + "_container";
 44     var listpeoplewrappergroupDefaultTemplate = listpeoplewrappergroup + "_default_template";
 45     var listpeoplewrappergroupSettings = listpeoplewrappergroup + "_settings";
 46     var listpeoplewrappergroupSettingsCancel = listpeoplewrappergroupSettings + "_cancel";
 47     var listpeoplewrappergroupSettingsFinish = listpeoplewrappergroupSettings + "_finish";
 48     var listpeoplewrappergroupSettingsSelect = listpeoplewrappergroupSettings + "_select";
 49 
 50     //////////////////////
 51     // Render functions //
 52     //////////////////////
 53 
 54     /**
 55      * Render the template
 56      */
 57     var renderTemplateListpeople = function() {
 58         var listTitle = listType.charAt(0).toUpperCase() + listType.substring(1,listType.length);
 59         $(listpeoplewrappergroupContainer, rootel).html($.TemplateRenderer(listpeoplewrappergroupDefaultTemplate, {listType: listType, tuid: tuid, listTitle: listTitle}));
 60         var newTitle = $(".listpeoplewrappergroup_header", rootel).text();
 61         if (sakai.api.Widgets.isOnDashboard(tuid)) {
 62             sakai.api.Widgets.changeWidgetTitle(tuid, newTitle);
 63         } else {
 64             $(".listpeoplewrappergroup_header", rootel).show();
 65         }
 66     };
 67 
 68     /**
 69      * Render Widget with group data
 70      */
 71     var loadGroupElements = function(){
 72         var pl_config = {"selectable":false, "subNameInfoUser": "", "subNameInfoGroup": "sakai:group-description", "sortOn": "lastName", "sortOrder": "ascending", "items": 1000, "function": "getSelection" },
 73             url = "";
 74 
 75         if (listType === "members") {
 76             // get group members
 77             url = "/system/userManager/group/" + sakai.currentgroup.data.authprofile["sakai:group-id"] + ".members.detailed.json";
 78         } else if (listType === "managers") {
 79             // get group managers
 80             url = "/system/userManager/group/" + sakai.currentgroup.data.authprofile["sakai:group-id"] + ".managers.detailed.json";
 81         } else if (listType === "content") {
 82             // get group content
 83             url = "/var/search/pool/files?group=" + sakai.currentgroup.data.authprofile["sakai:group-id"];
 84         }
 85 
 86         $(window).trigger("sakai-listpeople-render", {"tuid": listType+tuid, "listType": listType, "pl_config": pl_config, "url": url, "id": sakai.currentgroup.data.authprofile["sakai:group-id"]});
 87     };
 88 
 89     //////////////
 90     // Bindings //
 91     //////////////
 92 
 93     /**
 94      * Bind the listpeople widget
 95      */
 96     var addListpeopleBinding = function(){
 97         // Bind the listpeople widget
 98         if (sakai.listpeople && sakai.listpeople.isReady && sakai.data.listpeople[(listType+tuid)] && sakai.data.listpeople[(listType+tuid)].isReady) {
 99             loadGroupElements();
100         } else {
101             $(window).bind("sakai-listpeople-ready", function(e, iTuid) {
102                 if (iTuid === (listType+tuid)) {
103                     loadGroupElements();
104                 }
105             });
106         }
107     };
108 
109     /**
110      * Bind the save/cancel buttons
111      */
112     var addSettingBinding = function(){
113         $(listpeoplewrappergroupSettingsCancel, rootel).bind("click",function(e,ui){
114             sakai.api.Widgets.Container.informCancel(tuid, "listpeoplewrappergroup");
115         });
116         $(listpeoplewrappergroupSettingsFinish, rootel).bind("click",function(e,ui){
117             var listType = $(listpeoplewrappergroupSettingsSelect, rootel).val();
118             if (listType) {
119                 sakai.api.Widgets.saveWidgetData(tuid, {listType: listType}, function(success, data) {
120                     if ($(".sakai_dashboard_page").is(":visible")) {
121                         showSettings = false;
122                         showHideSettings(showSettings);
123                     }
124                     else {
125                         sakai.api.Widgets.Container.informFinish(tuid, "listpeoplewrappergroup");
126                     }
127                 });
128             }
129         });
130     };
131 
132     ////////////////////
133     // Initialization //
134     ////////////////////
135 
136     /**
137      * Shows or hides the settings screen
138      * @param {Object} show
139      */
140     var showHideSettings = function(show){
141         if(show){
142             addSettingBinding();
143             $(listpeoplewrappergroupSettings, rootel).show();
144         } else if (sakai.currentgroup.data) {
145             $(listpeoplewrappergroupSettings, rootel).hide();
146             renderTemplateListpeople();
147             addListpeopleBinding();
148         }
149     };
150 
151     /**
152      * doInit
153      */
154     var doInit = function(){
155         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
156             if (success) {
157                 resultJSON = data;
158                 if (data.listType) {
159                     listType = data.listType;
160                     $(listpeoplewrappergroupSettingsSelect, rootel).val(listType);
161                 }
162             }
163             showHideSettings(showSettings);
164 
165             // Inserts the listpeople widget
166             sakai.api.Widgets.widgetLoader.insertWidgets(tuid);
167         });
168     };
169 
170     doInit();
171 };
172 
173 sakai.api.Widgets.widgetLoader.informOnLoad("listpeoplewrappergroup");
