 20 /*global Querystring, Config, $,  set_cookie */
 21 
 22 
 23 var sakai = sakai || {};
 24 
 25 /**
 26  * @name sakai.login
 27  *
 28  * @param {String} tuid Unique id of the widget
 29  * @param {Boolean} showSettings Show the settings of the widget or not
 30  */
 31 sakai.login = function(){
 32 
 33 
 34     /////////////////////////////
 35     // Configuration variables //
 36     /////////////////////////////
 37 
 38     var redirectUrl = sakai.config.URL.MY_DASHBOARD_URL;
 39     var usernameField = "username";
 40     var passwordField = "password";
 41     var failMessage = "#login-failed";
 42     var loadingMessage = "#login-loader";
 43     var registerLink = "#register_here";
 44     var loginButton = "#loginbutton";
 45     var loginDefault = "#login-default";
 46     var loginExternal = "#login-external";
 47     var loginExternalSystem = loginExternal + "-system";
 48     var loginExternalButton = loginExternal + "-button";
 49     var loginForm = "#login-container";
 50 
 51     var currentUserName; 
 52     var currentPassword;
 53 
 54     /////////////////////
 55     // Login functions //
 56     /////////////////////
 57 
 58     /**
 59      * This will determine whether there is a valid session. If there is, we'll
 60      * redirect to the URL requested or the personal dashboard if nothing has been provided.
 61      */
 62     var decideLoggedIn = function(data){
 63 
 64         var mejson = (data === undefined ? sakai.data.me : data);
 65         if (mejson.user.userid) {
 66             document.location = redirectUrl;
 67         } else {
 68             $(loadingMessage).hide();
 69             
 70             // check if internal is true or internal account creation is true show login
 71             if (sakai.config.Authentication.internal || sakai.config.Authentication.allowInternalAccountCreation) {
 72                 $(loginButton).show();
 73                 $(registerLink).show();
 74                 $(loginDefault).show();
 75                 if (data) {
 76                     $(failMessage).show();
 77                     $("#username").addClass("error");
 78                     $("#password").addClass("error");
 79                 }
 80                 // Set the cursor in the username field
 81                 $("#" + usernameField).focus();
 82             } else {
 83                 // loop through and render each external authentication system
 84                 $.each(sakai.config.Authentication.external, function(index, value) {
 85                     var curNum = $('.login-external-system').length - 1;
 86                     var newNum = curNum + 1;
 87                     var newElem = $(loginExternalSystem + '0').clone().attr('id', 'login-external-system' + newNum);
 88                     newElem.children(':first').attr('id', 'login-external-button' + newNum).attr('login-external-button', 'login-external-button' + newNum);
 89                     $(loginExternalSystem + curNum).after(newElem);
 90                     $(loginExternalSystem + newNum + ' .login-external-label').text(sakai.api.Security.saneHTML(value.label));
 91                     if (value.description) {
 92                         $(loginExternalSystem + newNum + ' .login-external-description').text(sakai.api.Security.saneHTML(value.description));                    
 93                     }
 94                     $(loginExternalSystem + newNum + ' .login-external-url').text(sakai.api.Security.saneHTML(value.url));
 95                     $(loginExternalSystem + newNum).show();
 96                     
 97                     // bind external url
 98                     $(loginExternalButton + newNum).bind("click", function(){
 99                         document.location = value.url;
100                     });
101                 });
102                 $(loginExternal).show();
103             }
104         }
105 
106     };
107 
108     /**
109      * This will be executed after the post to the login service has finished.
110      * We send a new request to the Me service, explicity disabling cache by
111      * adding a random number behind the URL, becasue otherwise it would get
112      * the cached version of the me object which would still say I'm not logged
113      * in.
114      */
115     var checkLogInSuccess = function(){
116 
117         $.ajax({
118             url : sakai.config.URL.ME_SERVICE,
119             cache : false,
120             success : decideLoggedIn,
121             error: function(xhr, textStatus, thrownError) {
122                 throw "Me service has failed";
123             }
124         });
125 
126     };
127 
128     /**
129      * This will extract the username and password entered by the user, will hide the
130      * login button and the register here link andwill put up a Signing in ... message
131      * It will then call the login service and attempt to log you in. Once the login request
132      * has completed, we'll do a new request to the me service and check whether we're
133      * logged in
134      */
135     var performLogIn = function(){
136 
137         var values = $(loginForm).serializeObject();
138 
139         if (currentUserName !== values[usernameField] || currentPassword !== values[passwordField]) {
140             currentUserName = values[usernameField];
141             currentPassword = values[passwordField];
142             
143             $("#username").removeClass("error");
144             $("#password").removeClass("error");
145 
146             $(failMessage).hide();
147             $(loginButton).hide();
148             $(registerLink).hide();
149             $(loadingMessage).show();
150             var data = {
151                 "username": values[usernameField],
152                 "password": values[passwordField]
153             };
154             
155             // Perform the login operation
156             sakai.api.User.login(data, checkLogInSuccess);
157             
158         }
159         return false;
160 
161     };
162 
163 
164     ////////////////////
165     // Event Handlers //
166     ////////////////////
167 
168     /*
169      * When the user is trying to initiate the form submission,
170      * we initiate the login function
171      */
172     $(loginForm).submit(performLogIn);
173 
174 
175     /////////////////////////////
176     // Initialisation function //
177     /////////////////////////////
178 
179     var doInit = function(){
180 
181         /*
182          * If you were logged in to the system, and the session has expired,
183          * it will redirect you to the login page with the URL of the page you
184          * were on encoded into a "url" querystring variable. We check whether
185          * the "url" querystring parameter is there, and if it is, we'll decode
186          * that URL and redirect to that location on successful login. If it isn't
187          * set, we'll redirect to the personal dashboard
188          */
189         var qs = new Querystring();
190         var red = qs.get("url", false);
191         if (red !== false){
192             redirectUrl = decodeURIComponent(red);
193         }
194 
195         // Check whether we are already logged in
196         decideLoggedIn();
197     };
198 
199     doInit();
200 
201 };
202 
203 sakai.api.Widgets.Container.registerForLoad("sakai.login");
