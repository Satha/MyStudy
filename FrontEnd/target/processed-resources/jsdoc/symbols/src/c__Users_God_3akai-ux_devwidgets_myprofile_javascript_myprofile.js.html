 18 /*global $ */
 19 
 20 var sakai = sakai || {};
 21 
 22 /**
 23  * @name sakai.myprofile
 24  *
 25  * @class myprofile
 26  *
 27  * @description
 28  * Initialize the myprofile widget
 29  *
 30  * @version 0.0.1
 31  * @param {String} tuid Unique id of the widget
 32  * @param {Boolean} showSettings Show the settings of the widget or not
 33  */
 34 sakai.myprofile = function (tuid, showSettings) {
 35 
 36 
 37     /////////////////////////////
 38     // Configuration variables //
 39     /////////////////////////////
 40 
 41     var rootel = $("#" + tuid);
 42     var me = sakai.data.me;
 43     var json = me.profile;
 44 
 45     //    IDs
 46     var myprofileId = "#myprofile";
 47     var myprofileClass = ".myprofile";
 48 
 49     var profileNameID = myprofileId + "_name";
 50     var profilePictureID = myprofileId + "_pic";
 51     var profileDepartementID = myprofileId + "_dept";
 52 
 53     var profileChatStatusID = myprofileId + "_chat_status_";
 54     var profileChatStatus = "chat_available_status_";
 55     var profileStatusContainer = myprofileId + "_status";
 56     var profileChatStatusClass = myprofileClass + "_chat_status";
 57     var profileChatStatusDropDownLink = myprofileClass + "_chat_status_dropdown_link";
 58     var profileChatStatusPicker = myprofileClass + "_chat_status_picker";
 59     var profileWidget = myprofileClass + "_widget";
 60 
 61     var profilePreviewLink = "#myprofile_preview_profile";
 62 
 63     var headerChatUserId = "#user_link"; // The username in the chat bar.
 64 
 65     var chatStatus = "online";
 66 
 67     var authprofileURL;
 68 
 69     /////////////////
 70     // Chat status //
 71     /////////////////
 72 
 73     /**
 74      * Update the chat statuses all across the page.
 75      * @param {Object} status
 76      */
 77     var updateChatStatus = function (status) {
 78         $(profileChatStatusClass).hide();
 79         $(profileChatStatusID + status).show();
 80 
 81         // Trigger the chat_status_change event to update other widgets
 82         $(window).trigger("chat_status_change", chatStatus);
 83     };
 84 
 85     /**
 86      * Change the status of the currently logged in user.
 87      * @param {Object} status
 88      */
 89     var changeStatus = function (chatstatus) {
 90         if (chatStatus !== chatstatus){
 91             tempChatStatus = chatstatus;
 92 
 93             $(profileStatusContainer).toggle();
 94 
 95             var data = {
 96                 "chatstatus": chatstatus,
 97                 "_charset_": "utf-8"
 98             };
 99 
100             $.ajax({
101                 url: "/~" + sakai.data.me.profile["rep:userId"] + "/public/authprofile",
102                 type: "POST",
103                 data: data,
104                 success: function(data){
105                     updateChatStatus(status);
106                 },
107                 error: function(xhr, textStatus, thrownError){
108                     if ($.isFunction(callback)) {
109                         callback(false, xhr);
110                     }
111                     debug.error("Entity widget - An error occured when sending the status to the server.");
112                 }
113             });
114 
115             sakai.data.me.profile = $.extend(true, sakai.data.me.profile, {"chatstatus": chatstatus});
116 
117         }
118     };
119 
120 
121     //////////////////////////////
122     // Initialisation Functions    //
123     //////////////////////////////
124 
125     var doInit = function () {
126 
127         $(profilePreviewLink, rootel).attr("href", "/~" + sakai.data.me.user.userid);
128 
129         // Check if we have a first and last name
130         if (sakai.api.User.getDisplayName(json) !== "") {
131             $(profileNameID, rootel).text(sakai.api.Util.shortenString(sakai.api.User.getDisplayName(json), 30));
132             $(profileNameID, rootel).attr("href", "/~" + sakai.data.me.user.userid);
133         }
134         else {
135             $(profileNameID, rootel).text(sakai.api.Security.saneHTML(me.user.userid));
136         }
137 
138         authprofileURL = "/~" + sakai.data.me.user.userid + "/public/authprofile";
139 
140         // Do we have a picture
141         if (json.picture) {
142             var pict = $.parseJSON(json.picture);
143             if (pict.name) {
144                 $(profilePictureID, rootel).attr("src", "/~" + sakai.data.me.user.userid + "/public/profile/" + pict.name );
145             }
146         } else {
147             $(profilePictureID, rootel).attr("src", sakai.config.URL.USER_DEFAULT_ICON_URL);
148         }
149 
150         // Any extra information we may have.
151         var extra = "";
152         if (json.basic) {
153             var basic = $.parseJSON(json.basic);
154             if (json.unirole) {
155                 extra = json.unirole;
156             }
157             else if (json.unicollege) {
158                 extra = json.unicollege;
159             }
160             else if (json.unidepartment) {
161                 extra = json.unidepartment;
162             }
163             $(profileDepartementID, rootel).html(sakai.api.Security.saneHTML(extra));
164         }
165 
166         var chatstatus = "online";
167         chatStatus = "online";
168         // Get the user his chatstatus
169         if (me.profile.chatstatus) {
170             chatstatus = me.profile.chatstatus;
171             chatStatus = me.profile.chatstatus;
172         }
173 
174         // Set the status in front of the user his name/
175         $(profileNameID).addClass(profileChatStatus + chatstatus);
176         $(profileChatStatusID + chatstatus).show();
177 
178         // Show the widget after everything is loaded
179         $(profileWidget).show();
180     };
181 
182     /**
183      * Update the position of the status box which is position:absolute
184      */
185     var updateStatusContainerPosition = function(){
186         $(profileStatusContainer).css("left", $(".profile_chat_status_dropdown_link").offset().left + 65 + "px");
187         $(profileStatusContainer).css("top", $(".profile_chat_status_dropdown_link").offset().top + 30 + "px");
188     };
189 
190     /**
191      * Hide the status pop-up
192      */
193     var hideStatusContainer = function(){
194         $(profileStatusContainer).hide();
195     };
196 
197     /**
198      * Toggle the status pop-up
199      */
200     var toggleStatusContainer = function(){
201         updateStatusContainerPosition();
202         $(profileStatusContainer).toggle();
203     };
204 
205 
206     ////////////////////
207     // Event Handlers //
208     ////////////////////
209 
210     // Toggle
211     $(profileChatStatusDropDownLink).bind("click", function (ev) {
212         toggleStatusContainer();
213     });
214     $(profileChatStatusClass).bind("click", function (ev) {
215         toggleStatusContainer();
216     });
217 
218     /**
219      * Bind the document on click event and check if it if you didn't click on the profile chatstatus link
220      */
221     $(document).click(function(e){
222         var $clicked = $(e.target);
223 
224         // Check if one of the parents is the chatstatuscontainer
225         if(!$clicked.parents().is("#profile_chat_status")){
226             hideStatusContainer();
227         }
228 
229     });
230 
231     /**
232      * Bind the resize event
233      */
234     $(window).resize(function(){
235 
236         // If the box is visible, update the position of it
237         if($(profileStatusContainer).is(":visible")){
238             updateStatusContainerPosition();
239         }
240     });
241 
242     // A user selects his status
243     $(profileChatStatusPicker).live("click", function (ev) {
244         var statusChosen = this.id.split("_")[this.id.split("_").length - 1];
245         changeStatus(statusChosen);
246         chatStatus = statusChosen;
247     });
248 
249     // Add binding to set the status
250     $(window).bind("chat_status_change", function(event, currentChatStatus){
251         sakai.api.Util.updateChatStatusElement($(profileNameID), currentChatStatus);
252         chatStatus = currentChatStatus;
253         $(profileChatStatusClass).hide();
254         $(profileChatStatusID + currentChatStatus).show();
255     });
256 
257     doInit();
258 };
259 
260 sakai.api.Widgets.widgetLoader.informOnLoad("myprofile");
