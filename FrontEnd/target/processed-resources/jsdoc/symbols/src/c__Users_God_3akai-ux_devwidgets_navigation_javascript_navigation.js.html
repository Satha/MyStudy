111     var sortByURL = function(a,b){
112         if (a.path > b.path){
113             return 1;
114         } else if (a.path < b.path){
115             return -1;
116         } else {
117             return 0;
118         }

121     // Create arrays of full URL elements
122     var fullURLs = function(site_object) {
123 
124         var full_urls = [];
125 
126         for (var current_url_title in site_object) {
127 
128             if (site_object[current_url_title]["jcr:path"]) {
129 
130                 var raw_path_elements = site_object[current_url_title]["jcr:path"].split("/");
131                 var path_elements = [];
132                 var raw_path_element = "";
133 
134                 for (var j=1; j<sakai.sitespages.config.startlevel; j++) {
135                     raw_path_element += "/" + raw_path_elements[j];
136                 }
137 
138                 // Consider only elements below the start level, and discard "_pages" or empty entries
139                 for (var i=sakai.sitespages.config.startlevel , current_level = raw_path_elements.length; i<current_level; i++) {
140                     raw_path_element += "/" + raw_path_elements[i];
141                     if ((raw_path_elements[i] !== "_pages") && (raw_path_elements[i] !== "")) {
142                         path_elements.push(raw_path_element);
143                     }
144                 }
145 
146                 full_urls.push(path_elements);
147             }
148         }
149         return full_urls.sort();
150     };

152     // Get a page object by it's jcr path
153     var getPageInfoByLastURLElement = function(i_jcr_path) {
154         var return_object = {},
155             jcr_path;
156         for (var i in sakai.sitespages.site_info._pages) {
157             if (sakai.sitespages.site_info._pages.hasOwnProperty(i)) {
158                 if (sakai.sitespages.site_info._pages[i]["jcr:path"]) {
159                     jcr_path = sakai.sitespages.site_info._pages[i]["jcr:path"];
160                 } else {
161                     continue;
162                 }
163                 if (jcr_path === i_jcr_path) {
164                     return_object = sakai.sitespages.site_info._pages[i];
165                 }
166             }
167         }
168         return return_object;
169     };

171     // Converts array of URL elements to a hierarchical structure
172     var convertToHierarchy = function(url_array) {
173         var item, path;
174 
175         // Discard duplicates and set up parent/child relationships
176         var children = {};
177         var hasParent = {};
178         for (var i = 0, j = url_array.length; i<j; i++) {
179             path = url_array[i];
180             var parent = null;
181             for (var k = 0, l = path.length; k<l; k++)
182             {
183                 item = path[k];
184                 if (!children[item]) {
185                     children[item] = {};
186                 }
187                 if (parent) {
188                     children[parent][item] = true; /* dummy value */
189                     hasParent[item] = true;
190                 }
191                 parent = item;
192             }
193         }
194 
195         // Now build the hierarchy
196         var result = [];
197         for (item in children) {
198             if (!hasParent[item]) {
199                 result.push(buildNodeRecursive(item, children));
200             }
201         }
202         return result;
203     };
204 
205     // Recursive helper to create URL hierarchy
206     var buildNodeRecursive = function(url_fragment, children) {
207 
208         var page_info = getPageInfoByLastURLElement(url_fragment);
209 
210         // Navigation node data
211         var p_title = "";
212         var p_id = "";
213         var p_pagePosition;
214         if (page_info["pageTitle"]) {
215             p_title = sakai.api.Security.saneHTML(page_info["pageTitle"]);
216             p_id = "nav_" + page_info["pageURLName"];
217             p_pagePosition = parseInt(page_info.pagePosition, 10);
218         }
219 
220         var node = {
221             attr: { id: p_id },
222             data: {
223                 title: p_title,
224                 attr: {"href": "#"},
225                 pagePosition: p_pagePosition
226             },
227             children:[]
228         };
229         for (var child in children[url_fragment]) {
230             if (children[url_fragment].hasOwnProperty(child)) {
231                 node.children.push(buildNodeRecursive(child, children));
232             }
233         }
234         return node;
235     };

238     /**
239     * This function will sort the pages based on pagePosition
240     * @param {Object} site_objects, the page array
241     */
242     var sortOnPagePosition = function(site_objects){
243 
244         // Bublesort to srt the pages
245         for (var x = 0,l = site_objects.length ; x < l; x++) {
246             for (y = 0; y < (l - 1); y++) {
247                 if (parseFloat(site_objects[y].data.pagePosition,10) > parseFloat(site_objects[y + 1].data.pagePosition ,10)) {
248                     holder = site_objects[y + 1];
249                     site_objects[y + 1] = site_objects[y];
250                     site_objects[y] = holder;
251                 }
252             }
253         }
254         return site_objects;
255     };

520     /**
521      * Renders the pages navigation tree that is the core of the navigation widget
522      *
523      * @param {String} selectedPageUrlName id of the page to select upon initial
524      *   load of the navigation tree. If null, a default page is selected.
525      * @param {Object} site_info_object Contains an array with all the pages, each page is an object.
526      * @param {Boolean} allowDnd true if drag-and-drop editing should be enabled,
527      *   false if it should be disabled.
528      * @return None
529      */
530     var renderPages = function (selectedPageUrlName, site_info_object, allowDnd) {
531         // set the number of pages in the group
532         $pageCount.html("(" + sakai.sitespages.site_info.number_of_pages() + ")");
533 
534         // Create navigation data object
535         var navigationData = convertToHierarchy(fullURLs(site_info_object));
536         sortOnPagePosition(navigationData);
537 
538         // determine which page to initially select
539         var initiallySelect = navigationData[0].attr.id;
540         if(selectedPageUrlName) {
541             initiallySelect = "nav_" + selectedPageUrlName;
542         }
543 
544         // destroy any existing jstree instance
545         $navigationTree.jstree("destroy");
546 
547         // set up new jstree navigation tree
548         var pluginArray = allowDnd ?
549             [ "themes", "json_data", "ui", "cookies", "dnd" ] :
550             [ "themes", "json_data", "ui", "cookies" ];
551         $navigationTree.jstree({
552             "core": {
553                 "animation": 0,
554                 "html_titles": true
555             },
556             "cookies": {
557                 "save_selected": false
558             },
559             "json_data": {
560                 "data": navigationData
561             },
562             "themes": {
563                 "dots": false,
564                 "icons": true
565             },
566             "ui": {
567                 "select_limit": 1,
568                 "initially_select": [initiallySelect.toString()]
569             },
570             "plugins" : pluginArray
571         });
572 
573         // set up new jstree event bindings
574         addJstreeBindings();
575     };

578     /**
579      * Add event bindings for the jstree pages navigation tree
580      */
581     var addJstreeBindings = function () {
582         /**
583          * When a page is selected in the navigation tree, show it
584          */
585         $navigationTree.bind("select_node.jstree", function(e, data) {
586             var selectedPageUrl = $(data.rslt.obj[0]).attr("id").replace("nav_","");
587             // If page is not the current page load it
588             if (sakai.sitespages.selectedpage !== selectedPageUrl) {
589                 History.addBEvent(selectedPageUrl);
590             }
591         });
592 
593 
594         /**
595          * When a page is moved, update its position
596          */
597         $navigationTree.bind("move_node.jstree", function (e, data) {
598             var $moved_node = $(data.rslt.o[0]);      // the moved node
599             var $reference_node = $(data.rslt.r[0]);  // the node moved next to
600             var position = data.rslt.p;               // either: "before", "after", "inside"
601 
602             // the $moved_node has been moved <position = before | after | inside> the $reference_node
603 
604             // Source data - the element being moved
605             var src_url_name = $moved_node.attr("id").replace("nav_","");
606             var src_url = sakai.sitespages.site_info._pages[src_url_name]["jcr:path"];
607             var src_url_title = sakai.sitespages.site_info._pages[src_url_name]["pageURLTitle"];
608             var src_url_depth = sakai.sitespages.site_info._pages[src_url_name]["pageDepth"];
609 
610             // Reference data - the element being moved next to
611             var ref_url_name = $reference_node.attr("id").replace("nav_","");
612             var ref_url = sakai.sitespages.site_info._pages[ref_url_name]["jcr:path"];
613             var ref_url_title = sakai.sitespages.site_info._pages[ref_url_name]["pageURLTitle"];
614             var ref_url_depth = sakai.sitespages.site_info._pages[ref_url_name]["pageDepth"];
615 
616             // Construct target URL
617             var ref_url_elements = ref_url.split("/");
618 
619             // If we are moving a page inside a page add a "_pages" element to the url
620             if (position === "inside")  {
621                 ref_url_elements.push("_pages");
622             } else {
623                 ref_url_elements.pop();
624             }
625 
626             // Construct target URL
627             var tgt_url = ref_url_elements.join("/") + "/" + src_url_title;
628 
629             var changeNextNodes = false;
630 
631             // If there is a depth difference or putting a node inside another the move is a move within a hierarchy
632             if ((src_url_depth !== ref_url_depth) || (position === "inside")) {
633                 // Move page
634                 sakai.sitespages.movePage(src_url, tgt_url, function(newName){
635                     // update reference to the page in the nav
636                     var newID = "nav_" + newName;
637                     $moved_node.attr("id", newID);
638                     $navigationTree.jstree("open_node", $reference_node);
639                     $navigationTree.jstree("select_node", $moved_node);
640                 });
641 
642             } else if((position ==='before') ||(position ==='after')){
643                 var currentNodePage = parseFloat(sakai.sitespages.site_info._pages[src_url_name].pagePosition, 10);
644                 var referenceNodePage = parseFloat(sakai.sitespages.site_info._pages[ref_url_name].pagePosition, 10);
645 
646                 var toUpdatePages = [],
647                     nodePage;
648 
649                 //check if the node has been dropped infront of the reference node or behind
650                 if((position ==='before')){
651                     //Check if the user dragged the node to another node which is higher in the list or not
652                     if (currentNodePage < referenceNodePage) {
653                         // Loop over all the nodes
654                         for (var c in sakai.sitespages.site_info._pages) {
655                             if (sakai.sitespages.site_info._pages.hasOwnProperty(c)) {
656                                 nodePage = parseFloat(sakai.sitespages.site_info._pages[c].pagePosition, 10);
657                                 // make sure that the dropped node isn't in this list, because it has to be updated speratly
658                                 if (sakai.sitespages.site_info._pages[c].pageTitle !== sakai.sitespages.site_info._pages[src_url_name].pageTitle) {
659                                     // Check if the node in the list is smaller than the current node (dragged node) and the smaller than the reference node. Because these will have to get a lower position value
660                                     // These are in fact the nodes that are in front of the reference node
661                                     if ((nodePage > currentNodePage) && (nodePage < referenceNodePage)) {
662                                         sakai.sitespages.site_info._pages[c].pagePosition = nodePage - 200000;
663                                         toUpdatePages.push(sakai.sitespages.site_info._pages[c]);
664                                     }
665                                     // IF this is not the case this means that the node will be after the reference node and it just has to be parsed
666                                     else {
667                                         sakai.sitespages.site_info._pages[c].pagePosition = nodePage;
668                                         toUpdatePages.push(sakai.sitespages.site_info._pages[c]);
669                                     }
670                                 }
671                             }
672                         }
673                         // The node will get the value of the reference node - 2000000, because the node is dragged from underneath the reference node which means that all the nodes
674                         // underneath the referance node will have received a lower value because 1 is gone.
675                         sakai.sitespages.site_info._pages[src_url_name].pagePosition = referenceNodePage - 200000;
676                         toUpdatePages.push(sakai.sitespages.site_info._pages[src_url_name]);
677                         updatePagePosition(toUpdatePages);
678                     } else {
679                         // This happends when a user drags a node from the top, this means that nothing will change to the nodes that are under the reference node,only the nodes above the reference node will have to be updated
680                         sakai.sitespages.site_info._pages[src_url_name].pagePosition = referenceNodePage;
681                         //updateSite(sakai.sitespages.site_info._pages[src_url_name]);
682                         toUpdatePages.push(sakai.sitespages.site_info._pages[src_url_name]);
683                         for (var k in sakai.sitespages.site_info._pages) {
684                             if (sakai.sitespages.site_info._pages.hasOwnProperty(k)) {
685                                 nodePage = parseFloat(sakai.sitespages.site_info._pages[k].pagePosition,10);
686                                 if(nodePage >= parseFloat(sakai.sitespages.site_info._pages[src_url_name].pagePosition,10)&&(sakai.sitespages.site_info._pages[k].pageTitle !==sakai.sitespages.site_info._pages[src_url_name].pageTitle )){
687                                     sakai.sitespages.site_info._pages[k].pagePosition = nodePage + 200000;
688                                     toUpdatePages.push(sakai.sitespages.site_info._pages[k]);
689                                 }
690                             }
691                         }
692                         updatePagePosition(toUpdatePages);
693                     }
694                 } else {
695                     // This is almost exactly the same as the "before" part, there are small diffrences because the reference node is in front of the node when it is dropped
696                     // This means that the nodes before the reference node will have an extra node and the nodes after the reference node will have one less
697                     if (currentNodePage < referenceNodePage) {
698                         for (var z in sakai.sitespages.site_info._pages) {
699                             if (sakai.sitespages.site_info._pages.hasOwnProperty(z)) {
700                                 nodePage = parseFloat(sakai.sitespages.site_info._pages[z].pagePosition, 10);
701                                 if (sakai.sitespages.site_info._pages[z].pageTitle !== sakai.sitespages.site_info._pages[src_url_name].pageTitle) {
702                                     if ((nodePage > currentNodePage) && (nodePage <= referenceNodePage)) {
703                                         sakai.sitespages.site_info._pages[z].pagePosition = nodePage - 200000;
704                                         //updateSite(sakai.sitespages.site_info._pages[c]);
705                                         toUpdatePages.push(sakai.sitespages.site_info._pages[z]);
706                                     }
707                                     else {
708                                         sakai.sitespages.site_info._pages[z].pagePosition = nodePage;
709                                         toUpdatePages.push(sakai.sitespages.site_info._pages[z]);
710                                     }
711                                 }
712                             }
713                         }
714                         sakai.sitespages.site_info._pages[src_url_name].pagePosition = referenceNodePage + 200000;
715                         toUpdatePages.push(sakai.sitespages.site_info._pages[src_url_name]);
716                         updatePagePosition(toUpdatePages);
717                     }
718                     else {
719                         // This is the part where the user drags a node from the top of the list, which again means that only the nodes after the reference node will have to be updated
720                         sakai.sitespages.site_info._pages[src_url_name].pagePosition = referenceNodePage + 200000;
721                         //updateSite(sakai.sitespages.site_info._pages[src_url_name]);
722                         toUpdatePages.push(sakai.sitespages.site_info._pages[src_url_name]);
723                         for (var t in sakai.sitespages.site_info._pages) {
724                             if(parseFloat(sakai.sitespages.site_info._pages[t].pagePosition,10) >= parseFloat(sakai.sitespages.site_info._pages[src_url_name].pagePosition,10)&&(sakai.sitespages.site_info._pages[t].pageTitle !==sakai.sitespages.site_info._pages[src_url_name].pageTitle )){
725                                 sakai.sitespages.site_info._pages[t].pagePosition = parseFloat(sakai.sitespages.site_info._pages[t].pagePosition,10) + 200000;
726                                 //updateSite(sakai.sitespages.site_info._pages[c]);
727                                 toUpdatePages.push(sakai.sitespages.site_info._pages[t]);
728                             }
729                         }
730                         updatePagePosition(toUpdatePages);
731                     }
732                 }
733             }
734             $navigationTree.jstree("deselect_all");
735             $navigationTree.jstree("select_node", $moved_node);
736         });
737     };

740     /**
741      * Function that is available to other functions and called by site.js
742      * It fires the event to render the navigation
743      * @param {String} selectedPageUrlName id of the page to select upon initial
744      *   load of the navigation tree. If null, a default page is selected.
745      * @param {Object} site_info_object Contains an array with all the pages, each page is an object.
746      */
747     sakai.sitespages.navigation.renderNavigation = function(selectedPageUrlName, site_info_object) {
748         // check arguments
749         if(!site_info_object || site_info_object.length === 0) {
750             renderNoPages(true);
751             return;
752         }
753 
754         // determine what the current user is allowed to see
755         // only managers are allowed to edit pages
756         var pagesVisibility;
757         if ($.isEmptyObject(sakai.currentgroup.data)) {
758             pagesVisibility = sakai.config.Permissions.Groups.visible["public"];
759             $settingsIcon.remove();
760             $settingsMenu.remove();
761         } else {
762             pagesVisibility = sakai.currentgroup.data.authprofile["sakai:pages-visible"];
763         }
764 
765         if (sakai.show.canEdit() === true) {
766             // current user is a manager
767             renderReadWritePages(selectedPageUrlName, site_info_object);
768         } else if(pagesVisibility === sakai.config.Permissions.Groups.visible["public"] ||
769             (pagesVisibility === sakai.config.Permissions.Groups.visible.allusers && !sakai.data.me.user.anon) ||
770             (pagesVisibility === sakai.config.Permissions.Groups.visible.members && sakai.api.Groups.isCurrentUserAMember(sakai.currentgroup.id))) {
771             // we have a non-manager that can only view pages, not edit
772             renderReadOnlyPages(selectedPageUrlName, site_info_object);
773         } else {
774             // we have a non-manager that is not allowed to view pages
775             renderNoPages(false);
776         }
777     };
