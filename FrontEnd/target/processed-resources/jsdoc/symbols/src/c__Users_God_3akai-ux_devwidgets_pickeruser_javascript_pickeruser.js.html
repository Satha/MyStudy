105     /**
106      * Reset
107      * Resets the people picker to a default state
108      * @returns void

110     var reset = function() {
111         $pickeruser_content_search.html("");
112         $pickeruser_content_search.unbind("scroll");
113         $pickeruser_message.val("");

120     /**
121      * Render
122      * Renders the people picker
123      * @param iConfig {String} Config element for the widget
124      * @returns void
125      */
126     var render = function(iConfig) {
127         $pickeruser_add_button.attr("disabled", "disabled");
128         clearAutoSuggest();
129         // Merge user defined config with defaults
130         for (var element in iConfig) {
131             if (iConfig.hasOwnProperty(element) && pickerData.hasOwnProperty(element)) {
132                 pickerData[element] = iConfig[element];
133             }
134         }
135 
136         // bind elements, replace some text
137         $pickeruser_adding_titles.hide();
138         if (pickerData.type === 'content') {
139             $pickeruser_instruction.html($pickeruser_content_text.html());
140             $pickeruser_send_message.hide();
141             $pickeruser_container_search.addClass("no_message");
142             $pickeruser_adding_files.show();
143         } else {
144             $pickeruser_instruction.html($pickeruser_people_text.html());
145             $pickeruser_send_message.show();
146             $pickeruser_init_search.show();
147             $pickeruser_adding_people.show();
148             $pickeruser_container_search.removeClass("no_message");
149         }
150 
151         $pickeruser_add_header_what.html(pickerData.what);
152         $pickeruser_add_header_where.html(pickerData.where);
153         $pickeruser_search_query.focus();
154         $pickeruser_add_button.unbind("click");
155         $pickeruser_add_button.bind("click", function(){
156             addPeople(iConfig);
157             //reset form
158             reset();
159         });
160     };

169     var getSelectedList = function() {
170         var list = $("#as-values-" + tuid).val();
171         // this value is a comma-delimited list
172         // split it and get rid of any empty values in the array
173         list = list.split(",");
174         $(list).each(function(i, val) {
175            if (val === "") {
176                list.splice(i, 1);
177            }
178         });
179         return list;
180     };
181 
182     var addPeople = function(iConfig) {
183 
184       var userList = getSelectedList();
185 
186       // send the message if its not empty
187       var messageText = $.trim($pickeruser_message.val());
188       if (messageText !== "") {
189           var messageList = getSelectedList();
190           if ($pickeruser_copy_myself.is(':checked')) {
191             messageList.push(sakai.data.me.profile["rep:userId"]);
192           }
193           if (iConfig.URL){
194               messageText = messageText + "\n\n" + "<a href='" + iConfig.URL + "'>" + iConfig.URL + "</a>";
195           }
196           sakai.api.Communication.sendMessage(messageList, sakai.api.Security.saneHTML($("#pickeruser_subject_text").text())+ "," + iConfig.where, messageText);
197       }
198       $pickeruser_container.jqmHide();
199       $(window).trigger("sakai-pickeruser-finished", {"toAdd":userList});
200     };

202     var clearAutoSuggest = function() {

207     var setupAutoSuggest = function() {
208         $pickeruser_search_query.autoSuggest("",{
209             source: function(query, add) {
210                 var searchUrl = sakai.config.URL.SEARCH_USERS_GROUPS;
211                 if (pickerData.type === 'content') {
212                     searchUrl = sakai.config.URL.POOLED_CONTENT_MANAGER;
213                 }
214                 sakai.api.Server.loadJSON(searchUrl.replace(".json", ""), function(success, data){
215                     if (success) {
216                         var suggestions = [];
217                         var name, value, type;
218                         $.each(data.results, function(i) {
219                             if (pickerData.type === 'content') {
220                                 name = data.results[i]['sakai:pooled-content-file-name'];
221                                 value = data.results[i]['jcr:name'];
222                                 type = "file";
223                             } else if (data.results[i]["rep:userId"]) {
224                                 name = sakai.api.Security.saneHTML(sakai.api.User.getDisplayName(data.results[i]));
225                                 value = data.results[i]["rep:userId"];
226                                 type = "user";
227                             } else if (data.results[i]["sakai:group-id"]) {
228                                 name = data.results[i]["sakai:group-title"];
229                                 value = data.results[i]["sakai:group-id"];
230                                 type = "group";
231                             }
232                             if (pickerData.excludeList.length === 0 || $.inArray(value, pickerData.excludeList) === -1) {
233                                 suggestions.push({"value": value, "name": name, "type": type});
234                             }
235                         });
236                         add(suggestions);
237                     }
238                 }, {"q": "*" + query.replace(/\s+/g, "* OR *") + "*", "page": 0, "items": 15});
239             },
240             retrieveLimit: 10,
241             asHtmlID: tuid,
242             selectedItemProp: "name",
243             searchObjProps: "name",
244             formatList: function(data, elem) {
245                 // formats each line to be presented in autosuggest list
246                 // add the correct image, wrap name in a class
247                 var imgSrc = "/dev/images/user_avatar_icon_32x32.png";
248                 if(data.type === "group") {
249                     imgSrc = "/dev/images/group_avatar_icon_32x32.png";
250                 }
251                 var line_item = elem.html(
252                     '<img class="sm_suggestion_img" src="' + imgSrc + '" />' +
253                     '<span class="sm_suggestion_name">' + data.name + '</span>');
254                 return line_item;
255             },
256             resultClick: function(data) {
257                 $pickeruser_add_button.removeAttr("disabled");
258             },
259             selectionRemoved: function(elem) {
260                 elem.remove();
261                 if ($(".as-selection-item").length === 0) {
262                     $pickeruser_add_button.attr("disabled", "disabled");
263                 }
264             }
265         });
266     };
267     setupAutoSuggest();

269     var addChoicesFromPickeradvanced = function(data) {
270       $.each(data, function(i,val) {
271           var name = "";
272           var id = "";
273           if (val.entityType == "group") {
274               name = val["sakai:group-title"];
275               id = val["sakai:group-id"];
276           } else if (val.entityType == "user") {
277               name = sakai.api.User.getDisplayName(val);
278               id = val["rep:userId"];
279           } else if (val.entityType == "file") {
280               name = val["sakai:pooled-content-file-name"];
281               id = val["jcr:name"];
282           }
283           $pickeruser_search_query.autoSuggest.add_selected_item({"name": name, "value": id}, id);
284           $pickeruser_add_button.removeAttr("disabled");
285       });
286       $("input#" + tuid).val('').focus();
287     };

293     $(window).unbind("sakai-pickeruser-init");
294     $(window).bind("sakai-pickeruser-init", function(e, config, callbackFn) {
295 
296         // position dialog box at users scroll position
297         var htmlScrollPos = $("html").scrollTop();
298         var docScrollPos = $(document).scrollTop();
299         if (htmlScrollPos > 0) {
300             $pickeruser_container.css({"top": htmlScrollPos + 50 + "px"});
301         } else if (docScrollPos > 0) {
302             $pickeruser_container.css({"top": docScrollPos + 50 + "px"});
303         }
304 
305         $pickeruser_container.jqmShow();
306         render(config);
307         $(window).unbind("sakai-pickeradvanced-finished");
308         $(window).bind("sakai-pickeradvanced-finished", function(e, data) {
309             addChoicesFromPickeradvanced(data.toAdd);
310         });
311         callback = callbackFn;
312     });
