144     /**
145      * Convert to 2 numbers, we use this for converting hours and minutes
146      *     e.g.
147      *     2 -> 02
148      *     5 -> 05
149      *     23 -> 23
150      * @param {int} integer Number that needs to be converted
151      */
152     var extractToTwo = function(integer){
153         integer+= "";
154         try {
155             if(integer < 10){
156                 return "0" + integer;
157             } else {
158                 return integer;
159             }
160         }catch(err){
161             return integer;
162         }
163     };

165     /**
166      * Convert a date
167      * @param {String} date Date in format dd/mm/yyyy
168      * @param {String} hour Number of hours
169      * @param {String} minute Number of minutes
170      * @return {Date} the converted date
171      */
172     var convertDate = function(date, hour, minute){
173         var dArray = date.split("/");
174         var d = new Date();
175         d.setFullYear(dArray[2], dArray[1]-1, dArray[0]);
176         d.setHours(hour);
177         d.setMinutes(minute);
178         d.setSeconds(0);
179         return d;
180     };

194             json.poll.temp.polldateok = 1;
195             if(stopdate < today){
196                 json.poll.temp.polldateok = 0;

243     /**
244      * Process the poll information (options/votes)
245      * @param {Boolean} preview Is in preview mode or not
246      */
247     var processPoll = function(preview){
248         var totalvotes = 0;
249         var totalvotescount = 0;
250 
251         // Check if the poll is in preview mode or not
252         if(preview){
253 
254             // If the poll is in preview mode, we add some dummy data
255             json.poll.temp={};
256             json.poll.temp.options = [];
257             json.poll.temp.options = previewOptions;
258 
259             json.poll.temp.votes = [];
260             json.poll.temp.votes = previewVotes;
261 
262             totalvotes = json.poll.temp.votes.length;
263             totalvotescount = 0;
264             for(var i = 0; i < json.poll.temp.votes.length; i++){
265                 totalvotescount += parseInt(json.poll.temp.votes[i], 10);
266                 json.poll.temp.options[i].color = colors[i];
267             }
268 
269             json.poll.temp.processedVotes = [];
270             for(var j = 0; j < json.poll.temp.votes.length; j++){
271                 json.poll.temp.processedVotes[j] = {};
272                 json.poll.temp.processedVotes[j].percentage = Math.round(json.poll.temp.votes[j] / totalvotescount * 100);
273             }
274             json.poll.temp.question = previewQuestion;
275 
276         }else{
277 
278             // If the poll is not in preview mode, we add real data
279             totalvotes = json.poll.votes.length;
280             totalvotescount = 0;
281             for(var m = 0; m < json.poll.votes.length; m++){
282                 totalvotescount += parseInt(json.poll.votes[m], 10);
283             }
284             json.poll.processedVotes = [];
285 
286             if(totalvotescount === 0){
287                 // If the there are no votes, all the percentages should be 0
288                 for(var k = 0; k < json.poll.votes.length; k++){
289                     json.poll.processedVotes[k] = {};
290                     json.poll.processedVotes[k].percentage = 0;
291                 }
292             }else{
293                 // If there are votes, we calculate the percentages
294                 for(var l = 0; l < json.poll.votes.length; l++){
295                     json.poll.processedVotes[l] = {};
296                     json.poll.processedVotes[l].percentage = Math.round(json.poll.votes[l] / totalvotescount * 100);
297                 }
298             }
299         }
300         for(var n = 0, nl = json.poll.options.length; n < nl; n++){
301             if(typeof json.poll.options[n] === "string"){
302                 json.poll.options[n] = {
303                     "answer": json.poll.options[n]
304                 };
305             }
306         }
307     };

496     /**
497      * Register the vote of a user
498      */
499     var registerVote = function(){
500 
501         // Check if we can get the me object for the current user
502         if(me){
503 
504             // A user can only vote once
505             if($.inArray(me.user.userid, json.poll.users) === -1){
506 
507                 // Double check (it is already checked in the bindUnbindVote function)
508                 // if the user has checked an element. We do this double check to make it
509                 // more secure.
510                 if($('input[name=' + pollQuestionViewOptionsClass + ']:checked').length === 0){
511                     alert("Please select at least one option.");
512 
513                 // If the user selected one or multiple options
514                 }else{
515                     for(var i = 0; i < $('input[name=' + pollQuestionViewOptionsClass + ']:checked').length; i++){
516                         var pollIndex = parseInt($("#" + $('input[name=' + pollQuestionViewOptionsClass + ']:checked')[i].id).val(), 10);
517                         json.poll.votes[pollIndex] = parseInt(json.poll.votes[pollIndex], 10) + 1 + "";
518                     }
519 
520                     json.poll.users.push(me.user.userid);
521                     savePollToDatabase(finishVoting, false);
522                 }
523             }
524             else{
525                 alert("You already registered a vote.");
526             }
527         }else{
528             alert("Can not get the current user.");
529         }
530     };

675     /**
676      * Make and bind hours and minutes
677      * We do this by making a dynamic element that is added to the appropriate div
678      * @param {String} input Name of the elements that need to be created
679      * @param {Integer} count How many fields there need to be created
680      */
681     var makeBindDiv = function(input, count){
682         for(var i = 0; i < count; i++){
683 
684             // Create the dynamic element
685             var a = document.createElement('a');
686 
687             // Set the class and id
688             a.setAttribute('class', input);
689             a.setAttribute('id', input+i);
