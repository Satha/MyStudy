353     /**
354      * prepares the member objects..
355      */
356     var prepareSiteMembers = function(members){
357 
358         // Make sure we have properly formed data.
359         for (var i = 0, j = members.length; i < j; i++) {
360             var user = members[i];
361             // profile information
362             if (typeof user.firstName === "object") {
363                 user.firstName = user.firstName[0];
364             }
365             if (typeof user.lastName === "object") {
366                 user.lastName = user.lastName[0];
367             }
368             if (typeof user.email === "object") {
369                 user.email = user.email[0];
370             }
371             if (typeof user.picture === "object") {
372                 user.picture = user.picture[0];
373             } else if (user.picture && $.parseJSON(user.picture).name) {
374                 user.picture = "/~" + user["rep:userId"] + "/public/profile/" + $.parseJSON(user.picture).name;
375             }
376             if (typeof user["rep:userId"] === "object") {
377                 user.userid = user["rep:userId"][0];
378             }
379 
380             // Check if we are this user.
381             user.isMe = false;
382             if (user["rep:userId"] === me.user.userid) {
383                 user.isMe = true;
384             }
385             // Check to see if this user is a contact of us.
386             user.isContact = checkIfContact(user);
387 
388             // The fullname for this user. (User for filtering)
389             user.fullname = user.firstName + " " + user.lastName;
390 
391             user.basic = user.basic ? $.parseJSON(user.basic) : "";
392             user.aboutme = user.aboutme ? $.parseJSON(user.aboutme) : "";
393             user.education = user.education ? $.parseJSON(user.education) : "";
394             user.talks = user.talks ? $.parseJSON(user.talks) : "";
395             user.academic = user.academic ? $.parseJSON(user.academic) : "";
396             user.job = user.job ? $.parseJSON(user.job) : "";
397 
398 
399             // the information the admin wants displayed.
400             user.displayedData = [];
401             for (var di = 0, dj = widgetSettings.data.length; di < dj; di++) {
402                 var toShow = widgetSettings.data[di];
403                 var o = {};
404                 o.key = displaydata[toShow];
405 
406                 if (toShow === "aboutme") {
407                     o.value = displayAboutMe(user, "aboutme", sitemembers_normal_data_aboutme);
408                 }
409                 else
410                     if (toShow === "hobbies") {
411                         o.value = displayAboutMe(user, "hobbies", sitemembers_normal_data_hobbies);
412                     }
413                     else
414                         if (toShow === "academicinterests") {
415                             o.value = displayAboutMe(user, "academicinterests", sitemembers_normal_data_academicinterests);
416                         }
417                         else
418                             if (toShow === "personalinterests") {
419                                 o.value = displayAboutMe(user, "personalinterests", sitemembers_normal_data_personalinterests);
420                             }
421                             else
422                                 if (toShow === "degrees") {
423                                     o.value = displayDegrees(user);
424                                 }
425                                 else
426                                     if (toShow === "talks") {
427                                         o.value = displayTalks(user);
428                                     }
429                                     else
430                                         if (toShow === "publications") {
431                                             o.value = displayPublications(user);
432                                         }
433                                         else
434                                             if (toShow === "profexperience") {
435                                                 o.value = displayProfExperience(user);
436                                             }
437 
438                 user.displayedData.push(o);
439             }
440 
441             if (user.isContact) {
442                 myContactsOnSite.push(user);
443             }
444 
445             members[i] = user;
446         }
447 
448         return members;
449     };

464     /**
465      * Does a request to the site.members.json servlet and retrieves all the members for this site.
466      */
467     var getSiteMembers = function(){
468         // Show a prelaoder.
469         loader(true);
470         // Get a list of all the members.
471         var url = sakai.config.URL.SITE_GET_MEMBERS_SERVICE.replace("__SITE__", siteid) + "?items=" + nrOfItems + "&sort=" + sortOn + "," + sortOrder + "&start=" + startPos;
472         $.ajax({
473             url: url,
474             success: function(data){
475                 var json = data.results;
476 
477                 // If we get an emty list, then we assume we have received all the members.
478                 if (json.length === 0) {
479                     gotAllMembers = true;
480                     displayMore(false);
481                 }
482                 else {
483                     displayMore(true);
484                     // Set the start param for next get.
485                     startPos += json.length;
486                     $(sitemembers_normal_nrFetched, rootel).text(startPos);
487 
488                     if (totalMembers <= startPos) {
489                         displayMore(false);
490                     }
491 
492                     // Make sure the formatting is correct.
493                     var members = prepareSiteMembers(json);
494                     // Add the new members to the list.
495                     allMembers = allMembers.concat(members);
496                     // Render it to the page.
497                     renderAndAppendSiteMembers(members);
498                 }
499                 loader(false);
500             },
501             error: function(xhr, textStatus, thrownError) {
502                 if (xhr.status === 500) {
503                     gotAllMembers = true;
504                     displayMore(false);
505                     loader(false);
506                 }
507             }
508         });
509     };

566     /**
567      * Retrieves the settings object from JCR.
568      */
569     var getSiteMembersSettingsFromJCR = function(){
570         sakai.api.Widgets.loadWidgetData(tuid, function(success, data){
571             if (success) {
572                 widgetSettings = data;
573                 widgetSettings.data = $.parseJSON(widgetSettings.data);
574                 if (showSettings) {
575                     displaySettings(widgetSettings);
576                 }
577                 else {
578                     getTotalAmountOfMembers();
579                     doPageView();
580                 }
581             }
582             else {
583                 widgetSettings.data = [];
584                 widgetSettings.display = "compact";
585                 widgetSettings.sort = "lastname";
586                 getTotalAmountOfMembers();
587                 doPageView();
588             }
589         });

596     var createSiteMembersSettings = function(){

598         settings.sort = "lastname";

603         $(sitemembers_settings_dataClass + " input:checked").each(function(){

605         });
606         // Get the display mode.
607         if ($(sitemembers_settings_display_wide + ":checked", rootel).length === 1) {
608             settings.display = "wide";
609         }
610 
611         return settings;
