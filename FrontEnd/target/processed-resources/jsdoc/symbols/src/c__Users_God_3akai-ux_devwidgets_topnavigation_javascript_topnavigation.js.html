 91     /**
 92      * Get the number of messages that are unread and show it.
 93      */
 94     var getCountUnreadMessages = function(){
 95         //we have the number of unread messages as a part of the me-feed
 96         //so get it directly from me object.
 97         $(chatUnreadMessages).text(sakai.data.me.messages.unread);

100     /**
101      * Update the status on the page by firing an event that handles this
102      */
103     var updateChatStatus = function(){
104         // Trigger the chat_status_change event to update other widgets
105         $(window).trigger("chat_status_change", currentChatStatus);
106     };

108     /**
109      * Set the chatstatus of the user
110      * @param {String} chatstatus The chatstatus which should be
111      * online/offline or busy
112      */
113     var sendChatStatus = function(chatstatus){
114         if (currentChatStatus !== chatstatus){
115             currentChatStatus = chatstatus;
116 
117             var data = {
118                 "chatstatus": chatstatus,
119                 "_charset_": "utf-8"
120             };
121 
122             $.ajax({
123                 url: "/~" + sakai.data.me.profile["rep:userId"] + "/public/authprofile",
124                 type: "POST",
125                 data: data,
126                 success: function(data){
127                     updateChatStatus();
128                 },
129                 error: function(xhr, textStatus, thrownError){
130                     alert("An error occurend when sending the status to the server.");
131                 }
132             });
133         }
134     };

136     /**
137      * Show or hide the user link menu
138      * @param {Boolean} hideOnly
139      *  true: Hide the menu only
140      *  false: Show or hide the menu depending if it's already visible
141      */
142     var showHideUserLinkMenu = function(hideOnly){
143         if ($(userLinkMenu).is(":visible") || hideOnly) {
144             $(userLinkMenu).hide();
145         }
146         else {
147             $(userLinkMenu).css("left", Math.round($(userLink).offset().left) + "px");
148             $(userLinkMenu).css("top", (Math.round($(userLink).offset().top) + $(userLink).height() + 2) + "px");
149             $(userLinkMenu).css("width", ($(userLink).width() + 10) + "px");
150             $(userLinkMenu).show();
151         }
152     };
153 
154     /**
155      * Add binding to some elements
156      */
157     var addBinding = function(){
158         $(userLink).bind("click", function(){
159             showHideUserLinkMenu(false);
160         });
161 
162         $(userLinkChatStatusClass).bind("click", function(ev){
163             showHideUserLinkMenu(false);
164             var clicked = ev.currentTarget.id.split("_")[ev.currentTarget.id.split("_").length - 1];
165             sendChatStatus(clicked);
166         });
167 
168         $.each($(topNavigationBar + " a"), function(){
169             if (window.location.pathname === $(this).attr("href")){
170                 $(this).attr("href", "#");
171             }
172         });
173 
174         $(document).bind("click", function(e){
175             var $clicked = $(e.target);
176 
177             // Check if one of the parents is the userLink
178             if (!$clicked.parents().is(userLink)) {
179                 showHideUserLinkMenu(true);
180             }
181         });
182 
183         $(window).bind("chat_status_change", function(event, chatstatus){
184             currentChatStatus = chatstatus;
185             sakai.api.Util.updateChatStatusElement($(userLink), chatstatus);
186         });
187 
188         $(window).bind("click", function(e){
189             // if menu is visible and the target element clicked is not menu hide dropdown
190             if ($(userLinkMenu).is(":visible") && !$(e.target).parents().is(userLink)){
191                 showHideUserLinkMenu(true);
192             }
193         });
194     };

200     /**
201      * Select the page in the top navigation where you are currently on.
202      * This will apply a class to the selected navigation item based on the current URL
203      * @returns void;
204      */
205     var determineCurrentNav = function(){
206         var windowLocationPath = window.location.pathname.toLowerCase();
207 
208         // Remove all selected classes from nav elements
209         $(nav + " " + navSelectedNavItemClass).removeClass(navSelectedNavItemClass);
210 
211         // My Sakai
212         if ((windowLocationPath.indexOf(sakai.config.URL.MY_DASHBOARD_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.PUBLIC_MY_DASHBOARD_URL) !== -1)) {
213             $(navMySakaiLink).addClass(navSelectedNavItemClass);
214             return;
215         }
216 
217         // Content & Media
218         if ((windowLocationPath.indexOf(sakai.config.URL.CONTENT_MEDIA_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.PUBLIC_CONTENT_MEDIA_URL) !== -1)) {
219             $(navContentMediaLink).addClass(navSelectedNavItemClass);
220             return;
221         }
222 
223         // People
224         if ((windowLocationPath.indexOf(sakai.config.URL.PEOPLE_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.PUBLIC_PEOPLE_URL) !== -1)) {
225             $(navPeopleLink).addClass(navSelectedNavItemClass);
226             return;
227         }
228 
229         // Courses & Sites
230         if ((windowLocationPath.indexOf(sakai.config.URL.COURSES_SITES_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.PUBLIC_COURSES_SITES_URL) !== -1) || (windowLocationPath.indexOf("/sites/") !== -1)) {
231             $(navCoursesSitesLink).addClass(navSelectedNavItemClass);
232             return;
233         }
234 
235         // Calendar
236         if ((windowLocationPath.indexOf(sakai.config.URL.SEARCH_GENERAL_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.SEARCH_PEOPLE_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.SEARCH_SITES_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.SEARCH_CONTENT_URL) !== -1) || (windowLocationPath.indexOf(sakai.config.URL.PUBLIC_SEARCH_URL) !== -1)) {
237             $(navCalendarLink).addClass(navSelectedNavItemClass);
238             return;
239         }
240 
241     };

248     /**
249      * Execute the search for the value that is in the search input field
250      */
251     var doSearch = function(){
252         var tosearch = $general_search_input.val();
253         // Disable search button
254         $(generalSearchSubmitButton).attr("disabled", true);
255         // Check whether the search term is different than the default text. If not,
256         // we do a search for *
257         if (!tosearch || tosearch === $general_search_default_value.text()){
258             tosearch = "*";
259         }
260 
261         // Only enable button if the location is the search page
262         if (window.location.pathname.split("/")[2] === "search.html") {
263             $(generalSearchSubmitButton).attr("disabled", false);
264             // if user is on the search page use the history event to perform the search
265             History.addBEvent("1", encodeURIComponent(tosearch));
266         } else {
267             // Redirecting back to the general search page. This expects the URL to be
268             // in a format like this one: page.html#pageid|searchstring
269             document.location = sakai.config.URL.SEARCH_GENERAL_URL + "#q=" + tosearch;
270         }
271     };

273     /**
274      * If this is the first time the field gets focus, we'll make his text color black
275      * and remove the default value
276      */
277     $general_search_input.bind("focus", function(ev){
278         if (!searchFocus) {
279             $general_search_input.val("").addClass(searchInputFocusClass);
280             searchFocus = true;
281         }
282     });
283 
284     /**
285      * If we leave the field without filling out anything, we'll reinsert the default
286      * search inputbox value
287      */
288     $general_search_input.bind("blur", function(ev){
289 
290         if (!$general_search_input.val()) {
291             $general_search_input.removeClass(searchInputFocusClass);
292             $general_search_input.val($general_search_default_value.text());
293             searchFocus = false;
294         }
295     });

298      * Bind the submit event to the search form
299      * This event is triggered when you hit enter in the input field and
300      * when you click on the submit button
301      */
302     $general_search_form.bind("submit", function(){
303 
304         doSearch();
305         return false;
306     });

308     /**
309      * Switch navigation bar to anonymous mode
310      * @returns void
311      */
312     var switchToAnonymousMode = function(){
313 
314         // Show Nav Container
315         $(exploreNavigationContainer).show();
316 
317         // Hide things which are irrelvant for Anonymous user
318         $(".personal .mail").hide();
319         $(".personal .sign_out").hide();
320         $(".help").hide();
321         $("#user_link_container").hide();
322 
323         // Hide search bar
324         $general_search_container.hide();
325 
326         // Show anonymous elements
327         $("#other_logins_button_container").show();
328         $(".log_in").addClass("help_none");
329 
330         // if config.js is set to external, register link is hidden
331         if(!sakai.config.Authentication.internal) {
332             $("#register_button_container").hide();
333         }
334         else {
335             $("#register_button_container").show();
336         }
337         $("#login_button_container").show();
338 
339         // Set up public nav links
340         $("#nav_my_sakai_link a").attr("href", sakai.config.URL.PUBLIC_MY_DASHBOARD_URL);
341         $("#nav_content_media_link a").attr("href", sakai.config.URL.PUBLIC_CONTENT_MEDIA_URL_PAGE);
342         $("#nav_people_link a").attr("href", sakai.config.URL.PUBLIC_PEOPLE_URL);
343         $("#nav_courses_sites_link a").attr("href", sakai.config.URL.PUBLIC_COURSES_SITES_URL);
344         $("#nav_search_link a").attr("href", sakai.config.URL.PUBLIC_SEARCH_URL_PAGE);
345 
346         // Make the login page redirect to the current page after login
347         $(".log_in").attr("href", $(".log_in").attr("href") + "?url=" + escape(window.location.pathname + window.location.search + window.location.hash));
348

351     /**
352      * Set the presence for the current user
353      * We need to do this in order to let the server know the user is still available for chatting.
354      */
355     var setPresence = function(){
356 
357         var data = {
358             "sakai:status": "online",
359             "_charset_": "utf-8"
360         };
361 
362         $.ajax({
363             url: sakai.config.URL.PRESENCE_SERVICE,
364             type: "POST",
365             success: function(){
366                 setTimeout(setPresence, 120000);
367             },
368             data: data
369         });
370     };

372     /**
373      * Parse the chatstatus for a user
374      * @param {String} chatStatus The chatstatus which should be
375      * online, busy or offline
376      */
377     var parseChatStatus = function(chatStatus){
378         // Check if the status is defined
379         if (!chatStatus) {
380             chatStatus = "online";
381         }
382         return chatStatus;
383     };
384 
385     /**
386      * Get the chat status for the current user
387      */
388     var getChatStatus = function(){
389         if (sakai.data.me.profile) {
390             currentChatStatus = parseChatStatus(sakai.data.me.profile.chatstatus);
391         }
392         else {
393             currentChatStatus = "online";
394         }
395         updateChatStatus();
396     };

402     /**
403      * Contains all the functions and methods that need to be
404      * executed on the initial load of the page
405      */
406     var doInit = function(){
407 
408         $(navMyProfile).attr("href", "/~" + sakai.data.me.user.userid);
409 
410         var obj = {};
411         var menulinks = [];
412 
413         for (var i in sakai.config.Navigation) {
414 
415             // We need to add the hasOwnProperty to pass to JSLint and it is also a security issue
416             if (sakai.config.Navigation.hasOwnProperty(i)) {
417 
418                 var temp = {};
419                 temp.url = sakai.config.Navigation[i].url;
420                 temp.label = sakai.api.i18n.General.getValueForKey(sakai.config.Navigation[i].label);
421                 temp.cleanurl = temp.url || "";
422                 if (temp.cleanurl) {
423                     if (temp.cleanurl.indexOf('?') && temp.cleanurl.indexOf('?') > 0) {
424                         temp.cleanurl = temp.cleanurl.substring(0, temp.cleanurl.indexOf('?'));
425                     }
426                     if (temp.cleanurl.indexOf('#') && temp.cleanurl.indexOf('#') > 0) {
427                         temp.cleanurl = temp.cleanurl.substring(0, temp.cleanurl.indexOf('#'));
428                     }
429                 }
430                 if (i === 0) {
431                     temp.firstlink = true;
432                 }
433                 else {
434                     temp.firstlink = false;
435                 }
436                 menulinks.push(temp);
437             }
438         }
439         obj.links = menulinks;
440         // Get navigation and render menu template
441         $(".explore").html($.TemplateRenderer("navigation_template", obj));
442 
443         var person = sakai.data.me;
444 
445         $(exploreNavigationContainer).show();
446 
447         // Fill in the name of the user in the different fields
448         if (sakai.api.User.getDisplayName(person.profile) !== "") {
449             $(userIdLabel).text(sakai.api.Util.shortenString(sakai.api.User.getDisplayName(person.profile), 25));
450         }
451 
452         // Show the profile picture on the dashboard page
453         /** TODO : Remove the lines beneath if this functionality is inside changepic.js */
454         if (person.profile.picture) {
455             var picture = $.parseJSON(person.profile.picture);
456             if (picture.name) {
457                 $(pictureHolder).attr("src", "/~" + sakai.data.me.user.userid + "/public/" + picture.name);
458             }
459         }
460 
461         // Highlight current nav item
462         determineCurrentNav();
463 
464         // Set presence and bind things
465         addBinding();
466         getCountUnreadMessages();
467         setPresence();
468 
469         // Get chat status
470         getChatStatus();
471     };
472 
473     if (sakai.data.me.user.anon) {
474         // If a user is not logged in -> switch to anonymous mode
475         switchToAnonymousMode();
476     }
477     else {
478         doInit();
479     }
