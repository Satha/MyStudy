 73     /**
 74      * Change the mode of the current profile
 75      * @param {String} mode The mode for the profile (view | edit)
 76      */
 77     var setProfileMode = function(mode){
 78 
 79         // Check the mode parameter
 80         if ($.inArray(mode, sakai.profile.main.mode.options) !== -1) {
 81 
 82             // Set the correct profile mode
 83             sakai.profile.main.mode.value = mode;
 84 
 85         }
 86         else {
 87 
 88             // Use the standard profile mode
 89             sakai.profile.main.mode.value = sakai.profile.main.mode.options[0];
 90 
 91             // Print a log message that the supplied mode isn't valid
 92             debug.info("profile - setProfileMode - the supplied mode '" + mode + "' is not a valid profile mode. Using the default mode instead");
 93 
 94         }
 95 
 96     };

 98     /**
 99      * Get the profile mode from the querystring
100      */
101     var getProfileMode = function(){
102 
103         if (querystring.contains("mode")) {
104             return querystring.get("mode");
105         }
106         return false;
107 
108     };

110     /**
111      * Change the profile mode
112      * This will fire a redirect
113      * @param {String} mode The mode you want to change to
114      */
115     var changeProfileMode = function(mode){
116 
117          // Check the mode parameter
118         if ($.inArray(mode, sakai.profile.main.mode.options) !== -1) {
119 
120             // Perform the redirect
121             //window.location = window.location.pathname + "?mode=" + mode; // TODO fix this, jquery.bbq it, and do not force a refresh
122 
123             switch (mode) {
124                 case "edit":
125                     window.location = sakai.config.URL.PROFILE_EDIT_URL + "?user=" + sakai.profile.main.currentuser;
126                     break;
127                 case "view":
128                     window.location = "/~" + sakai.profile.main.currentuser;
129                     break;
130             }
131         }
132 
133     };

136     /**
137      * Check whether the user is editing/looking at it's own profile or not
138      * We do this because if it is the current user, we don't need to perform an extra search
139      */
140     var setIsMe = function(){
141 
142         // Check whether there is a user parameter in the querystring,
143         // if so, check whether the userid is not the same as the user parameter
144         if (querystring.contains("user") && querystring.get("user") !== sakai.data.me.user.userid) {
145             sakai.profile.main.isme = false;
146             sakai.profile.main.currentuser = querystring.get("user");
147         }
148         else {
149             sakai.profile.main.isme = true;
150             sakai.profile.main.currentuser = sakai.data.me.user.userid;
151         }
152 
153     };

245     /**
246      * Check whether there is a valid picture for the user
247      * @param {Object} profile The profile object that could contain the profile picture
248      * @return {String}
249      * The complete URL of the profile picture
250      * Will be an empty string if there is no picture
251      */
252     var constructProfilePicture = function(profile){
253 
254         if (profile.basic.elements.picture && profile.basic.elements.picture.value && profile["rep:userId"]) {
255             return "/~" + profile["rep:userId"] + "/public/profile/" + profile.basic.elements.picture.value.name;

263     /**
264      * Filter some JCR properties, we need to do this because some properties
265      * can not be used by the import operation in Slin
266      * @param {Object} i_object The object you want to filter
267      */
268     var filterJCRProperties = function(i_object){
269 
270         // Remove the "rep:policy" property
271         if (i_object["rep:policy"]) {
272             delete i_object["rep:policy"];
273         }
274 
275         // Also run over the other objects within this object
276         for (var i in i_object) {
277             if (i_object.hasOwnProperty(i) && $.isPlainObject(i_object[i])) {
278               filterJCRProperties(i_object[i]);
279             }
280         }
281 
282     };

285     /**
286      * Save the Access control list for the profile
287      */
288     var saveProfileACL = function(){
289 
290         var requests = []; // Variable used to contain all the information we need to send to the batch post
291 
292         // Remove the ACL's on the authprofile URL
293         requests[requests.length] = {
294             // Construct the right URL
295             "url": authprofileURL + ".deleteAce.html",
296             "method": "POST",
297             "parameters": {
298                 ":applyTo": [sakai.profile.main.currentuser, "anonymous", "everyone"]
299             }
300         };
301 
302         // Run over all the elements in the config file
303         for (var i in sakai.profile.main.config) {
304             if (sakai.profile.main.config.hasOwnProperty(i) && $.isPlainObject(sakai.profile.main.config[i])) {
305 
306                 // Create a sectionobject for caching purposes
307                 var sectionObject = sakai.profile.main.data[i];
308 
309                 // Check whether it is also in the data object
310                 if (sakai.profile.main.data[i] && $.isPlainObject(sakai.profile.main.data[i])) {
311 
312                     // Array containing the postparams for the specific access property
313                     var aclArray = sakai.profile.main.acls.options[sectionObject.access].postparams;
314 
315                     // Run over all the elements in the array
316                     for (var j = 0, jl = aclArray.length; j < jl; j++) {
317 
318                         // Add the object to the requests array
319                         requests[requests.length] = {
320                             // Construct the right URL
321                             "url": authprofileURL + "/" + i + ".modifyACE.json", // Todo change to JSON
322                             "method": "POST",
323                             "parameters": aclArray[j]
324                         };
325 
326                     }
327 
328                 }
329 
330             }
331         }
332 
333         /*for(var k = 0, kl = requests.length; k < kl; k++){
334 
335             $.ajax({
336                 url: requests[k].url,
337                 traditional: true,
338                 type: requests[k].method,
339                 data: requests[k].parameters,
340                 async: false,
341                 success: function(){
342 
343                     if(k === requests.length-1){
344                         alert("ok");
345                     }
346 
347                 }
348             });
349 
350         }*/
351 
352 
353         // Send the Ajax request to the batch servlet
354         // depends on KERN-909
355         $.ajax({
356             url: sakai.config.URL.BATCH,
357             traditional: true,
358             type: "POST",
359             data: {
360                 requests: $.toJSON(requests)
361             },
362             success: function(data){
363 
364                 // Show a successful notification to the user
365                 sakai.api.Util.notification.show("", $profile_message_form_successful.text() , sakai.api.Util.notification.type.INFORMATION);
366 
367                 // Wait for 2 seconds
368                 setTimeout(
369                     function(){
370                         // Change the profile mode if the save was successful
371                         changeProfileMode("view");
372                     }, 2000);
373 
374             },
375             error: function(xhr, textStatus, thrownError){
376 
377                 // Show an error message to the user
378                 sakai.api.Util.notification.show("", $profile_error_form_error_server.text() , sakai.api.Util.notification.type.ERROR);
379 
380                 // Log an error message
381                 debug.error("sakai.profile - saveProfileACL - the profile ACL's couldn't be saved successfully");
382 
383             }
384         });
385 
386     };

388     /**
389      * Save the current profile data to the repository
390      */
391     var saveProfileData = function(){
392 
393         // Trigger the profile save method, this is event is bound in every sakai section
394         $(window).trigger("sakai-profile-save");
395 
396     };
397 
398     $(window).bind("sakai-profile-data-ready", function(e, sectionName) {
399 
400         // keep track of all the sections that are ready
401         if ($.inArray(sectionName, readySections) < 0) {
402             readySections.push(sectionName);
403         }
404 
405         // if all sections are ready, we'll pass over this loop. otherwise, return and wait
406         for (var i in sakai.profile.main.config) {
407             if (sakai.profile.main.config.hasOwnProperty(i)) {
408                 if ($.inArray(i, readySections) < 0) {
409                     return;
410                 }
411             }
412         }
413 
414         // Filter some JCR properties
415         filterJCRProperties(sakai.profile.main.data);
416 
417         // Save the profile properties
418         sakai.api.Server.saveJSON(authprofileURL, sakai.profile.main.data, function(success, data){
419 
420             // Check whether is was successful
421             if (success) {
422 
423                 // Save the profile acl
424                 saveProfileACL();
425 
426             }
427             else {
428 
429                 // Show an error message to the user
430                 sakai.api.Util.notification.show("", $profile_error_form_error_server.text() , sakai.api.Util.notification.type.ERROR);
431 
432                 // Log an error message
433                 debug.error("sakai.profile - saveProfileData - the profile data couldn't be saved successfully");
434 
435             }
436 
437         });
438     });

534     /**
535      * Insert a profile section widget
536      * @param {String} sectionname The name of the section e.g. basic/talks/aboutme
537      */
538     var insertProfileSectionWidget = function(sectionname) {
539 
540         // Create a JSON object to pass the sectionname along
541         // Trimpath needs an object to be passed (not only a variable)
542         var sectionobject = {
543             "sectionname": "profilesection-" + sectionname
544         };
545 
546         // Construct the html for the widget
547         var toAppend = $.TemplateRenderer($profile_sectionwidgets_container_template, sectionobject);
548         $profile_sectionwidgets_container.append(toAppend);
549 
550         // Bind a global event that can be triggered by the profilesection widgets
551         $(window).bind("sakai-" + sectionobject.sectionname, function(eventtype, callback){
552 
553             if ($.isFunction(callback)) {
554                 callback(sectionname);
555             }
556 
557         });
558 
559     };

572             }
573         }

611     var doInit = function(){
612 
613         // Set the querystring object variable
614         // We use the following parameters:
615         //    mode -> mode of the profile
616         //    user -> the id of the user for which you want to see the profile
617         querystring = new Querystring();
618 
619         // Get and set the profile mode
620         var profilemode = getProfileMode();
621         if (profilemode) {
622             setProfileMode(profilemode);
623         }
624 
625         // Check if you are looking at the logged-in user
626         setIsMe();
627 
628         // Construct the authprofile URL
629         authprofileURL = "/~" + sakai.profile.main.currentuser + "/public/authprofile";
630 
631         // Construct the ACL list
632         constructACL();
633 
634         // Check user profile type
635         checkProfileType();
636 
637         renderTemplates();
638 
639         insertProfileSectionWidgets();
640 
641         addBinding();
642 
643 
644     };
